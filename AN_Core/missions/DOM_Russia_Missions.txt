mos_rus_handle_succession = {
	slot = 1
	generic = no
	ai = yes
	potential_on_load = {
		has_dlc = "Domination"
	}
	potential = {
		OR = {
			tag = MOS
			AND = {
				was_tag = MOS
				tag = RUS
			}
		}
		has_dlc = "Domination"
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	mos_the_civil_war = {
		icon = mission_dethrone_the_rurikovichs
		required_missions = {  }
		position = 1
		provinces_to_highlight = {
			owned_by = ROOT
			controlled_by = REB
		}
		trigger = {
			OR = {
				custom_trigger_tooltip = {
					tooltip = mos_the_civil_war_at
					NOT = { has_ruler_flag = has_vasily_ii }
				}
				custom_trigger_tooltip = {
					tooltip = mos_the_civil_war_bt
					has_country_flag = mos_civil_war_ended_flag
				}
			}
			if = {
				limit = { government = monarchy }
				legitimacy = 80
			}
			NOT = { num_of_rebel_armies = 1 }
			NOT = { num_of_rebel_controlled_provinces = 1 }
		}
		effect = {
			if = {
				limit = {
					dynasty = "Rurikovich"
				}
				add_country_modifier = {
					name = mos_legitimacy_of_the_rurikovich
					duration = 9125
				}
			}
			else = {
				add_country_modifier = {
					name = mos_a_new_russian_authority
					duration = 9125
				}
			}
		}

		ai_weight = {
			factor = 250
		}
	}
	
	mos_secure_muscovys_future = {
		icon = mission_sovereign_principality
