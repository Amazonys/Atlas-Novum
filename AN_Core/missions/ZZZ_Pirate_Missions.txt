	}

	pirate_trade_practices = {

		required_missions = { }

		effect = {
			add_country_modifier = {
				name = "pirate_trade_practices"

			}
		}
	}
	pirate_trade_ports = {
		icon = mission_dominate_home_trade_node
		required_missions = { pirate_trade_practices }
		
		provinces_to_highlight = {
			province_has_center_of_trade_of_level = 1
			NOT = { province_has_center_of_trade_of_level = 2 }
			owned_by = ROOT
		}
		
		trigger = {
			if = {
				limit = {
					has_dlc = "Dharma"
				}
				num_of_owned_provinces_with = {
					province_has_center_of_trade_of_level = 2
					value = 3
				}
			}
			else = {
				num_of_owned_provinces_with = {
					OR = {
						has_building = marketplace
						has_building = trade_depot
						has_building = stock_exchange
					}
					value = 5
				}
			}
		}
		
		effect = {
			add_country_modifier = {
				name = "pirate_trade_ports"
				duration = 7300
			}
		}
	}
	pirate_trade_in_gold = {

		}
		
		effect = {

			}
		}
	}
	pirate_triangle_trade = {

	pirate_an_old_fort = {
		icon = mission_conquer_50_development
		required_missions = { pirate_a_second_madagascar }
		
		provinces_to_highlight = {
			owned_by = ROOT
			is_capital = yes
			NOT = { fort_level = 3 }
		}

			}
		}
		
		effect = {

		icon = mission_unite_home_region
		required_missions = { pirate_an_old_fort }
		
		provinces_to_highlight = {

				area = cuba_area
				area = east_cuba_area
			}

			has_discovered = yes
		}
		
		trigger = {
			cuba_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
			east_cuba_area = {
				type = all

			carribeans_region = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = {  is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}

	pirate_pirates_of_the_caribbean = {
		icon = mission_establish_high_seas_navy
		required_missions = { pirate_running_down_to_cuba }
		
		provinces_to_highlight = {
			region = carribeans_region
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
			has_discovered = yes
		}

		trigger = {
			carribeans_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}

		effect = {
			custom_tooltip = all_gold_provinces_around_the_caribbean
			hidden_effect = {
				central_america_region = {

					}
					add_permanent_claim = ROOT
				}
				rio_grande_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
						trade_goods = gold
					}
					add_permanent_claim = ROOT
				}
				mississippi_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
						trade_goods = gold
					}
					add_permanent_claim = ROOT
				}
				southeast_america_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }

					}
					add_permanent_claim = ROOT
