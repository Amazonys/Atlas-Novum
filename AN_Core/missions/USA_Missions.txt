usa_missions_main = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = USA
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	usa_independence = {
		icon = mission_assemble_an_army
		required_missions = { }
		trigger = {
			is_subject = no
			is_at_war = no
		}
		effect = {
			change_government_reform_progress = 100
			if = {
				limit = { NOT = { legitimacy_equivalent = 90 } }
				add_legitimacy = 20
				add_horde_unity = 20
				add_devotion = 20
				add_republican_tradition = 10
			}
			else = { add_prestige = 15 }
			if = { #For bookmark start
				limit = {
					northeast_america_region = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
				}
				northeast_america_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
	defend_the_american_colonies = {
		icon = mission_cannons_firing
		required_missions = { usa_independence }
		provinces_to_highlight = {
			region = northeast_america_region
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
			has_discovered = ROOT
		}
		trigger = {
			northeast_america_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			add_army_tradition = 50
			add_stability = 2
			if = { #For bookmark start
				limit = {
					southeast_america_region = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
				}
				southeast_america_region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
		}
	}
	conquer_florida = {
		icon = mission_unite_home_region
		required_missions = { defend_the_american_colonies }
		provinces_to_highlight = {
			region = southeast_america_region
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
			has_discovered = ROOT
		}
		trigger = {
			southeast_america_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		effect = {
			add_country_modifier = {
				name = "military_victory"
				duration = 7300
			}
		}
	}
	reach_the_mississippi = {
		icon = mission_settlers_north_america
		required_missions = { conquer_florida }
		provinces_to_highlight = {
			region = mississippi_region
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
			has_discovered = ROOT
		}
		trigger = {
			num_of_owned_provinces_with = {
				region = mississippi_region
				value = 10
			}
			OR = {
				has_idea_group = exploration_ideas
				has_idea_group = expansion_ideas
			}
		}
		effect = {
			add_country_modifier = {
				name = "colonial_enthusiasm"
				duration = 7300
			}
		}
	}
