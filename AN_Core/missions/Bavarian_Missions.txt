bavarian_missions_main = {
	slot = 1
	generic = no
	ai = yes
	potential = {
		OR = {
			tag = BAV
			tag = ING
			tag = LBV
			tag = UBV
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	bav_reunite_bavaria = {
		icon = mission_conquer_50_development
		required_missions = { }
		completed_by = 1507.1.1
		ai_weight = {
			factor = 10
		}
		ai_priority = {
			factor = 10
		}

		provinces_to_highlight = {
			OR = {
				#other claimants are gone
				AND = {
					ROOT = { NOT = { tag = LBV } }
					LBV = {
						OR = {
							num_of_cities = 5
							NOT = { is_subject_of = ROOT }
						}
					}
					owned_by = LBV
				}
				AND = {
					ROOT = { NOT = { tag = ING } }
					ING = {
						OR = {
							num_of_cities = 5
							NOT = { is_subject_of = ROOT }
						}
					}
					owned_by = ING
				}
				AND = {
					ROOT = { NOT = { tag = UBV } }
					UBV = {
						OR = {
							num_of_cities = 5
							NOT = { is_subject_of = ROOT }
						}
					}
					owned_by = UBV
				}
				#make sure you are not benefitting from others' conquests
				AND = {
					province_id = 65
					NOT = { country_or_non_sovereign_subject_holds = ROOT }
				}
				AND = {
					province_id = 64
					NOT = { country_or_non_sovereign_subject_holds = ROOT }
				}
				AND = {
					province_id = 2955
					NOT = { country_or_non_sovereign_subject_holds = ROOT }
				}
			}
		}
		trigger = {
			tag = BAV
			army_size_percentage = 1
			all_owned_province = {
				NOT = { unrest = 1 }
			}
		}
		effect = {
			#Actually, the Bavaria formation decision will give you these, so it is just a tooltip
			1757 = { #Oberpfalz
				if = {
					limit = {
						NOT = { is_permanent_claim = root }
						NOT = { is_core = root }
					}
					add_permanent_claim = root
				}
			}
			76 = { #Salzburg
				if = {
					limit = {
						NOT = { is_permanent_claim = root }
						NOT = { is_core = root }
					}
					add_permanent_claim = root
				}
			}
			2957 = { #Regensburg
				if = {
					limit = {
						NOT = { is_permanent_claim = root }
						NOT = { is_core = root }
					}
					add_permanent_claim = root
				}
			}
			
			add_country_modifier = {
				name = bav_rising_power
				duration = 7300
			}
		}
	}
	bav_expand_in_bavaria = {
		icon = mission_unite_home_region
		required_missions = { bav_reunite_bavaria }
		position = 3
		completed_by = 1618.5.6
        provinces_to_highlight = {
			OR = {
				AND = {
					OR = {
						province_id = 1757
						province_id = 76
						province_id = 2957
					}
					ROOT = {
						NOT = { owns_or_non_sovereign_subject_of = 1757 }
						NOT = { owns_or_non_sovereign_subject_of = 76 }
						NOT = { owns_or_non_sovereign_subject_of = 2957 }
					}
				}
			}
        }
        trigger = {
            num_of_cities = 10
			OR = {
				1757 = { #Oberpfalz
					country_or_non_sovereign_subject_holds = ROOT
				}
				76 = { #Salzburg
					country_or_non_sovereign_subject_holds = ROOT
				}
				2957 = { #Regensburg
					country_or_non_sovereign_subject_holds = ROOT
				}
			}
        }
		effect = {
			if = {
				limit = {
					has_dlc = "Emperor"
					valid_for_personal_unions_trigger = yes
					exists = PAL
					PAL = {
						valid_for_personal_unions_trigger = yes
					}
				}
				add_casus_belli = {
					type = cb_restore_personal_union
					months = 300
					target = PAL
				}
			}
			else = {
				palatinate_area = {
					limit = {
						NOT = { is_permanent_claim = root }
						NOT = { is_core = root }
					}
					add_permanent_claim = root
				}
				if = {
					limit = {
						exists = PAL
					}
					custom_tooltip = gain_claims_on_palatinate_provinces
					hidden_effect = {
						PAL = {
							every_owned_province = {
								limit = {
									NOT = { area = palatinate_area }
									NOT = { is_permanent_claim = root }
									NOT = { is_core = root }
								}
								add_permanent_claim = root
							}
						}
					}
				}
			}
			franconia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = {  is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			upper_franconia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = {  is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}

	conquer_franken = {
		icon = mission_cannons_firing
		required_missions = { bav_expand_in_bavaria }
		completed_by = 1806.1.1
		provinces_to_highlight = {
			OR = {
				area = franconia_area
				area = upper_franconia_area
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 6
				OR = {
					area = franconia_area
					area = upper_franconia_area
				}
			}
		}
		effect = {
			add_prestige = 20
			add_country_modifier = {
				name = "bavarian_ambition"
				duration = 7300
			}
			lower_swabia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = {  is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
			upper_swabia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = {  is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}

	conquer_ansbach = {
		icon = mission_landsknecht_soldier
		required_missions = { conquer_franken }
		provinces_to_highlight = {
			OR = {
				area = lower_swabia_area
				area = upper_swabia_area
			}
			NOT = {
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 6
				OR = {
					area = lower_swabia_area
					area = upper_swabia_area
				}
			}
		}
		effect = {
			add_adm_power = 100
			upper_rhineland_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = {  is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
}

				}
			}

		effect = {
			add_prestige = 10

			palatinate_area = {
				limit = { owned_by = ROOT }
				add_province_modifier = {
					name = faster_integration
					duration = 7300
				}
			}
		}
	}
