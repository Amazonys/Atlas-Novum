ha_sunset_group = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		technology_group = high_american
		capital_scope = {
			OR = {
				continent = north_america
				continent = south_america
				continent = new_world
			}
		}
		NOT = { has_reform = pirate_republic_reform }
	}
	has_country_shield = yes
	
	gain_foothold_in_ireland = {
		icon = mission_conquer_50_development
		required_missions = { }
		position = 6
		
		provinces_to_highlight = {
			OR = {
				area = munster_area
				area = leinster_area
				area = connacht_area
				area = ulster_area
			}
			NOT = { owned_by = ROOT }
			has_discovered = ROOT
		}
		
		trigger = {
			custom_trigger_tooltip = {
				tooltip = gain_foothold_in_ireland.tooltip
				british_isles_region = {
					has_discovered = ROOT
				}
				OR = {
					munster_area = {
						owned_by = ROOT
						#has_port = yes
					}
					leinster_area = {
						owned_by = ROOT
						#has_port = yes
					}
					connacht_area = {
						owned_by = ROOT
						#has_port = yes
					}
					ulster_area = {
						owned_by = ROOT
						#has_port = yes
					}
				}
			}
		}
		
		effect = {
			add_prestige = 15
			british_isles_region = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	ha_conquer_ireland = {
		icon = mission_unite_home_region
		required_missions = { gain_foothold_in_ireland }
		position = 7
		
		provinces_to_highlight = {
			region = british_isles_region
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
			has_discovered = ROOT
		}
		
		trigger = {
			british_isles_region = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		
		effect = {
			add_treasury = 500
			add_adm_power = 60
			add_dip_power = 60
			add_mil_power = 60
		}
	}
}
