		}
		trigger = {

			OR = {
				alliance_with = ENG

					}
				}
			}
		}
		effect = {
			loire_area = {
				limit = {
					NOT = { is_core = ROOT }

		}
	}

		}
		effect = {
			normandy_area = {
				limit = {
					NOT = { is_core = ROOT }

				}
				add_permanent_claim = ROOT

		}

				type = all

			}
		}
		effect = {
			ile_de_france_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT

bri_2 = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		is_random_new_world = no
		tag = BRI
	}
	has_country_shield = yes
	
	bri_brest = {
		icon = mission_establish_high_seas_navy
		required_missions = {  }
		provinces_to_highlight = {
			province_id = 170
			OR = {
				has_dock_building_trigger = no
				NOT = { owned_by = ROOT }
				NOT = { controlled_by = ROOT }
			}
		}
		trigger = {
			170 = {
				owned_by = ROOT
				controlled_by = ROOT
				has_dock_building_trigger = yes
			}
		}
		effect = {
			if = {
				limit = {
					170 = {
						NOT = {
							has_building = drydock
						}
					}
				}
				170 = {
					hidden_effect = { remove_building = dock }
					add_building = drydock
				}
			}
			170 = {
				add_unit_construction = {
					type = heavy_ship
					amount = 5
					speed = 1
					cost = 0
				}
			}
			munster_area = {
				limit = {
					NOT = { is_claim = root }
					NOT = { is_core = root }
				}
				add_claim = root
			}
		}
	}
	bri_new_brittany = {
		icon = mission_invade_island
		required_missions = { bri_brest }
		provinces_to_highlight = {
			colonial_region = colonial_canada
			NOT = { country_or_non_sovereign_subject_holds = root }
			has_port = yes
			has_discovered = root
		}
		trigger = {
			custom_trigger_tooltip = {
				tooltip = has_colony_in_colonial_canada
				colonial_canada = {
					owner = {
						is_colonial_nation_of = root
					}
				}
			}
		}
		effect = {
			add_country_modifier = {
				name = bri_colonialism
				duration = 7300
			}
		}
	}
	bri_better_weather = {
		icon = mission_iberian_conquistador
		required_missions = { bri_new_brittany }
		provinces_to_highlight = {
			colonial_region = colonial_the_carribean
			NOT = { country_or_non_sovereign_subject_holds = root }
			has_discovered = root
		}
		trigger = {
			custom_trigger_tooltip = {
				tooltip = has_colony_in_colonial_caribbean
				colonial_the_carribean = {
					owner = {
						is_colonial_nation_of = root
					}
				}
			}
		}
		effect = {
			1516 = {
				add_trade_modifier = {
					who = root
					duration = 7300
					power = 30
					key = bri_breton_better_weather
				}
			}
		}
	}
	bri_atlantic_trade = {
		icon = mission_dominate_home_trade_node
		position = 5
		required_missions = { bri_better_weather }
		provinces_to_highlight = {
			#the two Atlantic trade nodes
			OR = {
				province_id = 1490
				province_id = 1516
			}
			NOT = { is_strongest_trade_power = root }
			has_discovered = root
		}
		trigger = {
			1490 = { is_strongest_trade_power = root }
			1516 = { is_strongest_trade_power = root }
			calc_true_if = {
				amount = 2
				all_subject_country = {
					is_colonial_nation_of = root
					num_of_cities = 10
				}
			}
		}
		effect = {
			if = {
				limit = {
					owns = 170 #Finistere
				}
				170 = {
					if = {
						limit = {
							NOT = { province_has_center_of_trade_of_level = 3 }
						}
						add_center_of_trade_level = 1
						add_base_production = 2
					}
					else = { add_base_production = 3 }
				}
			}
			else = { add_dip_power = 100 }
			if = {
				limit = {
					owns = 172 #Nantes
				}
				172 = {
					add_base_production = 2
				}
			}
			else = { add_adm_power = 100 }
		}
	}
