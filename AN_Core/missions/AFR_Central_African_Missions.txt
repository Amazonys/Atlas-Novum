central_africa_military_missions = {
	slot = 2
	generic = yes
	ai = yes
	
	potential_on_load = {
		has_dlc = "Origins"
	}
	
	potential = {
		has_dlc = "Origins"
		capital_scope = {
			OR = {
				region = central_africa_region
				region = kongo_region
			}
		}
		NOT = { TAG = KON }
	}
	
	c_a_prepare_for_war_mission = {
		icon = mission_ne_palisades
		required_missions = {  }
		position = 1
		trigger = {
			army_size_percentage = 1
			has_mil_advisor = yes
		}
		effect = {
			capital_scope = {
				area = { 
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_claim = ROOT 
				}
			}
			add_army_tradition = 10
			add_country_modifier = {
				name = c_a_preparing_for_war	
				duration = 9125
			}
		}
	}
	
	c_a_train_army_mission = {
		icon = mission_son_military_expertise
		required_missions = { c_a_prepare_for_war_mission }
		position = 2
		trigger = {
			if = {
				limit = { has_dlc = "Cradle of Civilization" }
				army_professionalism = 0.25
			}
			army_tradition = 50
		}
		effect = {
			add_prestige = 20
			if = {
				limit = { has_dlc = "Rule Britannia" }
				change_innovativeness = 3
			}
			add_country_modifier = {
				name = c_a_training
				duration = 9125
			}
		}
	}
	

	c_a_conquest_mission = {
		icon = mission_unite_home_region
		required_missions = { 
			c_a_train_army_mission
			c_a_war_mission
		}
		position = 3
		trigger = { grown_by_states = 5 }
		effect = {
			capital_scope = {
				region = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_claim = ROOT 
				}
			}
		}
	}
	
	c_a_military_hegemon_mission = {
		icon = mission_assemble_an_army
		required_missions = { c_a_conquest_mission }
		position = 4
		trigger = {
			custom_trigger_tooltip = {
				tooltip = c_a_military_hegemon_mission_ct
				all_known_country = {
					OR = {
						NOT = {
							capital_scope = {
								OR = {
									region = central_africa_region
									region = kongo_region
								}
							}
						}
						NOT = { army_size = ROOT }
					}
				}
			}
		}		
		effect = {
			add_power_projection = {
				type = mission_rewards_power_projection
				amount = 25
			}
			add_army_tradition = 20
			add_country_modifier = {
				name = c_a_hegemon	
				duration = 9125
			}
		}
	}

	c_a_become_empire_mission = {
		icon = mission_empire
		required_missions = { 
			c_a_humiliate_rival_mission
			c_a_military_hegemon_mission
		}
		position = 5
		trigger = { 
			if = {
				limit = { has_dlc = "Common Sense" }
				government_rank = 3 
			}
			else = { total_development = 1000 }		
		}
		effect = {
			add_adm_power = 100
			add_dip_power = 100
			add_mil_power = 100
			add_prestige = 20
		}
	}

}

	c_a_get_cults_mission = {

		required_missions = {  }
		position = 1
		trigger = {

			religion = shamanism
		}
		effect = {
			add_country_modifier = {
				name = c_a_united_cults
				duration = -1
			}
		}
	}
	
	c_a_contact_muslims_mission = {
		icon = mission_indian_stateman
		required_missions = {
			c_a_get_cults_mission
			c_a_ally_big_mission
		}
		position = 2
		trigger = {
			custom_trigger_tooltip = {
				tooltip = c_a_contact_muslims_mission_ct
				any_ally = {
					AND = {
						religion_group = muslim
						has_opinion = {
							who = ROOT
							value = 200
						}
					}
				}
			}
		}
		effect = {
			add_prestige = 20
			if = {
				limit = { has_dlc = "Rule Britannia" }
				change_innovativeness = 5
			}
		}
	}
	
	c_a_expand_religion_mission = {
		icon = mission_sw_against_the_desert
		required_missions = { c_a_contact_muslims_mission }
		position = 3
		trigger = {
			custom_trigger_tooltip = {
				tooltip = c_a_expand_religion_mission_ct
				any_owned_province = {
					any_neighbor_province = {
						NOT = { religion = ROOT }
					}
				}
			}
			at_war_with_religious_enemy = yes
		}
		effect = {
			every_known_country = {
				limit = {
					capital_scope = { superregion = southern_africa_superregion }
					NOT = { religion = ROOT } 
				}
				reverse_add_casus_belli = {
					target = ROOT
					type = cb_crusade_pheasants
					months = 60
				}
			}
		}
	}
	
	c_a_convert_mission = {
		icon = mission_hands_praying
		required_missions = { c_a_expand_religion_mission }
		position = 4
		trigger = {
			custom_trigger_tooltip = {
				tooltip = c_a_convert_mission_tt
				check_variable = { which = num_converted_religion value = 10 }
			}
		}
		effect = {
			add_country_modifier = {
				name = c_a_only_truth
				duration = -1
			}
		}
	}
