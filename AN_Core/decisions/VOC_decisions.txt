country_decisions = {
	establish_voc = {
		major = yes
		potential = {
			tag = NED
			has_dlc = "Winds of Change"
			NOT = { has_country_flag = established_voc_flag }
		}
		allow = {
			has_estate_privilege = estate_burghers_voc_privilege
			num_of_owned_provinces_with = {
				superregion = india_superregion
				value = 4
			}
		}
		effect = {
			custom_tooltip = NED_ESTABLISH_VOC_IN_INDIA
			hidden_effect = {
				set_country_flag = established_voc_flag
				every_owned_province = {
					limit = {
						 superregion = india_superregion
					}
					cede_province = VOC
					add_core = VOC
				}
				
				create_subject = {
					subject_type = commercial_enterprise
					subject = VOC
				}
				VOC = {
					change_government = republic
					add_government_reform = trade_company_government
					change_religion = ROOT
					change_primary_culture = ROOT
					add_adm_power = 100
					add_dip_power = 100
					add_mil_power = 100
					add_treasury = 200
					set_prestige_to_zero = yes
					add_republican_tradition = 100
					swap_free_idea_group = yes
				}
				india_superregion = { 
					limit = { 	
						NOT = { is_core = VOC } 
						NOT = { country_or_non_sovereign_subject_holds = VOC }
						NOT = { owned_by = VOC } 
					}
					add_permanent_claim = VOC 
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}
	
	expand_voc = {
		major = yes
		potential = {
			VOC = {
				is_subject_of_type_with_overlord = { who = ROOT type = commercial_enterprise }
			}
			NOT = {
				india_superregion = {
					type = all
					owned_by = VOC
				}
			}
		}
		allow = {
			if = {
				limit = {
					ai = yes
				}
				has_estate_privilege = estate_burghers_voc_privilege
			}
			num_of_owned_provinces_with = {
				superregion = india_superregion
				value = 1
			}
		}
		effect = {
			custom_tooltip = NED_CEDE_PROVINCES_TO_VOC
			hidden_effect = {
				set_country_flag = established_voc_flag
				every_owned_province = {
					limit = {
						 superregion = india_superregion
					}
					cede_province = VOC
					add_core = VOC
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}
}
