#Decision for loading of mission modules
country_decisions = {

	AN_module_load = {
		potential = {
			has_country_flag = AN_uses_modules
		}
		allow = {
			NOT = { has_country_modifier = AN_module_timeout }
			if = {
				limit = { ai = yes }
				always = no
			}
		}
		effect = {
			country_event = { id = AN_module_load.1 days = 0 }
		}

		ai_will_do = {
			factor = 1000
		}
	}
}
