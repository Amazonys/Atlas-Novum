divorce_consort_aspect = {
	sprite = "GFX_divorce_consort_icon"
	cost = 100
	potential = {
		has_dlc = "Rights of Man"
		government = monarchy
		NOT = { has_reform = iqta }
		NOT = { has_reform = mamluk_government }
	}
	trigger = {
		has_consort = yes
		is_female = no
		has_heir = no
		is_lesser_in_union = no
		has_regency = no
	}
	effect = {
		custom_tooltip = divorce_consort_tooltip
		hidden_effect = {
			export_to_variable = {
				which = consort_adm_var
				value = consort_adm
			}
			export_to_variable = {
				which = consort_dip_var
				value = consort_dip
			}
			export_to_variable = {
				which = consort_mil_var
				value = consort_mil
			}
			#Multiply with -12 because a) it should be the inverse stat which gives monarch power (so divorcing a 0/0/0 gets you the most monarch power) and b) we assume the consort would have ruled for 12 months
			multiply_variable = {
				which = consort_adm_var
				value = -12
			}
			multiply_variable = {
				which = consort_dip_var
				value = -12
			}
			multiply_variable = {
				which = consort_mil_var
				value = -12
			}
			change_variable = {
				which = consort_adm_var
				value = 72
			}
			change_variable = {
				which = consort_dip_var
				value = 72
			}
			change_variable = {
				which = consort_mil_var
				value = 72
			}
			while = {
				limit = {
					check_variable = {
						which = consort_adm_var
						value = 12
					}
				}
				add_adm_power = 12
				subtract_variable = {
					which = consort_adm_var
					value = 12
				}
			}
			while = {
				limit = {
					check_variable = {
						which = consort_dip_var
						value = 12
					}
				}
				add_dip_power = 12
				subtract_variable = {
					which = consort_dip_var
					value = 12
				}
			}
			while = {
				limit = {
					check_variable = {
						which = consort_mil_var
						value = 12
					}
				}
				add_mil_power = 12
				subtract_variable = {
					which = consort_mil_var
					value = 12
				}
			}
			remove_consort = yes
		}
	}
	ai_will_do = {
		factor = 2
		modifier = {
			factor = 0
			consort_total_stats = {
				stats = 9
			}
		}
	}
}
