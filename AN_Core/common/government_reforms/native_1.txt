	lock_level_when_selected = yes

		has_reform = primitive_mechanic

			}
		}
	}

	modifiers = {
		stability_cost_modifier = -0.15
		advisor_cost = -0.1
		enemy_core_creation = 0.5
	}
	ai = {
		factor = 1
	}
}
