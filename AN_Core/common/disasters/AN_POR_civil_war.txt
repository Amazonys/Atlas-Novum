AN_POR_civil_war = {
	potential = {
		current_age = age_of_discovery
		tag = POR
		NOT = { has_country_flag = AN_POR_civil_war_complete }
		exists = BGC
		exists = CIM
	}


	can_start = {
		is_at_war = no
		has_any_disaster = no
		has_regency = no
	}


	can_stop = {
		has_any_disaster = yes
	}




	progress = {
		modifier = {
			factor = 10
			has_regency = no
		}
		modifier = {
			factor = 10
			NOT = { months_of_ruling = 60 }
		}
	}





	can_end = {
		OR = {
			custom_trigger_tooltip = {
				tooltip = AN_POR_civil_war_coimbra_win
				if = {
					limit = { war_with = BGC }
					232 = { country_or_non_sovereign_subject_controls = yes }
					3104 = { country_or_non_sovereign_subject_controls = yes }
				}
				227 = { country_or_non_sovereign_subject_controls = yes }
				231 = { country_or_non_sovereign_subject_controls = yes }
			}
			custom_trigger_tooltip = {
				tooltip = AN_POR_civil_war_braganza_win
				if = {
					limit = { war_with = CIM }
					1851 = { country_or_non_sovereign_subject_controls = yes }
				}
				if = {
					limit = { war_with = VSU }
					228 = { country_or_non_sovereign_subject_controls = yes }
				}
				227 = { country_or_non_sovereign_subject_controls = yes }
				231 = { country_or_non_sovereign_subject_controls = yes }
			}
		}
	}




	modifier = {
		global_unrest = 5
		legitimacy = -1
	}


	on_start = AN_disaster_POR_civil_war.1
	on_end = AN_disaster_POR_civil_war.100

	on_monthly = {
		events = {
		}
		random_events = {
			120 = 0
			1 = AN_disaster_POR_civil_war.2 #
		}
	}
}
