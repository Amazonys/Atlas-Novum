counts_feud = {
	potential = {
		tag = DAN
		government = monarchy
		normal_or_historical_nations = yes
		is_free_or_tributary_trigger = yes
		uses_doom = no
		NOT = { has_country_flag = had_counts_feud  }
	}


	can_start = {
		current_age = age_of_reformation
		has_any_disaster = no
		is_religion_enabled = protestant
		NOT = { religious_unity = 0.9 }	
		has_regency = no	
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			religious_unity = 1
			has_regency = yes
		}
	}
	
	
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 90 }
		}
		modifier = {
			factor = 1
			has_heir = no
		}
		modifier = {
			factor = 2
			has_new_dynasty = yes
		}
		modifier = {
			factor = 3
			NOT = { months_of_ruling = 60 }
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		
		
	}




	
	can_end = {
		NOT = {
			has_spawned_rebels = noble_rebels
		}
		NOT = {
			has_spawned_rebels = pretender_rebels
		}
		stability = 3
		hidden_trigger = {
			NOT = { has_country_flag = counts_feud_start_event }
		}
	}
	



	modifier = {
		global_unrest = 7
	}	
	
	
	on_start = counts_feud.1
	on_end = counts_feud.100
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0
			100 = counts_feud.2
			#3,4,5 are events triggering for Sweden, and reaction
			100 = counts_feud.6
		
		}
	}
}
