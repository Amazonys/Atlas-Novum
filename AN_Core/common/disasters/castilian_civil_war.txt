#Castilian Civil War
#A mix of the civil wars with outside intervention and noble families siding against the crown between 1419 and 1479.

castilian_civil_war = {
	potential = {
		tag = CAS
		NOT = { has_country_flag = had_cas_civil_war }
		is_free_or_tributary_trigger = yes
		exists = ARA
		exists = POR
		ARA = {
			is_free_or_tributary_trigger = yes
		}
		POR = {
			is_free_or_tributary_trigger = yes
		}
		government = monarchy
		normal_or_historical_nations = yes
		uses_doom = no
	}


	can_start = {
		current_age = age_of_discovery
		is_at_war = no
		has_any_disaster = no
		has_heir = yes
		OR = {
			has_regency = yes
			NOT = { heir_claim = 80 }
			has_female_heir = yes
		}
		is_lesser_in_union = no
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_lesser_in_union = yes
			NOT = { government = monarchy }
		}
	}
	
	
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 90 }
		}
		modifier = {
			factor = 1
			has_heir = no
		}
		modifier = {
			factor = 3
			NOT = { months_of_ruling = 60 }
		}
		modifier = {
			factor = 1
			NOT = { stability = 1 }
		}
		modifier = {
			factor = 0.5
			NOT = { ADM = 2 }
		}		
		modifier = {
			factor = 0.5
			NOT = { DIP = 2 }
		}		
		modifier = {
			factor = 0.5
			NOT = { MIL = 2 }
		}	
		modifier = {
			factor = -1
			stability = 3
		}
	}




	
	can_end = {
		hidden_trigger = { has_country_flag = civil_war_in_castile }
		stability = 1
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = {
			has_spawned_rebels = pretender_rebels
		}
	}
	



	modifier = {
		global_unrest = 7
	}	
	
	
	on_start = castilian_civil_war.1
	on_end = castilian_civil_war.100
	
	on_monthly = {
		events = {
		}
		random_events = { 
			120 = 0
			1 = castilian_civil_war.2 #Civil War Spreads to Province
			#Events 3 - 8 related to foreign intervention
		}
	}
}
