estate_eunuchs_disaster = {
	potential = {
		has_estate = estate_eunuchs
		estate_influence = {
			estate = estate_eunuchs
			influence = 60
		}
	}


	can_start = { 
		estate_influence = {
			estate = estate_eunuchs
			influence = 100
		}
		corruption = 4 
	}
	
	can_stop = {
		OR = {
			NOT = {
				estate_influence = {
					estate = estate_eunuchs
					influence = 100
				}
			}
			NOT = { corruption = 2 } 
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_eunuchs
					influence = 100
				}
			}
		}
		modifier = {
			factor = 1
			NOT = { corruption = 2.5 } 
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_eunuchs
					loyalty = 50
				}
			}
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_eunuchs
					loyalty = 40
				}
			}
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_eunuchs
					loyalty = 30
				}
			}
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_eunuchs
					loyalty = 20
				}
			}
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_EUNUCHS_TT
			NOT = { has_country_flag = estate_eunuchs_in_power }
		}
	}
	
	modifier = {	
		imperial_mandate = -0.15
		global_monthly_devastation = 0.1
		meritocracy = -0.5
		global_unrest = 20
		administrative_efficiency = -0.25
		
	}

	on_start = estate_disasters.19
	on_end = estate_disasters.20
	
	on_monthly = { }
}
