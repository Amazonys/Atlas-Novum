revolution = {
	ended_by_country_breaking_to_rebels = no
	potential = {
		NOT = { has_country_flag = had_revolution }
		NOT = { tag = PAP }
		OR = {
			NOT = { government = monarchy }
			NOT = { tag = FRA }
		}
		#NOT = { has_reform = celestial_empire }
		is_free_or_tributary_trigger = yes
		uses_doom = no
		current_age = age_of_revolutions
		is_revolutionary = no
		if = {
			limit = { NOT = { has_disaster = tur_disaster_ottoman_power_struggle } }
			if = {
				limit = {
					has_dlc = "Emperor"
				}
				is_great_power = yes
				revolution_spread = 0.2
				capital_scope = {
					OR = {
						continent = europe
						has_revolution_in_province = yes
					}
				}
			}
			else = {
				num_of_cities = 30
				capital_scope = { continent = europe }
				NOT = { revolution_target_exists = yes }
			}
		}
	}

	
	
	can_start = {
		current_age = age_of_revolutions
		if = {
			limit = {
				NOT = { has_disaster = tur_disaster_ottoman_power_struggle }
			}
			has_any_disaster = no
		}
		else = {
			calc_true_if = {
				has_disaster = tur_disaster_eyalet_rebellion
				has_disaster = tur_disaster_janissary_coup
				has_disaster = tur_disaster_pasha_decadence
				has_disaster = tur_disaster_plot_of_the_harem
				amount = 3
			}
		}
		OR = {
			NOT = { stability = 2 }
			AND = {
				capital_scope = { controlled_by = REB }
				has_spawned_rebels = revolutionary_rebels
				num_of_rebel_controlled_provinces = 8
			}
		}
		OR = {
			NOT = { stability = 0 }
			revolution_spread = 0.5
			num_of_loans = 50
			is_bankrupt = yes
			inflation = 10
			war_exhaustion = 10
			NOT = { prestige = 0 }
			custom_trigger_tooltip = {
				tooltip = root_is_most_powerful_in_europe
				capital_scope = { continent = europe }
				NOT = {
					any_country = {
						capital_scope = { continent = europe }
						OR = {
							army_size = root
							total_own_and_non_tributary_subject_development = root
						}
					}
				}
			}
			AND = {
				capital_scope = { controlled_by = REB }
				has_spawned_rebels = revolutionary_rebels
				num_of_rebel_controlled_provinces = 8
			}
		}
	}
	
	can_stop = { 
		if = {
			limit = { NOT = { has_disaster = tur_disaster_ottoman_power_struggle } }
			OR = {
				has_any_disaster = yes
				stability = 3
			}
		}
		else = {
			stability = 3
		}
	}
	
	
	progress = {
		modifier = {
			factor = 0.5
			revolution_spread = 0.5
			NOT = { revolution_spread = 0.65 }
		}
		modifier = {
			factor = 0.75
			revolution_spread = 0.65
			NOT = { revolution_spread = 0.8 }
		}
		modifier = {
			factor = 1
			revolution_spread = 0.8
		}
		modifier = {
			factor = 1
			NOT = { prestige = 0 }
		}
		modifier = {
			factor = 1
			war_exhaustion = 5
		} 
		modifier = {
			factor = 0.5
			num_of_loans = 10
			NOT = { num_of_loans = 25 }
		}	
		modifier = {
			factor = 0.5
			absolutism = 25
		}
		modifier = {
			factor = 0.5
			absolutism = 40
		}
		modifier = {
			factor = 0.5
			absolutism = 50
		}
		modifier = {
			factor = 0.5
			absolutism = 60
		}
		modifier = {
			factor = 0.5
			absolutism = 70
		}
		modifier = {
			factor = 0.5
			absolutism = 80
		}
		modifier = {
			factor = 1
			num_of_loans = 25
		}
		modifier = {
			factor = 3
			capital_scope = { controlled_by = REB }
		}
		modifier = {
			factor = 5
			is_bankrupt = yes
		}
		modifier = {
			factor = 0.5
			custom_trigger_tooltip = {
				tooltip = root_is_most_powerful_in_europe_2
				capital_scope = { continent = europe }
				NOT = {
					any_country = {
						capital_scope = { continent = europe }
						OR = {
							army_size = root
							total_own_and_non_tributary_subject_development = root
						}
					}
				}
			}
		}
		modifier = {
			factor = 1
			capital_scope = {
				has_revolution_in_province = yes
			}
		}
	}
	
	can_end = {
		if = {
			limit = {
				has_dlc = "Emperor"
			}
			OR = {
				custom_trigger_tooltip = {
					tooltip = disaster_active_for_20_years_tooltip
					had_country_flag = {
						flag = had_revolution
						days = 7300
					}
				}
				stability = 3
			}
			NOT = { num_of_rebel_controlled_provinces = 1 }
			NOT = { num_of_rebel_armies = 1 }
		}
		else = {
			OR = {
				AND = {
					OR = {
						custom_trigger_tooltip = {
							tooltip = disaster_active_for_20_years_tooltip
							had_country_flag = {
								flag = had_revolution
								days = 7300
							}
						}
						stability = 3
					}
					NOT = { num_of_rebel_controlled_provinces = 1 }
					NOT = { num_of_rebel_armies = 1 }
					OR = {
						government = monarchy
						has_reform = revolutionary_republic_reform
					}
				}
				AND = {
					revolution_target_exists = yes
					NOT = { root = { is_revolution_target = yes } }
				}
			}
		}
	}
	

	modifier = {
		global_unrest = 10
		stability_cost_modifier = 0.1
	}
