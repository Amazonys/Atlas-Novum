# The custom guiButtonType/instantTextBoxType/iconType must have a 'scripted = yes' line to work
custom_button = {
    name = AN_census_tab #Must match a scripted guiButtonType in a .gui file
    potential = { always = yes } #Determines when the button is visible
    trigger = { hidden_trigger = { owned_by = FROM controlled_by = FROM } } #Determines when the button is clickable
    effect = { 
		hidden_effect = {
			AN_check_dev_variables = yes
			province_event = { id = census.1 days = 0 }
		}
	} #Effect
    tooltip = AN_census_tooltip #Can use customizable localisation
}

custom_button = {
    name = AN_wealth_button #Must match a scripted guiButtonType in a .gui file
    potential = { always = yes } #Determines when the button is visible
    trigger = { 
		hidden_trigger = { 
			owned_by = FROM 
			controlled_by = FROM 
			owner = { adm_power = 25 treasury = 25 } 
		} 
	} #Determines when the button is clickable
    effect = { 
		hidden_effect = {
			AN_check_dev_variables = yes
			add_base_tax = 1
			owner = { add_adm_power = -25 add_treasury = -25 }
		}
	} #Effect
    tooltip = AN_wealth_tooltip #Can use customizable localisation
}

custom_button = {
    name = AN_infra_button #Must match a scripted guiButtonType in a .gui file
    potential = { always = yes } #Determines when the button is visible
    trigger = { 
		hidden_trigger = { 
			owned_by = FROM 
			controlled_by = FROM 
			
			check_variable = {
        		which = AN_prov_infrastructure_cap
        		which = AN_prov_infrastructure
    		}
			check_variable = {
				which = AN_prov_wealth
				which = AN_prov_infrastructure
    		}		
			owner = { mil_power = 25 }
		} 
	} #Determines when the button is clickable
    effect = { 
		hidden_effect = {
			AN_check_dev_variables = yes
			add_base_tax = -2
			add_base_production = 1
			owner = { add_mil_power = -25 }
		}
	} #Effect
    tooltip = AN_infra_tooltip #Can use customizable localisation
}
