num_of_issues_passed = {
	if = {
		limit = {
			OR = {
				has_dlc = "Common Sense"
				has_dlc = "Domination"
			}
			has_parliament = yes
		}
		custom_trigger_tooltip = {
			tooltip = num_of_issues_passed_$amount$
			check_variable = {
				which = issues_passed
				value = $amount$
			}
		}
	}
	else = {
		[[alternative_trigger]$alternative_trigger$]
	}
}

	}
}
