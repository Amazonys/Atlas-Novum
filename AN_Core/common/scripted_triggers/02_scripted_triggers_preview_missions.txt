	has_country_flag = has_batch_$batch$_active
}

not_in_mission_preview_mode = {
	if = {
		limit = {
			check_variable = {
				which = can_preview_missions_var
				value = 1
			}
			can_preview_mission_of_key = { key = $key$ }
		}
		custom_trigger_tooltip = {
			tooltip = BYZ_we_are_in_preview
			always = no
		}
	}
}
