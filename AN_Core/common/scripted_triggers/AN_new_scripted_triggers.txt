control_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = country_holds_or_non_sovereign_subject_controls_highlighted
		NOT = {
			any_province = {
				$TYPE$ = $TARGET$
				NOT = {
					country_or_non_sovereign_subject_holds = ROOT
					controller = {
						OR = { 
							tag = ROOT
							is_subject_of = ROOT
						}
						NOT = { is_subject_of_type = tributary_state }  
					}
				}
			}
		}
	}
}

blockade_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = country_holds_or_non_sovereign_subject_blockades_highlighted
		NOT = {
			any_province = {
				$TYPE$ = $TARGET$
				NOT = {
					country_or_non_sovereign_subject_holds = ROOT
					OR = {
						is_blockaded_by = ROOT 
						country_or_non_sovereign_subject_holds = ROOT
					}
				}
			}
		}
	}
}

conquest_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		NOT = {
			any_province = {
				$PROVINCES$ = yes
				NOT = { $OWNER_TYPE$ = ROOT }
			}
		}
		NOT = {
			any_province = {
				$PROVINCES$ = yes
				NOT = { is_city = yes }
			}
		}
	}
}



discover_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		calc_true_if = {
			all_province = {
				$PROVINCES$ = yes
				has_discovered = ROOT
			}
			amount = $NUM$
		}
	}
}

num_colonized_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
			$PROVINCES$ = yes
			is_city = yes
			value = $NUM$
		}
	}
}

vassalize_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		if = { 
			limit = { $TAG$ = { exists = yes } }
			$TAG$ = { 
				is_subject_of = ROOT 
				NOT = { is_subject_of_type = tributary_state }
			}
			NOT = {
				any_province = {
					is_core = $TAG$
					NOT = {	country_or_non_sovereign_subject_holds = ROOT }
				}
			}
		}
		else_if = {
			limit = { $TAG$ = { exists = no } }
			NOT = {
				any_province = {
					is_core = $TAG$
					NOT = {	owned_by = ROOT }
				}
			}
		}
	}
}

event_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		OR = {
			has_country_flag = $EVENT$_happened
			is_year = $BYPASS_DATE$
		}
	}
}

AN_can_chop_forest = {
	OR = {
		AND = {
			has_province_modifier = $TYPE$_1
			owner = {
				OR = {
					ai = no
					treasury = 50
				}
			}
		}
		AND = {
			has_province_modifier = $TYPE$_2
			owner = {
				OR = {
					ai = no
					treasury = 100
				}
			}
		}
		AND = {
			has_province_modifier = $TYPE$_3
			owner = {
				OR = {
					ai = no
					treasury = 150
				}
			}
		}
	}
	NOT = { has_province_modifier = $TYPE$_chopped }
}

AN_can_reforest = {
	OR = {
		AND = {
			has_province_flag = $TYPE$_residual
			owner = {
				OR = {
					ai = no
					treasury = 25
				}
			}
		}
		AND = {
			has_province_modifier = $TYPE$_1
			owner = {
				OR = {
					ai = no
					treasury = 150
				}
			}
		}
		AND = {
			has_province_modifier = $TYPE$_2
			owner = {
				OR = {
					ai = no
					treasury = 200
				}
			}
		}
		any_neighbor_province = {
			OR = {
				has_province_modifier = $TYPE$_1
				has_province_modifier = $TYPE$_2
				has_province_modifier = $TYPE$_3
			}
		}
	}
	NOT = {
		has_province_modifier = $TYPE$_chopped
		has_terrain = desert
		has_terrain = coastal_desert
		has_terrain = glacier
	}
}
