native_palisade = { #Palisade
	cost = 50
	time = 12
	build_trigger = {
		owner = { government = native }
		OR = {
			has_building = native_village
			has_building = native_shrine
		}
	}
	keep_trigger = {
		owner = { government = native }
	}
	modifier = {
		fort_level = 1
		local_defensiveness = 0.1
	}
	government_specific = yes
	influencing_fort = yes

	onmap = yes

	ai_will_do = {
		factor = 100
		modifier = {
			factor = 2.0
			is_capital = yes
		}
	}
}
