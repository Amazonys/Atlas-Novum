
######################################################################
# Ghulams rebels.
######################################################################
ghulams_rebels = {
	color = { 170 95 25 }

	area = nation 					# Stays within the nation
	government = monarchy			# Likes monarchy
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit_transfer = yes
	gfx_type = culture_owner	
	will_relocate = yes	
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.3
	infantry = 0.6
	cavalry = 0.1
	
	morale = 1.25
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0.01
			is_overseas = yes
		}
		modifier = {
			factor = 2
			owner = { religion_group = christian }
		}
		modifier = {
			factor = 0
			has_owner_religion = yes
		}
		modifier = {
			factor = 2
			owner = {
				estate_influence = {
					estate = estate_ghulams
					influence = 80
				}
			}
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		NOT = { local_autonomy = 50 }
	}
	siege_won_effect = {
		add_local_autonomy = 10
		owner = {
			if = {
				limit = { has_estate = estate_ghulams }
				change_estate_land_share = {
					estate = estate_ghulams
					share = 1
					province = ROOT
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = ghulams_rebels_demands
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		if = {
			limit = {
				has_estate = estate_ghulams
				NOT = {
					has_estate_influence_modifier = {
						estate = estate_ghulams
						modifier = EST_VAL_NOBLES_EXTORTED_RIGHTS
					}
				}
			}
			add_estate_influence_modifier = {
				estate = estate_ghulams
				desc = EST_VAL_NOBLES_EXTORTED_RIGHTS
				influence = 15
				duration = 5475
			}
		}
		every_owned_province = {
			limit = {
				has_rebel_faction = REB
			}
			add_local_autonomy = 20
		}	
		add_prestige = -20
		add_legitimacy = -20
		add_horde_unity = -20
		add_devotion = -20
		add_republican_tradition = -10
		if = {
			limit = {
				has_disaster = court_and_country
			}
			set_country_flag = court_and_country_surrendered_to_rebels
			hidden_effect = { country_event = { id = court_and_country_events.100 } }
			end_disaster = court_and_country
		}
		if = {
			limit = { has_estate = estate_ghulams }
			change_estate_land_share = {
				estate = estate_ghulams
				share = 10
			}
		}
		add_absolutism = -20
		kill_ruler = yes
	}
}
