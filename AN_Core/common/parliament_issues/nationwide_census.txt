nationwide_census = {

	category = 3
	
	allow = {
		num_of_cities = 6
	}
	
	on_issue_taken = {
		every_owned_province = {
			clr_province_flag = nationwide_census_flag
		}
		random_owned_province = {
			limit = {
				is_in_capital_area = yes
			}
			set_province_flag = nationwide_census_flag
		}
	}
	
	effect = {
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_flag = nationwide_census_flag }
				add_base_manpower = 1
				clr_province_flag = nationwide_census_flag
			}
		}
		if = {
			limit = {
				NOT = {
					any_owned_province = {
						has_province_flag = nationwide_census_flag
					}
				}
			}
			tooltip = {
				custom_tooltip = nationwide_census_tooltip
			}
		}
		if = {
			limit = {
				any_owned_province = {
					has_province_flag = nationwide_census_flag
				}
			}
			tooltip = {
				every_owned_province = {
					limit = { has_province_flag = nationwide_census_flag }
					add_base_manpower = 1
				}
			}
		}
	}
	
	modifier = {
		global_manpower_modifier = 0.05
	}

	ai_will_do = {
		factor = 1	
	}	
}
