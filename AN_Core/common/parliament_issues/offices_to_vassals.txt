offices_to_vassals = {

	category = 3

	allow = {
		OR = {
			vassal = 2
			personal_union = 2
			AND = {
				vassal = 1
				personal_union = 1
			}
		}
		government = monarchy
	}
	
	on_issue_taken = {
		every_subject_country = {
			clr_country_flag = offices_to_vassals
		}
		random_subject_country = {
			limit = {
				OR = {
					is_vassal = yes
					is_lesser_in_union = yes
				}
			}
			set_country_flag = offices_to_vassals
		}
	}

	effect = {
		if = {
			limit = {
				NOT = {
					any_subject_country = {
						OR = {
							is_vassal = yes
							is_lesser_in_union = yes
						}
						has_country_flag = offices_to_vassals
					}
				}
			}
			tooltip = {
				custom_tooltip = offices_to_vassals_tooltip
			}
		}
		if = {
			limit = {
				any_subject_country = {
					OR = {
						is_vassal = yes
						is_lesser_in_union = yes
					}
					has_country_flag = offices_to_vassals
				}
			}
			tooltip = {
				random_subject_country = {
					limit = {
						OR = {
							is_vassal = yes
							is_lesser_in_union = yes
						}
						has_country_flag = offices_to_vassals
					}
					add_opinion = {
						who = FROM
						modifier = parliament_closer_ties
					}
				}
			}
		}
		hidden_effect = {
			if = {
				limit = {
					any_subject_country = {
						OR = {
							is_vassal = yes
							is_lesser_in_union = yes
						}
						has_country_flag = offices_to_vassals
					}
				}
				random_subject_country = {
					limit = {
						OR = {
							is_vassal = yes
							is_lesser_in_union = yes
						}
						has_country_flag = offices_to_vassals
					}
					add_opinion = {
						who = FROM
						modifier = parliament_closer_ties
					}
					clr_country_flag = offices_to_vassals
				}
			}
			else = {
				random_subject_country = {
					limit = {
						OR = {
							is_vassal = yes
							is_lesser_in_union = yes
						}
					}
					add_opinion = {
						who = FROM
						modifier = parliament_closer_ties
					}
				}
			}
		}
	}
	
	modifier = {
		diplomatic_annexation_cost = -0.15
		vassal_income = 0.05
	}

	chance = {
		factor = 1
		modifier = {
			factor = 2
			vassal = 3
		}
		modifier = {
			factor = 3
			vassal = 4
		}
		modifier = {
			factor = 4
			vassal = 5
		}
	}
	
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 2
			vassal = 4
		}
		modifier = {
			factor = 2
			personal_union = 2
		}	
	}
}
