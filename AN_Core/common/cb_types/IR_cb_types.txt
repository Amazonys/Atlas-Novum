cb_fief_vassal = {
	valid_for_subject = no

	prerequisites = {
		OR = {
			current_age = age_of_discovery
			current_age = age_of_reformation
		}
		is_fief_domain_tech = yes
		is_subject = no
		if = {
			limit = { is_part_of_hre = yes }
			FROM = {
				is_subject = no
				NOT = { num_of_cities = ROOT }
				NOT = { is_subject_of = ROOT }
				NOT = { is_great_power = yes }
				NOT = { government_rank = 4 }
				is_part_of_hre = yes
				is_elector = no
				NOT = { has_reform = imperial_city }
				is_emperor = no
				religion_group = ROOT
				is_neighbor_of = ROOT
				is_fief_domain_tech = yes
			}
		}
		else_if = {
			limit = { religion_group = muslim }
			FROM = {
				is_subject = no
				NOT = { num_of_cities = ROOT }
				NOT = { is_subject_of = ROOT }
				NOT = { is_great_power = yes }
				NOT = { government_rank = 4 }
				OR = {
					religion_group = ROOT
					culture_group = ROOT
				}
				is_neighbor_of = ROOT
				is_fief_domain_tech = yes
			}
		}
		else = {
			FROM = {
				is_subject = no
				NOT = { num_of_cities = ROOT }
				NOT = { is_subject_of = ROOT }
				NOT = { is_great_power = yes }
				NOT = { government_rank = 4 }
				culture_group = ROOT
				religion_group = ROOT
				is_neighbor_of = ROOT
				is_part_of_hre = no
				is_fief_domain_tech = yes
			}
		}
	}

	attacker_disabled_po = {
		po_demand_provinces
