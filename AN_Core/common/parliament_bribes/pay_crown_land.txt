pay_crown_land = {
	trigger = {
		has_reached_seat_threshold = no
		crown_land_share = 10
		NOT = { crown_land_share = 100 }
		has_not_bribe_blocking_debate_active = {
			type = crownland
		}
	}
	
	effect = {
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			change_estate_land_share = {
				estate = all
				share = 1
			}
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			change_estate_land_share = {
				estate = all
				share = 0.5
			}
		}
		else = {
			change_estate_land_share = {
				estate = all
				share = 0.25
			}
		}
	}
	
	chance = {
		factor = 1
	}

	ai_will_do = {
		factor = 0.75
		modifier = {
			factor = 0.5
			NOT = { crown_land_share = 50 }
		}
	}
}

national_pay_crown_land = {
	trigger = {
		has_reached_seat_threshold = yes
		crown_land_share = 10
		NOT = { crown_land_share = 100 }
		has_not_bribe_blocking_debate_active = {
			type = crownland
		}
	}
	
	effect = {
		back_seats_of_bribe_type = {
			bribe = pay_crown_land
		}
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			change_estate_land_share = {
				estate = all
				share = 5
			}
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			change_estate_land_share = {
				estate = all
				share = 2.5
			}
		}
		else = {
			change_estate_land_share = {
				estate = all
				share = 1.25
			}
		}
	}
	
	chance = {
		factor = 1
	}
	
	ai_will_do = {
		factor = 0.75
		modifier = {
			factor = 0.5
			NOT = { crown_land_share = 50 }
		}
	}
}
