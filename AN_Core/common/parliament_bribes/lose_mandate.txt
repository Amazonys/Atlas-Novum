lose_mandate = {
	trigger = {
		has_reached_seat_threshold = no
		has_dlc = "Mandate of Heaven"
		is_emperor_of_china = yes
		imperial_mandate = 2
		has_not_bribe_blocking_debate_active = {
			type = mandate
		}
	}
	
	effect = {
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			add_mandate = -4
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			add_mandate = -2
		}
		else = {
			add_mandate = -1
		}
	}
	
	chance = {
		factor = 1
	}
	
	ai_will_do = {
		factor = 1
	}
}

national_lose_mandate = {
	trigger = {
		has_reached_seat_threshold = yes
		has_dlc = "Mandate of Heaven"
		is_emperor_of_china = yes
		imperial_mandate = 6
		has_not_bribe_blocking_debate_active = {
			type = mandate
		}
	}
	
	effect = {
		back_seats_of_bribe_type = {
			bribe = lose_mandate
		}
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			add_mandate = -12
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			add_mandate = -6
		}
		else = {
			add_mandate = -3
		}
	}
	
	chance = {
		factor = 1
	}
	
	ai_will_do = {
		factor = 1
	}
}
