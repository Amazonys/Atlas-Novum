grant_local_autonomy = {
	trigger = {
		FROM = { is_capital = no }
		FROM = { NOT = { local_autonomy = 80 } }
	}
	
	effect = {
		if = {
			limit = { halved_parliament_bribes = yes }
			FROM = { add_local_autonomy = 15 }
		}
		else = {
			FROM = { add_local_autonomy = 30 }
		}
	}
	
	ai_will_do = {
		factor = 1
	}	
}
