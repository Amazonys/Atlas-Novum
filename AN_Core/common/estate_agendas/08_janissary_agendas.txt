#Generic Agendas - always available

#This one or the next one will always be valid
estate_janissaries_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_janissaries
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_janissaries
			}
		}
		else = {
			employed_advisor = {
				category = MIL
			}
			is_in_deficit = no
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_janissaries
		}
		add_estate_loyalty = {
			estate = estate_janissaries
			loyalty = 10
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_janissaries
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#Give us three privileges
estate_janissaries_three_privileges = {
	can_select = {
		NOT = {
			num_of_estate_privileges = {
				estate = estate_janissaries
				value = 2
			}
		}
	}
	selection_weight = {
		factor = 10
	}
	pre_effect = {
		set_variable = {
			which = estate_janissaries_three_privileges_var
			value = 0
		}
		if = {
			limit = {
				num_of_estate_privileges = {
					estate = estate_janissaries
					value = 1
				}
			}
			set_variable = {
				which = estate_janissaries_three_privileges_var
				value = 2
			}
		}
		else = {
			set_variable = {
				which = estate_janissaries_three_privileges_var
				value = 1
			}
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_janissaries
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_janissaries
			}
		}
		else = {
			if = {
				limit = {
					check_variable = {
						which = estate_janissaries_three_privileges_var
						value = 2
					}
				}
				num_of_estate_privileges = {
					estate = estate_janissaries
					value = 3
				}
			}
			else = {
				num_of_estate_privileges = {
					estate = estate_janissaries
					value = 2
				}
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {

            estate = estate_janissaries
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5

#Retake Core
estate_janissaries_retake_core = {
	can_select = {
		is_free_or_tributary_trigger = yes
		OR = {
			is_part_of_hre = no
			NOT = { hre_reform_passed = landfriede }
		}
		any_province = {
			is_core = root
			NOT = { country_or_non_sovereign_subject_holds = root }
			is_city = yes
			NOT = { owner = { ROOT = { truce_with = prev } } }
		}
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 0.5
			any_province = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				owner = {
					NOT = { ROOT = { truce_with = prev } }
					alliance_with = root
				}
			}
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				NOT = { owner = { alliance_with = root } }
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				OR = {
					AND = {
						has_port = yes
						root = {
							any_owned_province = {
								has_port = yes
							}
						}
					}
					any_neighbor_province = {
						owned_by = root
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				OR = {
					AND = {
						has_port = yes
						root = {
							any_owned_province = {
								has_port = yes
							}
						}
					}
					any_neighbor_province = {
						owned_by = root
					}
				}
				NOT = { owner = { alliance_with = root } }
			}
			save_event_target_as = agenda_province
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { is_core = root }
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_janissaries
				}

				estate = estate_janissaries
			}
		}
		else = {
			event_target:agenda_province = {
				owned_by = root
			}
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_janissaries
		}
		add_estate_loyalty = {
			estate = estate_janissaries
			loyalty = 10
		}

#Support independence of X (subject of rival)
estate_janissaries_support_independence = {
	can_select = {
		OR = {
			has_dlc = "Conquest of Paradise"
			has_dlc = "El Dorado"
		}
		is_at_war = no
		is_subject = no
		any_rival_country = {
			NOT = { has_reform = daimyo }
			NOT = { ROOT = { truce_with = prev } }
			any_subject_country = {
				is_subject_other_than_tributary_trigger = yes
				liberty_desire = 50
				has_opinion = {
					who = root
					value = -100
				}
				ROOT = {
					NOT = { is_supporting_independence_of = prev }
				}
			}
		}
	}
	selection_weight = {
		factor = 2
	}
	pre_effect = {
		random_rival_country = {
			limit = {
				NOT = { ROOT = { truce_with = prev } }
				any_subject_country = {
					NOT = { has_reform = daimyo }
					is_subject_other_than_tributary_trigger = yes
					liberty_desire = 50
					has_opinion = {
						who = root
						value = -100
					}
					ROOT = {
						NOT = { is_supporting_independence_of = prev }
					}
				}
			}
			save_event_target_as = agenda_rival_country
			random_subject_country = {
				limit = {
					NOT = { has_reform = daimyo }
					is_subject_other_than_tributary_trigger = yes
					liberty_desire = 50
					has_opinion = {
						who = root
						value = -100
					}
					ROOT = {
						NOT = { is_supporting_independence_of = prev }
					}
				}
				save_event_target_as = agenda_country
			}
		}
	}
	fail_if = {
		OR = {
			is_subject = yes
			event_target:agenda_country = {
				OR = {
					NOT = { is_subject_of = event_target:agenda_rival_country }
					NOT = { num_of_cities = 1 }
					war_with = root
				}
			}
		}
	}
	provinces_to_highlight = {
		is_capital_of = event_target:agenda_country
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_janissaries
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_janissaries
			}
		}
		else = {
			is_supporting_independence_of = event_target:agenda_country
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_janissaries
		}
		add_estate_loyalty = {
			estate = estate_janissaries
			loyalty = 10
		}
		add_dip_power = 50
	}
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_janissaries
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
