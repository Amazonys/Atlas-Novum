#Generic Agendas - always available

#This one or the next one will always be valid
estate_qizilbash_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}
		}
		else = {
			employed_advisor = {
				category = MIL
			}
			is_in_deficit = no
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_qizilbash
			loyalty = 10
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_qizilbash
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

estate_qizilbash_fire_advisor = {

	}
	selection_weight = {
		factor = 1

		}

		hidden_effect = {
			set_country_flag = need_to_fire_mil_advisor
		}
	}

		clr_country_flag = need_to_fire_mil_advisor
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}

			custom_trigger_tooltip = {
				tooltip = fire_mil_advisor_tooltip

			}
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_qizilbash
			loyalty = 10
		}
		clr_country_flag = just_fired_mil_advisor

	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
		NOT = { has_country_flag = just_fired_mil_advisor }
	}

        add_estate_loyalty_modifier = {
            estate = estate_qizilbash
			desc = EST_VAL_AGENDA_DENIED

			duration = 7300

}

#Retake Core
estate_qizilbash_retake_core = {
	can_select = {
		is_free_or_tributary_trigger = yes
		OR = {
			is_part_of_hre = no
			NOT = { hre_reform_passed = landfriede }
		}
		any_province = {
			is_core = root
			NOT = { country_or_non_sovereign_subject_holds = root }
			is_city = yes
			NOT = { owner = { ROOT = { truce_with = prev } } }
		}
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 0.5
			any_province = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				owner = {
					NOT = { ROOT = { truce_with = prev } }
					alliance_with = root
				}
			}
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				NOT = { owner = { alliance_with = root } }
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				OR = {
					AND = {
						has_port = yes
						root = {
							any_owned_province = {
								has_port = yes
							}
						}
					}
					any_neighbor_province = {
						owned_by = root
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_province = {
			limit = {
				is_core = root
				NOT = { country_or_non_sovereign_subject_holds = root }
				is_city = yes
				NOT = { owner = { ROOT = { truce_with = prev } } }
				OR = {
					AND = {
						has_port = yes
						root = {
							any_owned_province = {
								has_port = yes
							}
						}
					}
					any_neighbor_province = {
						owned_by = root
					}
				}
				NOT = { owner = { alliance_with = root } }
			}
			save_event_target_as = agenda_province
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { is_core = root }
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}
		}
		else = {
			event_target:agenda_province = {
				owned_by = root
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_qizilbash
			loyalty = 10
		}
		add_country_modifier = {
			name = military_victory
			duration = 3650
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_qizilbash
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

#More Qizilbash Units
estate_qizilbash_recruit_more_qizilbash = {
	can_select = {
		NOT = { has_country_modifier = thriving_arms_industry }
		NOT = { army_size_percentage = 0.5 }
		
		if = {
			limit = {
				is_emperor = yes
			}
			NOT = { army_size_percentage = 0.35 }
		}
	}
	selection_weight = {
		factor = 5
		modifier = {
			factor = 2
			any_rival_country = {
				NOT = { truce_with = root }
				army_size = root
			}
		}
		modifier = {
			factor = 2.0
			NOT = { army_size_percentage = 0.3 }
			NOT = { army_size = 150 }
		}
		modifier = { #this could be really annoying in the late game
			factor = 0.5
			army_size = 100
		}
		modifier = {
			factor = 0.25
			army_size = 200
		}
		modifier = {
			factor = 0
			army_size = 500
		}
		modifier = {
			factor = 0.5
			is_emperor = yes
		}
		modifier = {
			has_estate_privilege = estate_janissary_allow_other_professions
			factor = 0
		}
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}
		}
		else = {
			if = {
				limit = {
					is_emperor = yes
				}
				army_size_percentage = 0.65
			}
			else = {
				army_size_percentage = 0.75
			}
		}
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_qizilbash
			loyalty = 10
		}
		define_advisor = {
			type = commandant
			cost_multiplier = 0.25
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_qizilbash
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

		has_dlc = "Golden Century"
		any_owned_province = {
			is_state_core = root
			has_any_holy_order = no
		}
		can_create_holy_orders = yes
	}
	selection_weight = {
		factor = 2.5
	}
	pre_effect = {
		random_owned_province = {
			limit = {
				is_state_core = root
				has_any_holy_order = no
			}
			save_event_target_as = agenda_province
			area = {
				set_permanent_province_flag = qizilbash_safavid_order_flag
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}
		}
		else = {
			if = {
				limit = {
					has_estate_agenda_auto_completion = {
						estate = estate_qizilbash
					}
				}
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			else = {
				event_target:agenda_province = {
					has_any_holy_order = yes
				}
			}
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_qizilbash
		}
		add_estate_loyalty = {
			estate = estate_qizilbash
			loyalty = 10
		}
		add_years_of_owned_provinces_manpower = {
			custom_tooltip = FROM_AGENDA_PROVINCE_AREA
			years = 4
			trigger = {
				has_province_flag = qizilbash_safavid_order_flag
			}
		}
		hidden_effect = {
			event_target:agenda_province = {
				area = { clr_province_flag = qizilbash_safavid_order_flag }
			}
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_qizilbash
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		hidden_effect = {
			event_target:agenda_province = {
				area = { clr_province_flag = qizilbash_safavid_order_flag }
			}
		}
    }
}

#Defeat Large Army
estate_qizilbash_defeat_large_army = {
	can_select = {
		is_bankrupt = no
		army_size = 20
	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
	}
	provinces_to_highlight = {
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}
		}
		else = {
			if = {
				limit = {
					current_age = age_of_revolutions
				}
				check_variable = {
					which = land_units_killed
					value = 30000
				}
			}
			else_if = {
				limit = {
					current_age = age_of_absolutism
				}
				check_variable = {
					which = land_units_killed
					value = 22500
				}
			}
			else_if = {
				limit = {
					current_age = age_of_reformation
				}
				check_variable = {
					which = land_units_killed
					value = 15000
				}
			}
			else = {
				check_variable = {
					which = land_units_killed
					value = 7500
				}
			}
		}
	}
	fail_if = {
		NOT = { army_size = 1 }
		is_at_war = no
	}
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_qizilbash
		}
		add_estate_loyalty = {
			estate = estate_qizilbash
			loyalty = 10
		}
		if = {
			limit = {
				current_age = age_of_revolutions
			}
			add_army_tradition_or_mil_power = {
				amount = 20
			}
		}
		else_if = {
			limit = {
				current_age = age_of_absolutism
			}
			add_army_tradition_or_mil_power = {
				amount = 15
			}
		}
		else_if = {
			limit = {
				current_age = age_of_reformation
			}
			add_army_tradition_or_mil_power = {
				amount = 15
			}
		}
		else = {
			add_army_tradition_or_mil_power = {
				amount = 10
			}
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_qizilbash
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

	}
	selection_weight = {
		factor = 1
	}
	pre_effect = {
	}
	provinces_to_highlight = {
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_qizilbash
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_qizilbash
			}
		}
		else = {
			cavalry_fraction = 0.2
		}
	}

		is_bankrupt = yes

			estate = estate_qizilbash
			loyalty = 10
		}
		add_country_modifier = {

		}

    failing_effect = {
		on_failed_agenda_effect = yes

        add_estate_loyalty_modifier = {

			desc = EST_VAL_AGENDA_DENIED

			duration = 7300
        }
    }
}
