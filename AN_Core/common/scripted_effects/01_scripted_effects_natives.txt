# Scripted effects for Native Americans


add_permanent_claims_for_federation_nation = {
	# ORIGINAL = Original Tag
	# TAG = New Tag
	capital_scope = {
		region = {
			limit = {
				NOT = { owned_by = $ORIGINAL$ }
			}
			add_permanent_claim = $TAG$
		}
	}
}

	}
