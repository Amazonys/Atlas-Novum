apply_global_demand_modifier_for_trade_good = {
	export_to_variable = {
		which = tempvar
		value = modifier:good_$trade_good_id$_demand_modifier
	}
	change_variable = {
		which = tempvar
		value = 1
	}
	multiply_variable = {
		which = stock_market_demand_for_$trade_good_id$
		which = tempvar
	}

		which = tempvar2
		value = modifier:local_development_cost_modifier 
		who = PREV
	}
	export_to_variable = {
		which = tempvar3
		value = modifier:development_cost_modifier 
	}
	
	
	change_variable = {
		which = tempvar3
		which = tempvar2
	}
	set_variable = {
		which = tempvar2
		value = 1
	}
	subtract_variable = {
		which = tempvar2
		which = tempvar3
	}
	multiply_variable = { 
		which = stock_market_demand_for_$trade_good_id$
		which = tempvar2
	}
