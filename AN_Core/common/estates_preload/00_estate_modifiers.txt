estate_brahmins = {
	modifier_definition = {
		type = loyalty
		key = brahmins_hindu_loyalty_modifier
		trigger = {
			has_estate = estate_brahmins
			religion = hinduism
		}
	}
	modifier_definition = {
		type = loyalty
		key = brahmins_muslim_loyalty_modifier
		trigger = {
			has_estate = estate_brahmins
			religion_group = muslim
		}
	}
	modifier_definition = {
		type = loyalty
		key = brahmins_other_loyalty_modifier
		trigger = {
			has_estate = estate_brahmins
			NOT = { religion = hinduism }
			NOT = { religion_group = muslim }
		}
	}
	modifier_definition = {
		type = influence
		key = brahmins_hindu_influence_modifier
		trigger = {
			has_estate = estate_brahmins
			religion = hinduism
		}
	}
	modifier_definition = {
		type = influence
		key = brahmins_muslim_influence_modifier
		trigger = {
			has_estate = estate_brahmins
			religion_group = muslim
		}
	}
	modifier_definition = {
		type = influence
		key = brahmins_other_influence_modifier
		trigger = {
			has_estate = estate_brahmins
			NOT = { religion = hinduism }
			NOT = { religion_group = muslim }
		}
	}

	modifier_definition = {
		type = privileges
		key = brahmins_privilege_slots
		trigger = {
			has_estate = estate_brahmins
		}
	}
}
