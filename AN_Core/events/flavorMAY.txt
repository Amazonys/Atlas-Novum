# Popol Vuh
country_event = {
	id = flavor_may.1
	title = flavor_may.1.t
	desc = flavor_may.1.d
	picture = BIG_BOOK_eventPicture
	
	fire_only_once = yes

	trigger = {
		OR = {
			tag = MAY
			tag = KIC
			tag = ITZ
			tag = XIU
			tag = COC
		}
		religion_group = christian
		any_owned_province = {
			NOT = {
				religion_group = christian
			}
		}
		NOT = {
			has_country_modifier = may_popol_vuh
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = flavor_may.1.a
		add_country_modifier = {
			name = "may_popol_vuh"
			duration = 3650
		}
	}
	option = {
		name = flavor_may.1.b
		add_prestige = 5
		add_papal_influence = 2
	}
}

# League of Mayapan
country_event = {
	id = flavor_may.2
	title = flavor_may.2.t
	desc = flavor_may.2.d
	picture = CIVIL_WAR_eventPicture
	
	trigger = {
		religion = mesoamerican_religion
		NOT = { has_country_flag = may_league_mayapan_flag }
	}
	
	mean_time_to_happen = {
		months = 22
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = may_league_mayapan_flag
		}
	}
	
	option = {
		name = flavor_may.2.a
		add_prestige = 10
		every_country = {
			limit = {
				religion = mesoamerican_religion
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = stronger_alone }
		}
	}
	option = {
		name = flavor_may.2.b
		every_country = {
			limit = {
				religion = mesoamerican_religion
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = league_mayapan }
		}	
	}
}

# Kan Ek and the Prophecy
country_event = {
	id = flavor_may.3
	title = flavor_may.3.t
	desc = flavor_may.3.d
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	fire_only_once = yes

	trigger = {
		religion = mesoamerican_religion
		NOT = {
			religion = catholic
		}
		knows_country = SPA
		SPA = {
			religion = catholic
			knows_country = ROOT
			any_owned_province = {
				OR = {
					region = carribeans_region
					region = colombia_region
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 32
	}
	
	option = {
		name = flavor_may.3.a
		change_religion = catholic
		add_stability = -1
		SPA = {
			add_opinion = { who = ROOT modifier = may_convert }
			country_event = {
				id = flavor_may.4
				days = 7
				tooltip = flavor_may.4.tt
			}
		}
	}
	option = {
		name = flavor_may.3.b
		add_prestige = 5
		set_country_flag = may_kan_ek_flag
	}
}

# Maya Sacrifice
country_event = {
	id = flavor_may.6
	title = flavor_may.6.t
	desc = flavor_may.6.d
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	trigger = {
		NOT = { has_country_flag = maya_sacrifice_flag }
		religion = mesoamerican_religion
		war_with = SPA
		SPA = {
			religion = catholic
			OR = {
				any_owned_province = {
					any_neighbor_province = {
						owner = {
							religion = mesoamerican_religion
						}
					}
				}
				ROOT = {
					had_country_flag = { flag = may_kan_ek_flag days = 7300 }
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 32
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = maya_sacrifice_flag
		}
	}
	
	option = {
		name = flavor_may.6.a
		change_religion = catholic
		add_stability = -1
		SPA = {
			add_opinion = { who = ROOT modifier = may_convert }
			country_event = {
				id = flavor_may.4
				days = 7
				tooltip = flavor_may.4.tt
			}
		}
	}
	option = {
		name = flavor_may.6.b
		add_prestige = 10
		tooltip = {
			random_owned_province = {
				limit = {
					OR = {
						has_port = yes
						any_neighbor_province = {
							owned_by = SPA
						}
					}
				}
				add_claim = SPA
			}
		}
		SPA = {
			country_event = {
				id = flavor_may.7
				days = 7
				tooltip = flavor_may.7.tt
			}
		}
	}
}
