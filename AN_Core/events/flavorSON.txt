# Rise of Askia Muhammed
country_event = {
	id = flavor_son.4
	title = flavor_son.4.t
	desc = flavor_son.4.d
	picture = ASKIA_eventPicture
	fire_only_once = yes

	trigger = {
		tag = SON
		is_year = 1480
		NOT = { is_year = 1525 }
		OR = {
			AND = {
				NOT = { adm = 3 }
				NOT = { dip = 3 }
				NOT = { mil = 3 }
			}
			dynasty = "Sonni"
			NOT = { legitimacy = 50 }
		}
	}

	mean_time_to_happen = {
		months = 200
		modifier = {
			factor = 0.1
			is_year = 1490
		}
	}

	option = { 
		name = flavor_son.4.a
		add_stability = -1
		#kill_heir = yes
		define_ruler = {
			name = "Muhammad"
			dynasty = "Askia"
			min_age = 25
			max_age = 40
			claim = 75
			adm = 6
			dip = 3
			mil = 6
			culture = soninke
			religion = sunni
		}
		hidden_effect = {
			define_ruler_to_general = { fire = 3 shock = 4 manuever = 4 siege = 1 trait = master_of_arms_personality }
			set_ruler_flag = son_askia_the_great
		}

		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				OR = {
					adm = 5
					dip = 5
					mil = 5
				}
			}
		}
	}

	option = { 
		name = flavor_son.4.b

				type = pretender_rebels

			}
		}

		ai_chance = {
			factor = 100
			modifier = {
				factor = 0
				NOT = { adm = 5 }
				NOT = { dip = 5 }
				NOT = { mil = 5 }
			}
		}
	}	
}

# Increase goods of all salt and cloth provinces

		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		every_owned_province = {
			limit = {
				OR = {
					trade_goods = cloth
					trade_goods = salt
				}
				NOT = { has_province_modifier = son_salt_cloth_production }
			}

				name = son_salt_cloth_production
				duration = -1

# Remove all the province modifeirs when invalid
country_event = {
	id = flavor_son.101
	title = none
	desc = none
	picture = none
	hidden = yes

	trigger = {
		OR = {
			AND = {
				NOT = { has_country_modifier = son_salt_cloth_trader }
				any_owned_province = {
					has_province_modifier = son_salt_cloth_production
				}
			}
			AND = {
				has_country_modifier = son_salt_cloth_trader
				any_owned_province = {
					has_province_modifier = son_salt_cloth_production
					NOT = { trade_goods = cloth }
					NOT = { trade_goods = salt }
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		if = {
			limit = {
				has_country_modifier = son_salt_cloth_trader
			}
			every_owned_province = {
				limit = {
					has_province_modifier = son_salt_cloth_production
					NOT = { trade_goods = cloth }
					NOT = { trade_goods = salt }
				}
				remove_province_modifier = son_salt_cloth_production
			}
		}
		else = {
			every_owned_province = {
				limit = {
					has_province_modifier = son_salt_cloth_production
				}
				remove_province_modifier = son_salt_cloth_production
			}
		}

		ai_chance = {
			factor = 1
		}
	}
}
