# Respect our flag
country_event = {
	id = trade.1
	title = trade.1.t
	desc = trade.1.d
	picture = NAVAL_MILITARY_eventPicture
	
	is_triggered_only = yes
	trigger = {
		num_of_merchants = 3
		capital_scope = {
			has_trader = root
			NOT = { is_strongest_trade_power = root }
		}
		
		has_idea_group = maritime_ideas
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 1.5
			trade_ideas = 1
		}
		modifier = {
			factor = 3
			trade_ideas = 3
		}
		modifier = {
			factor = 5
			trade_ideas = 7
		}
	}
	
	option = {
		name = trade.1.a
		add_prestige = -30
		random_active_trade_node = {
			limit = {
				any_trade_node_member_province = {
					owned_by = ROOT
					is_capital = yes
				}
			}
			strongest_trade_power = { 
				reverse_add_casus_belli = {
					target = ROOT
					type = cb_trade_war_triggered
					months = 60
				}
			}
		}
	}
	option = {
		name = trade.1.b
		add_stability = -1
	}
}

country_event = {
	id = 9520
	title = "EVTNAME9520"
	desc = "EVTDESC9520"
	picture = TRADEGOODS_eventPicture

	trigger = {
		has_country_modifier = the_abolish_slavery_act
		any_owned_province = { trade_goods = slaves }
	}

	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = "EVTOPTA9520"
		every_owned_province = {
			limit = { trade_goods = slaves }
			change_trade_goods = random
			remove_building = tradecompany
			if = {
				limit = { has_province_modifier = slave_entrepot }
				remove_province_modifier = slave_entrepot
			}
		}
	}
}
