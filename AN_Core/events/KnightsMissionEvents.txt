# Restoration of the Hospitalers
country_event = {
	id = knights_missions.1
	title = knights_missions.1.t
	desc = {
		trigger = {
			tag = KNI
		}
		desc = knights_missions.1.d
	}
	desc = {
		trigger = {
			NOT = { tag = KNI }
		}
		desc = knights_missions.1.d2
	}
	picture = RELIGIOUS_WARS_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = "knights_missions.1.a"
		trigger = {
			NOT = { tag = KNI }
			OR = {
				NOT = { exists = KNI }
				KNI = {
					exists = yes
					ai = yes
					religion = catholic
					NOT = {	num_of_cities = 3 }
					NOT = { is_rival = ROOT }
					NOT = { war_with = ROOT }
					OR = {
						is_subject = no
						is_subject_of = ROOT
					}
				}
			}
		}
		2313 = {
			add_core = KNI
			cede_province = KNI
		}
		if = {
			limit = {
				NOT = {
					KNI = { is_subject_of = ROOT }
				}
			}
			create_march = KNI
		}
		KNI = {
			add_country_modifier = {
				name = "restored_hospitaler_knights"
				duration = -1
			}
		}
	}
	option = {
		name = "knights_missions.1.b"
		trigger = {
			tag = KNI
		}
		add_country_modifier = {
			name = "restored_hospitaler_knights"
			duration = -1
		}
	}
	option = {
		name = "knights_missions.1.c"
		trigger = {
			NOT = { tag = KNI }
		}
		add_prestige = 10
		add_papal_influence = 10
		add_army_tradition = 10
	}
}

	title = knights_missions.3.t
	desc = knights_missions.3.d
	picture = RELIGIOUS_WARS_eventPicture 
	
	fire_only_once = yes
	is_triggered_only = yes
	
	option = {
		name = "knights_missions.3.a"
		if = {
			limit = {
				NOT = { has_reform = monastic_order_reform }
				NOT = { government_rank = 3 }
			}
			set_government_rank = 3
		}
		else = {
			add_prestige = 50
		}
	}
}

# The Ark of the Covenant
country_event = {
	id = knights_missions.4
	title = knights_missions.4.t
	desc = knights_missions.4.d
	picture = DEATH_OF_HEIR_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes

	option = { # open the Ark
		name = "knights_missions.4.a"
		capital_scope = {
			add_province_modifier = {
				name = "kni_ark_of_the_covenant"
				duration = -1
			}
		}
		if = {
			limit = {
				OR = {
					ruler_has_personality_ancestor = { key = just }
					ruler_has_personality_ancestor = { key = righteous }
					ruler_has_personality_ancestor = { key = lawgiver }
				}
			}
			random_list = {
				25 = {
					kill_ruler = yes
				}
				75 = {
					add_country_modifier = {
						name = "kni_ten_commandments"
						duration = -1
					}
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					ruler_has_personality = sinner_personality
					ruler_has_personality = cruel_personality
					ruler_has_personality = malevolent_personality
				}
			}
			random_list = {
				75 = {
					kill_ruler = yes
				}
				25 = {
					add_country_modifier = {
						name = "kni_ten_commandments"
						duration = -1
					}
				}
			}
		}
		else = {
			random_list = {
				50 = {
					kill_ruler = yes
				}
				50 = {
					add_country_modifier = {
						name = "kni_ten_commandments"
						duration = -1
					}
				}
			}
		}
	}
	option = { # step away
		name = "knights_missions.4.b"
		capital_scope = {
			add_province_modifier = {
				name = "kni_ark_of_the_covenant"
				duration = -1
			}
		}
	}
}
