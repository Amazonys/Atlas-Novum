# Discontent Among Mercenaries
country_event = {
	id = mercs.1
	title = mercs.1.t
	desc = mercs.1.d
	picture = SCOTTISH_MERCENARIES_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		any_hired_mercenary_company = {
			is_foreign_company = yes
			hired_for_months = 3
			home_province = {
				NOT = { unrest = 3 }
				owner = {
					war_with = ROOT
				}
			}
			location = {
				owner = {
					war_with = ROOT
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_hired_mercenary_company = {
				limit = {
					is_foreign_company = yes
					hired_for_months = 3
					home_province = {
						NOT = { unrest = 3 }
						owner = {
							war_with = ROOT
						}
					}
					location = {
						owner = {
							war_with = ROOT
						}
					}
				}
				save_event_target_as = patriotic_mercs
				home_province = {
					save_event_target_as = merc_home_province
					owner = {
						save_event_target_as = merc_home_country
					}
				}
			}
		}
	}
	
	option = {
		name = mercs.1.a
		event_target:patriotic_mercs = {
			add_company_manpower = -0.25
			hidden_effect = {
				kill_mercenary_leader = THIS
			}
			custom_tooltip = patriotic_company_leader_deserts
		}
	}
	option = { # all Scots are Jake
		name = mercs.1.b
		highlight = yes
		trigger = {
			event_target:patriotic_mercs = {
				home_province = {
					OR = {
						culture = highland_scottish
						culture = scottish
					}
				}
			}
		}
		add_treasury = -5
	}
}
