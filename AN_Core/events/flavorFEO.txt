	}
	option = {	
		name = "flavor_feo.4.b"
		
		ai_chance = { factor = 4 }
		
		add_country_modifier = {
			name = feo_pricipate_guard_payments_secured
			duration = 3650
		}
		if = {
			limit = {
				has_dlc = "Cradle of Civilization"
			}
			add_army_professionalism = 0.05
		}
		else = {
			add_army_tradition = 10
		}
	}
	option = {	
		name = "flavor_feo.4.c"
		
		ai_chance = { factor = 2 }
		
		capital_scope = {
			add_devastation = -100
			if = {
				limit = {
					has_dlc = "Mandate of Heaven"
				}
				add_prosperity = 100
			}
			else = {
				add_base_manpower = 1
				add_base_production = 1
				add_base_tax = 1
			}
		}
	}
}
