			FUL = {
				define_ruler = {
					name = "Koli Tengella"
					dynasty = "Denianke"

					limit = {
						has_dlc = "Rights of Man" 
					}

				}
				adopt_reform_progress = ROOT
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.7

					}
