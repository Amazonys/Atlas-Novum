# The Marathi Revolt
country_event = {
	id = flavor_mug.1
	title = "flavor_mug.EVTNAME1"
	desc = "flavor_mug.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = MUG
		religion_group = muslim
		is_year = 1650
		NOT = { exists = MAR } # Marathas
		any_owned_province = {
			OR = {
					area = maidan_area
					area = desh_area
					culture = marathi
				}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_mug.EVTOPTA1" # Crush the rebels
		every_owned_province = {
			limit = {
				OR = {
					area = maidan_area
					area = desh_area
					culture = marathi
				}
				is_capital = no
			}
			add_core = MAR
			add_province_modifier = {
				name = add_unrest_10_modifier
				duration = 1825
			}
		}
	}
}
