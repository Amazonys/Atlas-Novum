# Flavour events for Pegu by Daniel Tolman

namespace = flavor_peg
normal_or_historical_nations = yes


# Shin Sawbu
country_event = {
	id = flavor_peg.1
	title = flavor_peg.1.t
	desc = {	
		trigger = {
			has_regency = yes
		}
		desc = flavor_peg.1.d1
	}
	desc = {	
		trigger = {
			has_heir = no
		}
		desc = flavor_peg.1.d2
	}
	picture = NUSANTARA_QUEEN_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = PEG
		government = monarchy
		NOT = { is_year = 1471 }
		OR = {
			has_regency = yes
			has_heir = no
		}
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = "flavor_peg.1.a"
		if = {
			limit = {
				has_heir = no
			}
			define_ruler = {
				name = "Shin Sawbu"
				dynasty = "Wareru"
				regency = no
				female = yes
				age = 60
				claim = 100
				adm = 5
				dip = 4
				mil = 2
			}
		}
		else = {
			define_ruler = {
				name = "Shin Sawbu"
				dynasty = "Wareru"
				regency = yes
				adm = 5
				dip = 4
				mil = 2
				female = yes
				age = 60
			}
		}
	}
	option = {
		name = flavor_peg.1.b
		add_prestige = -10
	}
}

# Dhammazedi
country_event = {
	id = flavor_peg.2
	title = flavor_peg.2.t
	desc = flavor_peg.2.d
	picture = NUSANTARA_QUEEN_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = PEG
		NOT = { is_year = 1492 }
		OR = {
			has_ruler = "Shin Sawbu"
			has_regency = yes
		}
		has_heir = no
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	option = { # Retire the throne to Dhammazedi now
		name = flavor_peg.2.a
		define_ruler = {
			name = "Dhammazedi"
			dynasty = "Wareru"
			adm = 5
			dip = 4
			mil = 3
			min_age = 35
		}
		define_heir = {
			name = "Binnya Ran"
			dynasty = "Wareru"
			claim = 70
			birth_date = 1409.1.1
			adm = 3
			dip = 4
			mil = 5
		}
	}
	option = { # Set Dhammazedi as heir
		name = flavor_peg.2.b
		define_heir = {
			name = "Dhammazedi"
			dynasty = "Wareru"
			female = yes
			claim = 100
			birth_date = 1409.1.1
			adm = 4
			dip = 4
			mil = 3
		}
	}
	option = { # ignore him
		name = flavor_peg.2.c
		add_prestige = -10
	}
}
