	desc = siamese_rev.1.d
	picture = PORTUGUESE_MERCHANTS_eventPicture

	fire_only_once = yes

		hidden_effect = {
			set_country_flag = siam_revolution_start_flag

				limit = {

						}

						superregion = east_indies_superregion
					}
				}

			}
		}
	}
	
	option = {
		name = siamese_rev.1.a
		add_stability = -2
	}
}
