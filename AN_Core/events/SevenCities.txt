# El Dorado
province_event = {
	id = seven_cities_events.1
	title = seven_cities_events.1.t
	desc = seven_cities_events.1.d
	picture = LAKE_TREASURE_eventPicture

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "El Dorado"
		FROM = {
			NOT = {
				OR = {
					has_country_flag = sce_eventchain_flag
					has_country_flag = sce_end_eldorado_flag
				}
			}
		}
		OR = {
			continent = new_world
			continent = south_america
			any_neighbor_province = {
				continent = south_america
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		
		modifier = {
			factor = 2.0
			is_empty = no
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_eldorado
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_fountainofyouth
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_cibola
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_cityofcaesars
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_sierradelaplata
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_norumbega
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_saguenay
		}
		modifier = {
			factor = 0.5
			has_global_flag = sce_ultimate_quivira
		}
	}
	
	option = {
		name = seven_cities_events.1.a
		FROM = {
			set_country_flag = sce_eventchain_flag
			add_country_modifier = {
				name = sce_search_eldorado
				duration = -1
			}
		}
	}
	option = {
		name = seven_cities_events.1.b
		FROM = {
			set_country_flag = sce_end_eldorado_flag
			add_adm_power = 50
		}
	}

# Explorer dies
province_event = {
	id = seven_cities_events.100
	title = seven_cities_events.100.t
	desc = seven_cities_events.100.d
	picture = GUARDED_TREASURE_eventPicture

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "El Dorado"
		is_empty = yes
		FROM = {
			has_country_flag = sce_eventchain_flag
			NOT = {
				has_country_flag = sce_action01_flag
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			FROM = {
				set_country_flag = sce_action01_flag
				clr_country_flag = sce_eventchain_flag
				
				remove_country_modifier = sce_search_eldorado
	 			remove_country_modifier = sce_search_fountainofyouth
	 			remove_country_modifier = sce_search_cibola
	 			remove_country_modifier = sce_search_cityofcaesars
	 			remove_country_modifier = sce_search_sierradelaplata
	 			remove_country_modifier = sce_search_norumbega
				remove_country_modifier = sce_search_saguenay
				remove_country_modifier = sce_search_quivira
			}
			kill_leader = { 
				type = conquistador
			}
		}
	}
	
	option = {
		name = seven_cities_events.100.a
	}
}
