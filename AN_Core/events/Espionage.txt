# Half the Loyalty, Double the Agent
country_event = {
	id = espionage_events.1
	title = "espionage_events.1.t"
	desc = "espionage_events.1.d"
	picture = SPY_eventPicture
	goto = espionage_event_country_capital

	trigger = {
		has_dlc = "Mare Nostrum"
		any_known_country = {
			has_spy_network_in = {
				who = ROOT
				value = 25
			}
			has_spy_network_from = {
				who = ROOT
				value = 25
			}
			corruption = 10
		}
	}

	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			capital_scope = {
				save_event_target_as = espionage_event_country_2_capital
			}
			random_known_country = {
				limit = {
					has_spy_network_in = {
						who = ROOT
						value = 25
					}
					has_spy_network_from = {
						who = ROOT
						value = 25
					}
					corruption = 10
				}
				save_event_target_as = espionage_event_country
				capital_scope = {
					save_event_target_as = espionage_event_country_capital
				}
			}
		}
	}

	# Recruit the spy
	option = {
		name = "espionage_events.1.a"
		add_treasury = -15
		event_target:espionage_event_country = {
			add_spy_network_from = {
				who = ROOT
				value = 30
			}
			random = {
				chance = 30
				country_event = {
					id = espionage_events.2 # Enemy discovers your plot
					days = 120
				}
			}
		}
	}

	# Execute the spy
	option = {
		name = "espionage_events.1.b"
		event_target:espionage_event_country = {
			add_spy_network_in = {
				who = ROOT
				value = -25
			}
		}
	}
}

	id = espionage_events.3
	title = "espionage_events.3.t"
	desc = "espionage_events.3.d"
	picture = DIPLOMAT_INTERROGATION_eventPicture
	goto = espionage_event_country_capital

	is_triggered_only = yes

	# Kill the agent
	option = {
		name = "espionage_events.3.a"
		custom_tooltip = espionage_events.3.tt_1
		tooltip = {
			add_spy_network_from = {
				who = FROM
				value = -25
			}
		}
		custom_tooltip = espionage_events.3.tt_2
		tooltip = {
			add_spy_network_in = {
				who = FROM
				value = -25
			}
		}
		hidden_effect = {
			if = {
				limit = { has_country_flag = re_doubled_agent }
				add_spy_network_from = {
					who = FROM
					value = -25
				}
			}
			if = {
				limit = { NOT = { has_country_flag = re_doubled_agent } }
				add_spy_network_in = {
					who = FROM
					value = -25
				}
			}
			clr_country_flag = re_doubled_agent
		}
	}

	# Continue operations
	option = {
		name = "espionage_events.3.b"
		custom_tooltip = espionage_events.3.tt_1
		tooltip = {
			add_spy_network_from = {
				who = FROM
				value = 15
			}
		}
		custom_tooltip = espionage_events.3.tt_2
		tooltip = {
			add_spy_network_in = {
				who = FROM
				value = 15
			}
		}
		hidden_effect = {

					who = FROM

				}
			}
			clr_country_flag = re_doubled_agent
		}
	}

			}

		}
	}
	
	immediate = {
		hidden_effect = {
			random_known_country = {
				limit = {
					has_spy_network_from = {
						who = ROOT
						value = 10
					}
					religion = catholic
				}
				save_event_target_as = victim_country
				capital_scope = {
					save_event_target_as = victim_country_capital
				}
			}
		}
	}
	
	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
	}

	# 
	option = {
		name = "espionage_events.4.a"
		add_spy_network_in = {
			who = event_target:victim_country
			value = 25
		}
		if = {
			limit = {
				NOT = { TAG = PAP }
			}
			random = {
				chance = 30
				country_event = {
					id = espionage_events.5 # Pope discovers your plot
					days = 60
				}
			}
		}
	}

	# 
	option = {
		name = "espionage_events.4.b"
		add_dip_power = 10
	}
}

# $SPYMASTER$ Suspects Neighbors
country_event = {
	id = espionage_events.8
	title = "espionage_events.8.t"
	desc = "espionage_events.8.d"
	picture = SPY_eventPicture

	trigger = {
		has_dlc = "Mare Nostrum"
		advisor = spymaster
		any_known_country = {
			is_neighbor_of = ROOT
			has_spy_network_in = {
				who = ROOT
				value = 25
			}
		}
		any_known_country = {
			is_neighbor_of = ROOT
			NOT = {
				has_spy_network_in = {
					who = ROOT
					value = 1
				}
			}
		}
	}

	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			random_list = {
				50 = {
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							has_spy_network_in = {
								who = ROOT
								value = 25
							}
						}
						save_event_target_as = neighbor_1
						capital_scope = {
							save_event_target_as = neighbor_1_capital
						}
					}
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							NOT = {
								has_spy_network_in = {
									who = ROOT
									value = 1
								}
							}
						}
						save_event_target_as = neighbor_2
						capital_scope = {
							save_event_target_as = neighbor_2_capital
						}
					}
				}
				50 = {
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							has_spy_network_in = {
								who = ROOT
								value = 25
							}
						}
						save_event_target_as = neighbor_2
						capital_scope = {
							save_event_target_as = neighbor_2_capital
						}
					}
					random_known_country = {
						limit = {
							is_neighbor_of = ROOT
							NOT = {
								has_spy_network_in = {
									who = ROOT
									value = 1
								}
							}
						}
						save_event_target_as = neighbor_1
						capital_scope = {
							save_event_target_as = neighbor_1_capital
						}
					}
				}
			}
		}
	}

	# 
	option = {
		name = "espionage_events.8.a"
		goto = neighbor_1_capital
		event_target:neighbor_1 = {
			add_opinion = {
				who = ROOT
				modifier = counterintelligence_operation
			}
			hidden_effect = {
				if = {
					limit = {
						has_spy_network_in = {
							who = ROOT
							value = 25
						}
					}
					add_spy_network_in = {
						who = ROOT
						value = -25
					}
				}
			}
		}
	}

	# 
	option = {
		name = "espionage_events.8.b"
		goto = neighbor_2_capital
		event_target:neighbor_2 = {
			add_opinion = {
				who = ROOT
				modifier = counterintelligence_operation
			}
			hidden_effect = {
				if = {
					limit = {
						has_spy_network_in = {
							who = ROOT
							value = 25
						}
					}
					add_spy_network_in = {
						who = ROOT
						value = -25
					}
				}
			}
		}
	}
