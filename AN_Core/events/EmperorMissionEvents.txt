# The Geneva Conventions
country_event = {
	id = emp_mission_events.3
	title = emp_mission_events.3.t
	desc = emp_mission_events.3.d
	picture = RELIGIOUS_CONVERSION_eventPicture
	
	is_triggered_only = yes
	major = yes
	
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					OR = {
						religion = protestant
						religion = reformed
					}
					capital_scope = {
						continent = europe
					}
				}
				save_event_target_as = geneva_conventions_1
				set_country_flag = geneva_conventions_target
			}
			random_country = {
				limit = {
					OR = {
						religion = protestant
						religion = reformed
					}
					capital_scope = {
						continent = europe
					}
					NOT = { has_country_flag = geneva_conventions_target }
					has_regency = no
				}
				set_country_flag = geneva_conventions_target
				save_event_target_as = geneva_conventions_2
			}
			random_country = {
				limit = {
					OR = {
						religion = protestant
						religion = reformed
					}
					capital_scope = {
						continent = europe
					}
					NOT = { has_country_flag = geneva_conventions_target }
				}
				save_event_target_as = geneva_conventions_3
			}
		}
	}
	
	option = {
		name = emp_mission_events.3.a
		custom_tooltip = geneva_religious_center
		4720 = {
			add_province_triggered_modifier = religious_center
		}
		every_country = {
			limit = {
				OR = {
					religion = protestant
					religion = reformed
					religion = anglican
					religion = hussite
				}
				capital_scope = {
					continent = europe
				}
				NOT = { tag = SWI }
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_geneva_conventions
			}
		}
	}
}

# The Duke of Alençon by Daniel Tolman

country_event = {
	id = emp_mission_events.10
	title = emp_mission_events.10.t
	desc = "emp_mission_events.10.d"
	picture = REFORM_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = FRA
		has_estate_privilege = estate_nobles_french_strong_duchies
		owns = 1879
		1879 = { is_capital = no }
		is_at_war = no
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					NOT = {
						1879 = {
							NOT = { is_core = ALE }
						}
					}
				}
				1879 = { add_core = ALE	}
			}
		}
	}
	
	option = { # restore the duke
		name = emp_mission_events.10.a
		hidden_effect = {
			release = ALE
		}
		if = {
			limit = { has_dlc = "Domination" } 
			create_subject = {
				subject_type = appanage
				subject = ALE 
			}
		}
		else = { vassalize = ALE } 
		hidden_effect = {
			ALE = {
				change_government = monarchy
				if = { 
					limit = { 
						FRA = { has_reform = feudal_france_reform }
					}
					add_government_reform = appanage_reform
				}
				else = { add_government_reform = feudalism_reform }
			}
		}
		if = {
			limit = {
				NOT = {
					is_year = 1476
				}
			}
			ALE = {
				define_ruler = {
					name = "Jean II"
					dynasty = "de Valois"
				}
			}
		}
		else = {
			ALE = {
				define_ruler = {
					dynasty = "de Valois"
				}
			}
		}
		if = {
			limit = {
				1879 = {
					is_core = FRA
				}
			}
			1879 = { remove_core = FRA }
		}
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_DUKE_OF_ALENCON
			influence = 10
			duration = 7300
		}
	}
	
	option = {
		name = emp_mission_events.10.b
		ai_chance = { factor = 0 }
		add_stability = -1
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -20
		}
	}
}
