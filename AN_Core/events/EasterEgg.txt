# Delicious Tortoise

namespace = tortoise

country_event = {
	id = tortoise.1
	title = tortoise.1.t
	desc = tortoise.1.d
	picture = FARMING_eventPicture
	goto = 2002
	
	is_triggered_only = yes
	
	trigger = {
		2002 = {
			trade_goods = livestock
			owner = {
				ai = no
				tag = ROOT
			}
			NOT = { has_province_modifier = delicious_tortoises }
		}
	}
	
	option = {
		name = tortoise.1.a
		2002 = {
			add_province_modifier = { name = delicious_tortoises duration = -1 }
		}
	}
}

namespace = dodo
province_event = {
	id = dodo.1
	title = dodo.1.name
	desc = dodo.1.desc
	picture = FARMING_eventPicture
	is_triggered_only = yes
	fire_only_once = yes
	goto = root

	trigger = {
		province_id = 1102
		owner = {
			natural_scientist = 3
			capital_scope = {
				continent = europe
			}
		}
		NOT = { colonysize = 800 }
	}

	option = {
		name = dodo.1.A
		owner = { add_adm_power = -30 }
		custom_tooltip = dodo.1.A.tt
		hidden_effect = {
			owner = {
				random_list = {
					2 = {
						country_event = { id = dodo.2 days = 180 }
					}
					1 = {
						country_event = { id = dodo.3 days = 180 }
					}
				}
			}
		}
	}
	option = {
		name = dodo.1.B
		add_province_modifier = {
			name = edible_dodos
			duration = 3650
		}
	}
}
