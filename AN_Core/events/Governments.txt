# Election of a new King
country_event = {
	id = government_events.3
	title = government_events.3.t
	desc = government_events.3.desc
	picture = COURT_eventPicture
	trigger = {
		has_government_attribute = has_ruler_election
		NOT = { is_subject_of_type = personal_union }
		NOT = { has_country_flag = had_election_event }
	}
	is_triggered_only = yes
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			set_country_flag = had_election_event
			select_biggest_country_of_scope_type = {
				limit = "
					has_regency = no
					is_free_or_tributary_trigger = yes
					OR = {
						has_estate = estate_nobles
						has_estate = estate_brahmins
						has_estate = estate_rajput
					}
					government = monarchy
					NOT = { war_with = ROOT }
					if = {
						limit = {
							ROOT = { religion_group = christian }
						}
						religion_group = christian
					}
				"
				trigger_scope = "any_neighbor_country"
				effect_scope = "random_neighbor_country"
			}
		}
	}
	after = {
		clr_country_flag = had_election_event
		if = {
			limit = {
				has_country_flag = has_internal_election_first
			}
			clr_country_flag = has_internal_election_first
			country_event = {
				id = 900
			}
		}
	}

	#Keep the current ruler
	option = {
		name = government_events.3.a
		trigger = {
			has_regency = no
		}
		add_legitimacy = 5
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { adm = 3 }
				NOT = { dip = 3 }
				NOT = { mil = 3 }
			}
			modifier = {
				factor = 0
				OR = {
					AND = {
						NOT = { adm = 4 }
						NOT = { dip = 2 }
						NOT = { mil = 2 }
					}
					AND = {
						NOT = { adm = 2 }
						NOT = { dip = 4 }
						NOT = { mil = 2 }
					}
					AND = {
						NOT = { adm = 2 }
						NOT = { dip = 2 }
						NOT = { mil = 4 }
					}
				}
			}
		}
	}

	#The heir shall rule us!
	option = {
		name = government_events.3.aa
		trigger = {
			has_regency = yes
			has_heir = yes
		}
		add_heir_claim = 10
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { heir_adm = 3 }
				NOT = { heir_dip = 3 }
				NOT = { heir_mil = 3 }
			}
			modifier = {
				factor = 0
				OR = {
					AND = {
						NOT = { heir_adm = 4 }
						NOT = { heir_dip = 2 }
						NOT = { heir_mil = 2 }
					}
					AND = {
						NOT = { heir_adm = 2 }
						NOT = { heir_dip = 4 }
						NOT = { heir_mil = 2 }
					}
					AND = {
						NOT = { heir_adm = 2 }
						NOT = { heir_dip = 2 }
						NOT = { heir_mil = 4 }
					}
				}
			}
		}
	}

	#Get a foreign noble
	option = {
		name = government_events.3.b
		trigger = {
			has_saved_event_target = selected_biggest_country
		}
		if = {
			limit = { has_government_attribute = has_improved_ruler_election }
			define_ruler = {
				change_adm = -1
				change_dip = -1
				change_mil = -1
				dynasty = event_target:selected_biggest_country
				culture = event_target:selected_biggest_country
				religion = event_target:selected_biggest_country
				claim = 75
				hide_skills = yes
			}
		}
		else = {
			define_ruler = {
				change_adm = -2
				change_dip = -2
				change_mil = -2
				dynasty = event_target:selected_biggest_country
				culture = event_target:selected_biggest_country
				religion = event_target:selected_biggest_country
				claim = 50
				hide_skills = yes
			}
		}
		kill_heir = { allow_new_heir = no }
		
		ai_chance = {
			factor = 1
		}
	}

	#Get our own noble
	option = {
		name = government_events.3.c
		trigger = {
			OR = {
				has_estate = estate_nobles
				has_estate = estate_maratha
				has_estate = estate_rajput
				AND = {
					has_regency = yes
					has_heir = no
				}
			}
		}
		if = {
			limit = { has_government_attribute = has_improved_ruler_election }
			define_ruler = {
				claim = 75
			}
		}
		else = {
			define_ruler = {
				change_adm = -1
				change_dip = -1
				change_mil = -1
				claim = 50
			}
		}
		if = {
			limit = { has_estate = estate_nobles }
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = 20
			}
		}
		else_if = {
			limit = { has_estate = estate_maratha }
			add_estate_loyalty = {
				estate = estate_maratha
				loyalty = 20
			}
		}
		else_if = {
			limit = { has_estate = estate_rajput }
			add_estate_loyalty = {
				estate = estate_rajput
				loyalty = 20
			}
		}
		kill_heir = { allow_new_heir = no }
		set_ruler_flag = has_elected_ruler
		ai_chance = {
			factor = 1
		}
	}

	#Somebody from the Army
	option = {
		name = government_events.3.d
		trigger = {
			army_size = 30
		}
		if = {
			limit = { has_government_attribute = has_improved_ruler_election }
			define_ruler = {
				change_mil = 2
				claim = 75
			}
		}
		else = {
			define_ruler = {
				change_adm = -1
				change_dip = -1
				change_mil = 2
				claim = 50
			}
		}
		add_army_tradition = 10
		hidden_effect = {
			convert_ruler_to_general = 50
		}
		kill_heir = { allow_new_heir = no }
		ai_chance = {
			factor = 1
		}
	}

	#Somebody from the Navy
	option = {
		name = government_events.3.e
		trigger = {
			navy_size = 30
		}
		if = {
			limit = { has_government_attribute = has_improved_ruler_election }
			define_ruler = {
				change_dip = 2
				claim = 75
			}
		}
		else = {
			define_ruler = {
				change_adm = -1
				change_dip = 2
				change_mil = -1
				claim = 50
			}
		}
		add_navy_tradition = 10
		convert_ruler_to_admiral = 50
		kill_heir = { allow_new_heir = no }
		ai_chance = {
			factor = 1
		}
	}

	#An influential Burgher
	option = {
		name = government_events.3.f
		trigger = {
			OR = {
				has_estate = estate_burghers
				has_estate = estate_vaisyas
			}
			OR = {
				estate_influence = {
					influence = 80
					estate = estate_burghers
				}
				estate_influence = {
					influence = 80
					estate = estate_vaisyas
				}
			}
		}
		if = {
			limit = { has_government_attribute = has_improved_ruler_election }
			define_ruler = {
				change_adm = 2
				claim = 75
			}
		}
		else = {
			define_ruler = {
				change_adm = 2
				change_dip = -1
				change_mil = -1
				claim = 50
			}
		}
		add_years_of_income = 0.1
		kill_heir = { allow_new_heir = no }
		ai_chance = {
			factor = 1
		}
	}

	#A member of the family
	option = {
		name = government_events.3.g
		trigger = {
			has_government_attribute = can_elect_from_relative
		}
		if = {
			limit = { has_government_attribute = has_improved_ruler_election }
			define_ruler = {
				dynasty = ROOT
				change_adm = 1
				claim = 100
			}
		}
		else = {
			define_ruler = {
				dynasty = ROOT
				change_adm = -1
				change_dip = -1
				change_mil = -1
				claim = 75
			}
		}
		add_prestige = 10
		kill_heir = { allow_new_heir = no }
		ai_chance = {
			factor = 1
		}
	}
}
