# Revolutionary Empire events by Daniel Tolman

namespace = napoleon

# The Emperor is present on the battlefield
country_event = {
	id = napoleon.1
	title = napoleon.1.t
	desc = napoleon.1.d
	picture = BATTLE_eventPicture
	
	is_triggered_only = yes # pulse
	
	trigger = {
		has_reform = revolutionary_empire_reform
		OR = {
			has_leader_with = {
				is_monarch_leader = yes
				siege = 5
			}
			has_leader_with = {
				is_monarch_leader = yes
				manuever = 5
			}
			has_leader_with = {
				is_monarch_leader = yes
				shock = 5
			}
			has_leader_with = {
				is_monarch_leader = yes
				fire = 5
			}
		}
		NOT = { has_country_modifier = napoleon_siege }
		NOT = { has_country_modifier = napoleon_manuever }
		NOT = { has_country_modifier = napoleon_shock }
		NOT = { has_country_modifier = napoleon_fire }
	}
	
	option = {
		name = napoleon.1.a
		highlight = yes
		trigger = {
			has_leader_with = {
				is_monarch_leader = yes
				siege = 5
			}
		}
		add_revolutionary_zeal = 5
		add_ruler_modifier = {
			name = "napoleon_siege" # siege ability
			duration = 1825
		}
	}
	
	option = {
		name = napoleon.1.b
		highlight = yes
		trigger = {
			has_leader_with = {
				is_monarch_leader = yes
				manuever = 5
			}
		}
		add_revolutionary_zeal = 5
		add_ruler_modifier = {
			name = "napoleon_manuever" # movement speed and attrition
			duration = 1825
		}
	}
	
	option = {
		name = napoleon.1.c
		highlight = yes
		trigger = {
			has_leader_with = {
				is_monarch_leader = yes
				shock = 5
			}
		}
		add_revolutionary_zeal = 5
		add_ruler_modifier = { # shock damage
			name = "napoleon_shock"
			duration = 1825
		}
	}
	
	option = {
		name = napoleon.1.da
		highlight = yes
		trigger = {
			has_leader_with = {
				is_monarch_leader = yes
				fire = 5
			}
		}
		add_revolutionary_zeal = 5
		add_ruler_modifier = { # fire damage
			name = "napoleon_fire"
			duration = 1825
		}
	}
}
