# Civil war
country_event = {
	id = civil_war.1
	title = "EVTNAME3091"
	desc = "EVTDESC3091"
	picture = CIVIL_WAR_eventPicture
	goto = rebel_province
	
	major = yes
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			set_saved_name = {
				key = upstart_pretender
				type = leader
			}
		}
		random_owned_province = {
			limit = {
				is_core = ROOT 
				development = 10
				is_capital = no
			}
			spawn_rebels = {
				type = 	pretender_rebels
				leader = upstart_pretender
				religion = ROOT
				size = 2
				win = yes
			}
			add_local_autonomy = 25

			hidden_effect = { save_event_target_as = rebel_province }
		}
		set_country_flag = had_civil_war
	}
	
	
	option = {
		name = "EVTOPTA3091"			# Crush them!
		add_prestige = -10
		if = {
			limit = {
				tag = LXA
			}
			custom_tooltip = lxa_warning_tt
		}
	}
	after = {
		clear_saved_name = upstart_pretender
	}
}
