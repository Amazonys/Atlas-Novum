country_event = {
	id = AN_seize.11
	title = "AN_seize.11.t"
	desc = "AN_seize.11.d"
	picture = BATTLE_eventPicture


	is_triggered_only = yes

	option = {
		name = "AN_seize.11.a"
		ai_chance = { factor = 100 }
		
		every_owned_province = {
			limit = { has_province_flag = cede_prov@PREV }
			cede_province = PREV
			clr_province_flag = cede_prov@PREV
		}
		white_peace = PREV
		hidden_effect = {
			PREV = { clr_country_flag = cede_to_tag }
			clr_country_flag = cede_from_tag
			clear_global_event_target = cede_to_tag
			clear_global_event_target = cede_from_tag
		}
	}
	
	option = { #Ignore ultimatum - player only
		name = "AN_seize.11.b"
		trigger = { ai = no }
		every_owned_province = {
			limit = { has_province_flag = cede_prov@PREV }
			add_territorial_core = PREV
			clr_province_flag = cede_prov@PREV
		}
		hidden_effect = {
			PREV = { clr_country_flag = cede_to_tag }
			clr_country_flag = cede_from_tag
			clear_global_event_target = cede_to_tag
			clear_global_event_target = cede_from_tag
		}
	}
}
