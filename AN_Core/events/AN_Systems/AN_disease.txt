#MEASLES
province_event = {
	id = AN_disease.1
	title = "AN_disease.1.t"
	desc = "AN_disease.1.d"
	picture = PLAGUE_eventPicture

	goto = ROOT

	#immediate = {
	#	save_event_target_as = IR_famine_loc
	#}

	trigger = {
		is_city = yes
		has_global_flag = AN_dev_init
		OR = { has_building = village has_building = town has_building = city has_building = metropolis }
		AN_disease_free = yes
		if = { #ONLY SPREADS INTO NEW WORLD AFTER OLD WORLD DISCOVERS
			limit = { capital_scope = { in_new_world = yes } }
			any_country = {
                capital_scope = { in_old_world = yes }
                has_discovered = ROOT
			}
		}
	}

	is_mtth_scaled_to_size = 1

	mean_time_to_happen = {
		months = 1200 #Once every 50 years
		modifier = {
			factor = 0.9
			has_building = town
		}
		modifier = {
			factor = 0.8
			has_building = city
		}
		modifier = {
			factor = 0.7
			has_building = metropolis
		}
		modifier = {
			factor = 0.5
			OR = {
				trade_goods = livestock #FROM CHICKENS OR CATTLE
				trade_goods = poultry #FROM CHICKENS OR CATTLE
				trade_goods = birds #FROM CHICKENS OR CATTLE
			}
		}
		modifier = { #UNLIKELY TO SPAWN TWICE AT THE SAME TIME
			factor = 4
			owner = { any_owned_province = { has_province_modifier = measles } }
		}
	}

	option = { #let the AN_disease spawn without response
		name = "AN_disease.x.a"
		add_devastation = 20
		add_permanent_province_modifier = {
			name = measles
   			duration = 730
		}
		AN_reduce_wealth_eighth = yes AN_reduce_pop_eighth = yes
		AN_check_dev_variables = yes
		custom_tooltip = AN_disease_chance_80
		hidden_effect = {
			every_neighbor_province = {
				limit = { AN_disease_free = yes }
				random = {
					chance = 80 #HIGH INFECTION
					province_event = { id = AN_disease.1 days = 5 random = 45 }
				}
			}
		}
	}
	option = { #Quarantine the province
		name = "AN_disease.x.b"
		owner = {
			add_adm_power = -50
			add_dip_power = -50
			add_mil_power = -50
		}
		add_devastation = 20
		add_permanent_province_modifier = {
			name = measles
   			duration = 730
		}
		AN_reduce_wealth_eighth = yes AN_reduce_pop_eighth = yes
		AN_check_dev_variables = yes
		custom_tooltip = AN_disease_chance_40
		hidden_effect = {
			every_neighbor_province = {
				limit = { AN_disease_free = yes }
				random = {
					chance = 40 # HALF INFECTION CHANCE
					province_event = { id = AN_disease.1 days = 5 random = 45 }
				}
			}
		}
	}
}

#PLAGUE
province_event = {
	id = AN_disease.2
	title = "AN_disease.2.t"
	desc = "AN_disease.2.d"
	picture = PLAGUE_eventPicture

	trigger = {
		is_city = yes
		has_global_flag = AN_dev_init
		OR = { has_building = village has_building = town has_building = city has_building = metropolis }
		AN_disease_free = yes
		capital_scope = { OR = { continent = europe region = north_china_region region = south_china_region region = anatolia_region region = mesopotamia_region region = mashriq_region region = egypt_region } } #ONLY EUROPE AND MIDDLE EAST AND CHINA
		is_year = 1460 #ONLY REEMERGES AFTER 1460
	}

	is_mtth_scaled_to_size = 1

	mean_time_to_happen = {
		months = 960 #Approximately every 40 years when possible
		modifier = {
			factor = 0.9
			has_building = town
		}
		modifier = {
			factor = 0.8
			has_building = city
		}
		modifier = {
			factor = 0.7
			has_building = metropolis
		}
		modifier = {
			factor = 2
			capital_scope = { OR = { region = scandinavia_region region = poland_region region = north_german_region } } #RESISTANCE IN POLAND, N GERMANY, SCANDINAVIA
		}
		modifier = {
			factor = 0.25
			capital_scope = { OR = { region = british_isles_region region = ireland_region region = france_region } } #FOCUSED IN BRITISH ISLES AND FRANCE INITIALLY
			is_year = 1465
			NOT = { is_year = 1625 }
		}
		modifier = {
			factor = 0.25
			capital_scope = { OR = { region = italy_region region = iberia_region region = russia_region } } #THEN SOUTHERN EUROPE AND RUSSIA
			is_year = 1625
			NOT = { is_year = 1680 }
		}
		modifier = {
			factor = 0.25
			capital_scope = { OR = { region = south_german_region region = baltic_region region = carpathia_region } } #THEN SOUTH GERMANY, DANUBE, AND BALTIC
			is_year = 1680
			NOT = { is_year = 1770 }
		}
		modifier = {
			factor = 0.25
			capital_scope = { OR = { region = dinaric_region region = balkan_region region = anatolia_region region = mesopotamia_region region = mashriq_region region = egypt_region } } #MIDDLE EAST AND BALKANS IN LATEGAME
			is_year = 1770
		}
		modifier = {
			factor = 0.1
			capital_scope = { OR = { region = north_china_region region = south_china_region }} #CHINA VERY LIKELY TO GET IT VERY LATE FOR THIRD PLAGUE PANDEMIC ~1850s

		}
		modifier = { #UNLIKELY TO SPAWN TWICE AT THE SAME TIME
			factor = 4
			owner = { any_owned_province = { has_province_modifier = plague } }
		}
	}

	option = { #let the AN_disease spawn without response
		name = "AN_disease.x.a"
		add_devastation = 20
		add_permanent_province_modifier = {
			name = plague
   			duration = 365
		}
		AN_reduce_wealth_fourth = yes AN_reduce_pop_fourth = yes
		AN_check_dev_variables = yes
		custom_tooltip = AN_disease_chance_50
		hidden_effect = {
			every_neighbor_province = {
				limit = { AN_disease_free = yes }
				random = {
					chance = 50 #MODERATE INFECTION
					province_event = { id = AN_disease.2 days = 5 random = 45 }
				}
			}
		}
	}
	option = { #Quarantine the province
		name = "AN_disease.x.b"
		owner = {
			add_adm_power = -50
