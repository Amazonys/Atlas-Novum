# The Iberian Wedding
country_event = {
	id = AN_marriages.1
	title = "AN_marriages.1.t"
	desc = "AN_marriages.1.d"
	picture = DIPLOMACY_eventPicture

	major = yes

	fire_only_once = yes

	trigger = {
		NOT = { exists = SPA }
		OR = {
			AND = {
				tag = CAS
				NOT = { war_with = ARA }
				ARA = {
					exists = yes
					ai = yes
					was_player = no
					is_subject = no
					is_neighbor_of = ROOT
				}
			}
			AND = {
				tag = ARA
				NOT = { war_with = CAS }
				ai = no
				CAS = {
					exists = yes
					ai = yes
					was_player = no
					is_subject = no
					is_neighbor_of = ROOT
				}
			}
		}
		is_year = 1469
		is_subject = no
		num_of_cities = 10
		NOT = { has_country_flag = AN_marriages.1_happened }
	}

	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			OR = {
				AND = {
					tag = CAS
					ARA = {
						has_regency = yes
					}
				}
				AND = {
					tag = ARA
					CAS = {
						has_regency = yes
					}
				}
			}
		}
	}

	immediate = { set_country_flag = AN_marriages.1_happened }


	option = {		# Let us bind their Dynasty to ours
		name = "flavor_spa.EVTOPTA3716"
		ai_chance = { factor = 100 }
		if = {
			limit = { tag = CAS }
			hidden_effect = {
				CAS = { release = ARA }
				ARA = { release = CAS }
			}
			#tag = CAT tag = VAL tag = MJO tag = SAR tag = SIC
			create_union = ARA
			#if = {
			#	limit = { NAP = { is_subject_of = ARA } }
			#	create_union = NAP
			#}
			hidden_effect = {
				if = {
					limit = { LLI = { is_subject_of = ARA } }
					ARA = { inherit = LLI }
				}
			}
			add_opinion = {
				who = ARA
				modifier = spa_iberian_wedding
			}
			ARA = {
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding
				}
				add_liberty_desire = -50
			}
			every_known_country = {
				limit = {
					marriage_with = ARA
				}
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding_negative
				}
			}
			define_ruler = {
 				name = "Isabella"
 				dynasty = "de Trastámara"
				age = 18
				claim = 75
				regency = no
				fixed = yes
				female = yes
				adm = 4
				dip = 5
				mil = 2
			}
			define_consort = {
 				name = "Ferdinand"
 				dynasty = "de Trastámara"
				age = 17
  				adm = 4
 				dip = 4
 				mil = 3
 				female = no
			}
		}
		if = {
			limit = { tag = ARA }
			hidden_effect = {
				CAS = { release = ARA }
				ARA = { release = CAS }
			}
			create_union = CAS
			hidden_effect = {
				if = {
					limit = { LLI = { is_subject_of = ARA } }
					ARA = { inherit = LLI }
				}
			}
			add_opinion = {
				who = CAS
				modifier = spa_iberian_wedding
			}
			CAS = {
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding
				}
				add_liberty_desire = -50
			}
			every_known_country = {
				limit = {
					marriage_with = CAS
				}
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding_negative
				}
			}
			define_ruler = {
 				name = "Ferdinand"
 				dynasty = "de Trastámara"
				age = 17
				claim = 75
				adm = 3
 				dip = 4
 				mil = 3
				fixed = yes
 				female = no
			}
			define_consort = {
 				name = "Isabella"
 				dynasty = "de Trastámara"
				age = 18
				female = yes
				adm = 4
				dip = 5
				mil = 2
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
	#	hidden_effect = { every_country = { limit = { OR = { tag = TLD tag = CRD tag = SEV tag = JAE tag = MUR tag = VAL tag = CAT } } } ROOT = { create_union = PREV } }
	}
	option = {		# No let us marry a local talent instead
		name = "flavor_spa.EVTOPTB3716"
		ai_chance = { factor = 1 }
		add_stability_or_adm_power = yes
		if = {
			limit = { tag = CAS }
			ARA = {
				add_opinion = {
					who = ROOT
					modifier = spa_declined_iberian_wedding
				}
			}
		}
		if = {
			limit = { tag = ARA }
			CAS = {
				add_opinion = {
					who = ROOT
					modifier = spa_declined_iberian_wedding
				}
			}
		}
		if = {
			limit = {
				has_consort = no
			}
			define_consort = {
				ADM = 3
				MIL = 3
				DIP = 3
			}
		}
	}
}


#Brittany Union
country_event = {
	id = AN_marriages.3
	title = AN_marriages.3.t
	desc = AN_marriages.3.d

	picture = DIPLOMACY_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		tag = FRA
		is_subject = no
		has_regency = yes
		BRI = {
			ai = yes
			exists = yes
			was_player = no
			is_subject = no
			NOT = { junior_union_with = FRA }
		}
		is_year = 1485
		NOT = { is_year = 1500 }
	}

	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.5
			is_year = 1488
		}
		modifier = {
			factor = 0.01
			is_year = 1490
		}
		modifier = {
			factor = 0.001
			is_year = 1492
		}
	}

	option = {
		name = AN_marriages.3.a #Treaty of Sable
		ai_chance = {
			factor = 99
		}
		create_union = BRI
	}

	option = {
		name = AN_marriages.3.b #Appoint a Local Noble.
		ai_chance = {
			factor = 1
		}
		define_ruler = {
			DIP = 4
			ADM = 4
			MIL = 4
			claim = 60
		}
	}
}
