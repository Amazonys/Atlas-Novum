# The Ambrosian Republic
country_event = {
	id = flavor_mlo.1
	title = flavor_mlo.1.t
	desc = flavor_mlo.1.d
	picture = DEBATE_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {	
			set_country_flag = had_ambrosian_republic
			change_government_to_republic = yes
			add_government_reform = ambrosian_republic
		}
	}
	
	option = {
		name = flavor_mlo.1.a
		ai_chance = { factor = 90 }
		tooltip = {
			change_government = republic
			add_government_reform = ambrosian_republic
		}
		add_republican_tradition = 5
		add_stability = -2
		every_neighbor_country  = {
			limit = {
				is_subject = no
				government = monarchy
				religion = catholic
			}
			add_opinion = {
				who = ROOT
				modifier = mlo_against_ambrosian
			}
			if = {
				limit = {
					government = monarchy
					NOT = { has_reform = elective_monarchy }
				}
				
				add_casus_belli = {	
					target = MLO
					type = cb_change_government
					months = 120
				}		
			}
		}
		hidden_effect = {
			if = {
				limit = {
					has_regency = yes
				}
				country_event = { id = 700 }
			}
		}
	}
}

# Niccolo Macchiavelli
country_event = {
	id = flavor_mlo.6
	title = flavor_mlo.6.t
	desc = flavor_mlo.6.d
	picture = ADVISOR_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = TUS
			tag = LAN
		}
		is_year = 1490
		NOT = { advisor_exists = 796 }
		government = republic
		owns = 116
		NOT = { is_year = 1515 }
	}
	
	mean_time_to_happen = {
		months = 12
	}	

	immediate = {
		set_country_flag = we_have_macchiavelli	
	}
	
	option = {
		name = flavor_mlo.6.a
		ai_chance = { factor = 90 }
		if = {
			limit = {
				NOT = { has_completed_all_reforms_trigger = yes }
			}

		}
		define_advisor = {
			type = philosopher
			culture = tuscan
			name = "Niccolò Machiavelli"
			discount = yes
			location = 116
			skill = 2
		}	
		add_prestige = 20
	}
	option = {
		name = flavor_mlo.6.b
		if = {
			limit = {
				NOT = { has_completed_all_reforms_trigger = yes }

		}

# Silk in Como?
province_event = {
	id = new_flavour_mlo.15
	title = new_flavour_mlo.15.name
	desc = new_flavour_mlo.15.desc
	picture = TRADE_GOODS_PLANTATION_GOODS_IN_WAREHOUSE_eventPicture
	
	is_triggered_only = yes
	
	goto = 4740

	immediate = {
		hidden_effect = {
			owner = {
				set_saved_name = {
					key = mlo_silk_guy
					type = advisor
					scope = this
				}
			}
		}
	}
	
	option = {
		name = new_flavour_mlo.15.a
		goto = 4740
		owner = {
			add_dip_power = -100
		}

		add_base_production = 2
		change_trade_goods = silk

		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = new_flavour_mlo.15.b #Channel his inventive energies elsewhere.

		owner = {
			if = {
				limit = {
					NOT = { monthly_income = 15 }
				}
				define_advisor = {
					type = natural_scientist
					name = mlo_silk_guy
					skill = 1
					discount = yes
				}
			}
			else_if = {
				limit = {
					NOT = { monthly_income = 25 }
				}
				define_advisor = {
					type = natural_scientist
					name = mlo_silk_guy
					skill = 2
					discount = yes
				}
			}
			else = {
				define_advisor = {
					type = natural_scientist
					name = mlo_silk_guy
					skill = 3
					discount = yes
				}
			}
		}
		
		ai_chance = {
			factor = 0
		}
	}
	after = {
		owner = {
			clear_saved_name = mlo_silk_guy
		}
	}
}
