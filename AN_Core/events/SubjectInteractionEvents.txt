# Pretender from previous dynasty rises
country_event = {
	id = subject_interaction_events.1
	title = subject_interaction_events.1.t
	desc = subject_interaction_events.1.d
	picture = ANGRY_MOB_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Common Sense"
		is_subject_other_than_tributary_trigger = yes
		government = monarchy
		has_new_dynasty = yes
		any_known_country = {
			overlord_of = ROOT
			dynasty = ROOT
			NOT = {
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_sent_help }
				reverse_has_opinion_modifier = { who = ROOT modifier = opinion_refused_send_help }
			}
		}
		NOT = {
			num_of_rebel_armies = 1
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = subject_interaction_events.1.a # Ask for help from Overlord
		random_owned_province = {
			limit = {
				is_capital = yes
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 3
			}
		}
		random_known_country = {
			limit = {
				overlord_of = ROOT
			}
			country_event = {
				id = subject_interaction_events.2
			}
		}
	}
	option = {
		name = subject_interaction_events.1.b # Deal with it ourselves
		add_prestige = 10
		random_owned_province = {
			limit = {
				is_capital = yes
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 2
			}
		}
	}
}

country_event = {
	id = subject_interaction_events.6
	title = subject_interaction_events.6.t
	desc = subject_interaction_events.6.d
	picture = ANGRY_MOB_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Common Sense"
		has_regency = no
		government = monarchy
		NOT = { has_reform = steppe_horde }
		religion_group = christian
		num_of_cities = 2
		NOT = {
			legitimacy = 50
		}
		any_subject_country = {
			OR = {
				is_vassal = yes
				is_march = yes
			}
			government = monarchy
			has_regency = no
			has_new_dynasty = yes
			dynasty = ROOT
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_subject_country = {
				limit = {
					government = monarchy
					has_regency = no
					has_new_dynasty = yes
					dynasty = ROOT
					OR = {
						is_vassal = yes
						is_march = yes
					}
				}
				set_country_flag = si_subject_state
			}
		}
	}
	
	option = {
		name = subject_interaction_events.6.a # Depose our cousin!
		random_subject_country = {
			limit = {
				has_country_flag = si_subject_state
			}
			define_ruler = {
				dynasty = ROOT
				culture = ROOT
				religion = ROOT
			}
			random_province = {
				limit = {
					owned_by = ROOT
					NOT = {
						is_capital = yes
					}
				}
				spawn_rebels = {
					type = pretender_rebels
					friend = PREV
					size = 1
				}
			}
			clr_country_flag = si_subject_state
		}
	}
	option = {
		name = subject_interaction_events.6.b
		add_corruption = 0.5
		add_treasury = -100
		random_subject_country = {
			limit = {
				has_country_flag = si_subject_state
			}
			add_treasury = 100
			clr_country_flag = si_subject_state
		}
	}
}

	is_triggered_only = yes

			NOT = {
				has_opinion_modifier = { who = ROOT modifier = opinion_smuggling_hindered }
				has_opinion_modifier = { who = ROOT modifier = opinion_small_smuggling_ok }
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_subject_country = {

					has_embargo_rivals = yes
					has_country_modifier = si_state_smuggling
				}
				set_country_flag = si_smuggler_vassal_flag
			}
		}
	}
	
	option = {
		name = subject_interaction_events.15.a
		ai_chance = {
			factor = 1
		}
		add_prestige = 10
		random_subject_country = {
			limit = {
				has_country_flag = si_smuggler_vassal_flag
			}
			set_country_flag = si_stop_smuggling_flag
			country_event = {
				id = subject_interaction_events.16
				tooltip = subject_interaction_events.16.tt1
			}
			add_opinion = {
				who = ROOT
				modifier = opinion_smuggling_hindered
			}
			tooltip = {
				remove_country_modifier = si_state_smuggling
			}
			clr_country_flag = si_smuggler_vassal_flag
		}
	}
	option = {
		name = subject_interaction_events.15.b
		ai_chance = {
			factor = 0
		}
		add_years_of_income = 0.25
		add_prestige = -10
		every_subject_country = {
			limit = { NOT = { has_country_flag = si_smuggler_vassal_flag } }
			add_liberty_desire = 5
		}
