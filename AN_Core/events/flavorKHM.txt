# Religious future of Khmer
country_event = {
	id = khmer.3
	title = khmer.3.t
	desc = khmer.3.d
	picture = BUDDHA_eventPicture

	is_triggered_only = yes
	
	option = {
		name = khmer.3.a
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				religion = hinduism
			}
		}
		if = {
			limit = {
				religion = buddhism
			}
			add_prestige = 10
		}
		else = {
			change_religion = buddhism
			add_stability = -1
		}
		set_estate_privilege = estate_church_brahmins_at_court
		every_owned_province = {
			limit = {
				region = indo_china_region
				religion = hinduism
			}
			change_religion = ROOT
		}
	}

	option = {
		name = khmer.3.b
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { religion = hinduism }
			}
		}
		if = {
			limit = {
				NOT = { religion = hinduism }
			}
			change_religion = hinduism
			add_stability = -1
		}
		else = {
			add_prestige = 10
		}
		if = {
			limit = {
				has_dlc = "Wealth of Nations"
				religion = hinduism
			}
			custom_tooltip = buddha_deity_tt
		}
		if = {
			limit = { 
				capital_scope = {
					NOT = { religion = hinduism }
				}
			}
			capital_scope = {
				change_religion = hinduism
			}
		}
		random_owned_province = {
			limit = {
				region = indo_china_region
				NOT = { religion = hinduism }
			}
			change_religion = hinduism
		}
		random_owned_province = {
			limit = {
				region = indo_china_region
				NOT = { religion = hinduism }
			}
			change_religion = hinduism
		}
	}
	
	option = {
		name = khmer.3.c
		ai_chance = {
			factor = 0
		}
		if = {
			limit = {
				NOT = {
					religion = mahayana
				}
			}
			change_religion = mahayana
		}
		else = {
			add_prestige = 10
		}
		if = {
			limit = {
				capital_scope = {
					NOT = { religion = mahayana }
				}
			}
			capital_scope = {
				change_religion = mahayana
			}
		}
	}
}
