	desc = {
		trigger = {
			is_colony = yes
		}
		desc = "EVTDESC6138"
	}

		trigger = {
			is_colony = no
		}
		desc = "colonial.1.d"

			random_empty_neighbor_province = {

		owner = { NOT = { num_of_colonies = 2 } }
	}

	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "EVTOPTA6138"
		goto = empty_neighbor_target
		event_target:empty_neighbor_target = {
			create_colony = 200
			add_province_modifier = {
				name = "colonial_boom"
				duration = 7300
			}
		}
	}
	option = {
		name = "EVTOPTB6138"
		trigger = {
			owner = {
			num_of_colonies = 1
			}
		}
		goto = root
		owner = {
			random_owned_province = {
				limit = {
					is_colony = yes
				}
				add_province_modifier = {
					name = "colonial_boom"

				}

			}
		}
	}
}

province_event = {
	id = colonial.5
	title = "EVTNAME1423"
	desc = "EVTDESC1423"
	picture = STREET_SPEECH_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		is_overseas = yes
		is_colony = yes
		colonysize = 600
		religion = owner
		NOT = { has_province_modifier = religious_migration }
		owner = {
			NOT = {
				any_owned_province = {
					expulsion_target = {
						province_id = root
					}
				}
			}
			any_owned_province = {
				is_overseas = no
				religion_group = ROOT
				NOT = { religion = ROOT }
				expelling_minorities = no
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			owner = {
				random_owned_province = {
					limit = {
						is_overseas = no
						religion_group = ROOT
						NOT = { religion = ROOT }
						expelling_minorities = no
					}
					save_event_target_as = heretics_from_here
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "EVTOPTA1423"
		goto = ROOT
		change_religion = event_target:heretics_from_here
		change_culture = owner
		add_colonysize = 400
		add_province_modifier = {
			name = "religious_migration"
			duration = 8760
		}
	}
	option = {
		name = "colonial.5.b"
		goto = heretics_from_here
		event_target:heretics_from_here = {
			add_province_modifier = {
				name = "enforcing_unity"
				duration = 3650
			}
		}
		add_colonysize = -100
	}
}

		days = 1
