# The Rise of Demak
country_event = {
	id = majapahit.5
	title = majapahit.5.t
	desc = majapahit.5.d
	picture = CIVIL_WAR_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	major = yes
	
	trigger = {
		NOT = { has_country_flag = maj_no_more_bad_religion_events }
		NOT = { exists = DMK }
		NOT = { legitimacy = 75 }
		629 = {
			religion = sunni
			owned_by = ROOT
		}
		num_of_owned_provinces_with = {
			value = 4
			religion = sunni
			region = indonesia_region
		}
	}
	
	immediate = {
		hidden_effect = {
			DMK = {
				set_saved_name = {
					key = demak_sultan
					type = leader
				}
			}
			629 = {
				add_core = DMK
			}
			if = {
				limit = {
					4792 = {
						owned_by = ROOT
					}
				}
				4792 = {
					add_core = DMK
				}
			}
			if = {
				limit = {
					4793 = {
						owned_by = ROOT
					}
				}
				4793 = {
					add_core = DMK
				}
			}
			if = {
				limit = {
					2689 = {
						owned_by = ROOT
					}
				}
				2689 = {
					add_core = DMK
				}
			}
			release = DMK
			DMK = {
				change_religion = sunni
				change_government = monarchy
				add_government_reform = mandala_reform
				define_ruler = {
					name = demak_sultan
					dynasty = ROOT
					claim = 95
				}
			}
			add_truce_with = DMK	
		}
	}
	
	option = {
		name = majapahit.5.a
		every_owned_province = {
			limit = {
				region = indonesia_region
			}
			add_claim = DMK
		}
		hidden_effect = {
			DMK = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.6
						cavalry = 0.3
						artillery = 0.1
					}
				}
				random_owned_province = {
					limit = {
						has_port = yes
					}
					build_to_forcelimit = {
						galley = 0.5
						light_ship = 0.2
						transport = 0.2
					}
				}
			}
		}
	}
}

# Subject liberty desire
country_event = {
	id = majapahit.7
	title = majapahit.7.t
	desc = majapahit.7.d
	picture = {
		trigger = { has_dlc = "Star and Crescent" }
		picture = CONQUEST_eventPicture 
	}
	picture = {
		trigger = { NOT = { has_dlc = "Star and Crescent" } }
		picture = TRADEGOODS_eventPicture
	}
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_flag = maj_no_more_subject_liberty }
		any_subject_country = {
			liberty_desire = 50
			ai = yes
			NOT = { truce_with = ROOT }
			NOT = { ROOT = { truce_with = PREV } }
		}
	}
	
	immediate = {
		hidden_effect = {
			random_subject_country = {
				limit = {
					is_subject_of_type = tributary_state
					liberty_desire = 50
					ai = yes
				}
				save_event_target_as = maj_tributary_country
			}
		}
	}
	
	option = {
		name = majapahit.7.a
		ai_chance = {
			factor = 1
		}
		if = {
			limit = {
				event_target:maj_tributary_country = {
					liberty_desire = 75
				}
			}
			event_target:maj_tributary_country = {
				grant_independence = yes
			}
		}
		else = {
			event_target:maj_tributary_country = {
				add_liberty_desire = 15
			}
		}
	}
	option = {
		name = majapahit.7.b
		trigger = {
			event_target:maj_tributary_country = {
				NOT = { liberty_desire = 80 }
			}
		}
		ai_chance = {
			factor = 0
		}
		add_legitimacy = -10
		event_target:maj_tributary_country = {
			add_liberty_desire = -5
		}
	}
}

# Pretender to Majapahit
country_event = {
	id = majapahit.8
	title = majapahit.8.t
	desc = majapahit.8.d
	picture = NUSANTARA_THRONE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_disaster = fall_of_majapahit
		NOT = { has_country_flag = maj_no_more_pretenders }
		any_owned_province = {
			region = indonesia_region
			controlled_by = ROOT
		}
	}
	immediate = {
		hidden_effect = {
			set_saved_name = {
				key = maj_pretender
				type = leader
			}
			random_owned_province = {
				limit = {
					region = indonesia_region
					controlled_by = ROOT
				}
				save_event_target_as = maj_pretender_province
				spawn_rebels = {
					type = pretender_rebels
					leader = maj_pretender
					size = 2
				}
			}
			set_country_flag = maj_pretender_event
		}
	}
	
	option = {
		name = majapahit.8.a
		tooltip = {
			event_target:maj_pretender_province = {
				spawn_rebels = {
					type = pretender_rebels
					leader = maj_pretender
					size = 2
				}
			}
		}
	}
}

# Spread of Islam in [province]
country_event = {
	id = majapahit.10
	title = majapahit.10.t
	desc = majapahit.10.d
	picture = {
		trigger = { has_dlc = "Star and Crescent" }
		picture = PIETY_eventPicture
	}
	picture = {
		trigger = { NOT = { has_dlc = "Star and Crescent" } }
		picture = RELIGION_eventPicture
	}
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			religion = sunni
			NOT = { has_country_flag = maj_no_more_bad_religion_events }
		}
		has_country_flag = maj_accepted_muslim_help
		any_owned_province = {
			region = indonesia_region
			NOT = { religion = sunni }
			OR = {
				has_port = yes
				any_neighbor_province = {
					religion = sunni
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					region = indonesia_region
					NOT = { religion = sunni }
					OR = {
						has_port = yes
						any_neighbor_province = {
							religion = sunni
						}
					}
				}
				save_event_target_as = maj_sunni_province
			}
		}
	}
	
	option = {
		name = majapahit.10.a
		event_target:maj_sunni_province = {
			change_religion = sunni
		}
	}
}
