#Colonisation of the Steppes
#Cossacks vs Nobles
country_event = {
	id = cossack_estate_events.1
	title = cossack_estate_events.1.t
	desc = cossack_estate_events.1.d
	picture = COSSACK_ESTATE_eventPicture
	
	trigger = {
		has_estate = estate_cossacks
		any_owned_province = {
			has_terrain = steppe
		}
		estate_influence = {
			estate = estate_nobles
			influence = 50
		}
		estate_influence = {
			estate = estate_cossacks
			influence = 25
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_cossacks
				modifier = EST_VAL_SERFS_HUNTED
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_cossacks
				modifier = EST_VAL_SERFS_TO_COSSACKS
			}
		}
		any_owned_province = {
			base_production = 2
			OR = {
				has_terrain = steppe
				any_neighbor_province = {
					owned_by = ROOT
					has_terrain = steppe
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			set_country_flag = colonization_of_the_steppes_flag
			random_owned_province = {
				limit = {
					OR = {
						has_terrain = steppe
						any_neighbor_province = {
							owned_by = ROOT
							has_terrain = steppe
						}
					}
					is_city = yes
				}
				set_province_flag = serfs_are_fleeing
			}
			random_owned_province = {
				limit = {
					has_terrain = steppe
				}
				set_province_flag = serfs_are_arriving
			}
		}
	}
	
	option = {
		name = cossack_estate_events.1.a #These Serfs must be brought back.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_cossacks
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_cossacks
					influence = 60
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_cossacks
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_cossacks
			desc = EST_VAL_SERFS_HUNTED
			influence = -10
			duration = 3650
		}
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_SERFS_BROUGHT_BACK
			influence = 10
			duration = 3650
		}
		random_owned_province = {
			limit = {
				has_province_flag = serfs_are_fleeing
			}
			add_base_production = 1
			clr_province_flag = serfs_are_fleeing
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_province_flag = serfs_are_arriving
				}
				clr_province_flag = serfs_are_arriving
			}
		}
	}
	option = {
		name = cossack_estate_events.1.b #We cannot afford to anger the Cossacks.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.2
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_cossacks
					influence = 60
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_cossacks
			desc = EST_VAL_SERFS_TO_COSSACKS
			influence = 10
			duration = 3650
		}
		add_estate_influence_modifier = {
			estate = estate_nobles
			desc = EST_VAL_SERFS_DEFECTING
			influence = -10
			duration = 3650
		}
		random_owned_province = {
			limit = {
				has_province_flag = serfs_are_arriving
			}
			add_base_manpower = 1
			clr_province_flag = serfs_are_arriving
		}
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_province_flag = serfs_are_fleeing
				}
				clr_province_flag = serfs_are_fleeing
			}
		}
	}

					}

							estate = estate_cossacks

					NOT = { region = west_siberia_region }
					NOT = { superregion = europe_superregion }

						NOT = { owned_by = ROOT }
						is_city = yes
						owner = {
							NOT = { war_with = ROOT }
							estate_influence = {
								estate = estate_cossacks
								influence = 15
							}
							OR = {
								is_rival = root
								ai_attitude = {
									who = root
									attitude = attitude_disloyal
								}
								ai_attitude = {
									who = root
									attitude = attitude_outraged
								}
								ai_attitude = {
									who = root
									attitude = attitude_rebellious
								}
								ai_attitude = {
									who = root
									attitude = attitude_hostile
								}
							}
							NOT = {
								has_estate_influence_modifier = {
									estate = estate_cossacks
									modifier = EST_VAL_COSSACK_AGGRESSIONS
								}
							}
							NOT = {
								has_estate_influence_modifier = {
									estate = estate_cossacks
									modifier = EST_VAL_STOPPED_COSSACK_AGGRESSIONS
								}
							}
						}
					}
				}
				random_neighbor_province = {
					limit = {
						NOT = { owned_by = ROOT }
						is_city = yes
						owner = {
							NOT = { war_with = ROOT }
							estate_influence = {
								estate = estate_cossacks
								influence = 15
							}
							OR = {
								is_rival = root
								ai_attitude = {
									who = root
									attitude = attitude_disloyal
								}
								ai_attitude = {
									who = root
									attitude = attitude_outraged
								}
								ai_attitude = {
									who = root
									attitude = attitude_rebellious
								}
								ai_attitude = {

							NOT = {
								has_estate_influence_modifier = {
									estate = estate_cossacks
									modifier = EST_VAL_COSSACK_AGGRESSIONS
								}
							}
							NOT = {
								has_estate_influence_modifier = {
									estate = estate_cossacks
									modifier = EST_VAL_STOPPED_COSSACK_AGGRESSIONS
								}
							}
						}
					}
					owner = { set_country_flag = cossack_raids_started_here }
					owner = { save_event_target_as = cossacks_here_target }
				}
			}
		}
	}
	
	option = {
		name = cossack_estate_events.2.a #Demand that the $ESTATE_COSSACKS$ be brought to Justice!
		ai_chance = {
			factor = 100
		}
		random_neighbor_country = {
			limit = { has_country_flag = cossack_raids_started_here }
			country_event  = { id = cossack_estate_events.3 }
		}
	}
	option = {
		name = cossack_estate_events.2.b #We will just have to let this slide.
		ai_chance = {
			factor = 0
		}
		if = {
			limit = { has_estate = estate_burghers }
			add_estate_loyalty = {
				estate = estate_burghers
				loyalty = -10
			}
		}
		if = {
			limit = { has_estate = estate_nobles }
			add_estate_loyalty = {
				estate = estate_nobles
				loyalty = -10
			}
		}
		if = {
			limit = { has_estate = estate_nomadic_tribes }
			add_estate_loyalty = {
				estate = estate_nomadic_tribes
				loyalty = -10
			}
		}
		random_owned_province = {
			limit = {
				base_tax = 2
				any_neighbor_province = {
					owner = {
						has_country_flag = cossack_raids_started_here
					}
				}
			}
			add_base_tax = -1
			owner = {
				clr_country_flag = cossack_raids_started_here
			}
			if = {
				limit = {
					owner = {
						NOT = {
							OR = {
								AND = {
									has_estate = estate_burghers
									has_estate = estate_nobles
								}
								has_estate = estate_nomadic_tribes
							}
						}
					}
				}
				add_unrest = 5
			}
		}
	}
