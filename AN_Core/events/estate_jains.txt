# Jains resist persecution
country_event = {
	id = jains_estate_events.4
	title = jains_estate_events.4.t
	desc = jains_estate_events.4.d
	picture = 10_Jain_Estate_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_estate = estate_jains
		NOT = {
			estate_loyalty = {
				estate = estate_jains
				loyalty = 30
			}
		}
		NOT = {
			has_country_modifier = jain_resistance
		}
		estate_territory = { 
			estate = estate_jains
			territory = 1
		}
		any_owned_province = {
			NOT = { has_province_modifier = passive_resistance }
			OR = {
				religion_group = dharmic
				religion = buddhism
				religion = vajrayana
				religion = mahayana
			}
			is_capital = no
			is_city = yes
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			while = {
				limit = {
					NOT = { 
						development_in_provinces = {
							has_province_flag = jain_passive_resistance
							value = estate
							estate = estate_jains
						}
					}
					any_owned_province = {
						NOT = { has_province_modifier = passive_resistance }
						OR = {
							religion_group = dharmic
							religion = buddhism
							religion = vajrayana
							religion = mahayana
						}
						is_capital = no
						is_city = yes
						NOT = { has_province_flag = jain_passive_resistance }
					}
				}
				random_owned_province = {
					limit = {
						NOT = { has_province_modifier = passive_resistance }
						OR = {
							religion_group = dharmic
							religion = buddhism
							religion = vajrayana
							religion = mahayana
						}
						is_capital = no
						is_city = yes
						NOT = { has_province_flag = jain_passive_resistance }
					}
					set_province_flag = jain_passive_resistance
				}
			}
		}
	}

	option = {
		name = jains_estate_events.4.a
		add_country_modifier = {
			name = jain_resistance
			duration = 5475
		}
		every_owned_province = {
			limit = {
				has_province_flag = jain_passive_resistance
			}
			add_province_modifier = {
				name = passive_resistance
				duration = 5475
			}
		}
	}

	after = {
		every_owned_province = {
			limit = {
				has_province_flag = jain_passive_resistance
			}
			clr_province_flag = jain_passive_resistance
		}
	}
}
