#Passover
country_event = {
	id = jewish_flavor.1
	title = jewish_flavor.1.t
	desc = jewish_flavor.1.d
	picture = FEAST_eventPicture
	
	trigger = {
		religion = jewish
		is_month = 2 #March. Passover is at the end of March-beginning of April
		NOT = { is_month = 4 }
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	immediate = { set_country_flag = shalosh_regalim }
	
	option = {
		name = jewish_flavor.1.a
		if = {
			limit = {
				any_owned_province = { trade_goods = livestock }
			}
			custom_tooltip = passover_tooltip
			hidden_effect = {
				every_owned_province = {
					limit = { trade_goods = livestock }
					add_province_modifier = { 
						name = shalosh_regalim_modifier
						duration = 365
					}
				}
			}
		}
		else = {
			add_country_modifier = {
				name = passover_modifier
				duration = 365
			}
		}
	}
}

#Bar Mitzvah
country_event = {
	id = jewish_flavor.4
	title = jewish_flavor.4.t
	desc = jewish_flavor.4.desc
	picture = FEAST_eventPicture
	
	trigger = {
		religion = jewish
		government = monarchy
		OR = {
			AND = {
				has_female_heir = yes
				heir_age = 12
			}
			AND = {
				heir_age = 13
				NOT = { has_female_heir = yes }
			}
		}
		NOT = { heir_age = 14 }
		NOT = { has_country_flag = had_bmv }
	}
	
	mean_time_to_happen = { days = 1 }
	
	immediate = { set_country_flag = had_bmv }
	
	option = {#Spend lavishly
		name = jewish_flavor.4.a
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2
				ruler_has_personality = drunkard_personality
			}
			modifier = {
				factor = 0
				ruler_has_personality = greedy_personality
			}
		}
		add_years_of_income = -1
		add_prestige = 20
		add_country_modifier = {
			name = lavish_bmv
			duration = 365		#Option to have these modifiers until heir gets to the throne?
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 15
		}
	}
	
	option = {#Do a normal celebration
		name = jewish_flavor.4.b
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2
				ruler_has_personality = pious_personality
			}
		}
		add_years_of_income = -0.66
		add_prestige = 10
		add_country_modifier = {
			name = normal_bmv
			duration = 365		#Option to have these modifiers until heir gets to the throne?
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = 10
		}
	}

	option = {#Be cheap
		name = jewish_flavor.4.c
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2
				OR = {
					ruler_has_personality = greedy_personality
					ruler_has_personality = embezzler_personality
				}
			}
		}
		add_country_modifier = {
			name = cheap_bmv
			duration = 365		#Option to have these modifiers until heir gets to the throne?
		}
		add_years_of_income = -0.33
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -5
		}
	}

	option = {#Do not celebrate
		name = jewish_flavor.4.d
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2
				OR = {
					ruler_has_personality = greedy_personality
					ruler_has_personality = embezzler_personality
				}
			}
			modifier = {
				factor = 0
				ruler_has_personality = pious_personality
			}
		}
		add_country_modifier = {
			name = no_bmv
			duration = 365		#Option to have these modifiers until heir gets to the throne?
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -10
		}
		add_prestige = -20
	}
}

# The Third Temple
country_event = {
	id = jewish_flavor.5
	title = jewish_flavor.5.t
	desc = jewish_flavor.5.d
	picture = JUDAISM_TEMPLE_eventPicture
	
	major = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			set_country_flag = has_third_temple_event_going
		}
	}

	after = {
		clr_country_flag = has_third_temple_event_going
	}
	
	option = { #We will build the Third Temple
		name = jewish_flavor.5.a
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = {
					treasury = 4000
				}
			}
		}
		add_treasury = -4000
		add_prestige = 20
		hidden_effect = {
			379 = {
				set_province_flag = third_temple_built
			}
		}
		if = {
			limit = {
				has_dlc = "Origins"
			}
			add_church_power = 20
		}
		if = {
			limit = {
				has_dlc = "Leviathan"
			}
			379 = {
				set_great_project_tier_3 = {
					type = holy_city_jerusalem
				}
			}
		}
		else = {
			palestine_area = {
				add_province_modifier = {
					name = "third_temple_provincial_modifier"
					duration = -1
				}
			}
			add_country_modifier = {
				name = "third_temple_country_modifier"
				duration = -1
			}
		}
		custom_tooltip = third_temple_consequences_tooltip
		hidden_effect = {
			every_country = { 
				limit = {
					religion_group = muslim
				}
				add_opinion = {
					who = ROOT
					modifier = destroyed_al_aqsa_opinion
				}
				add_casus_belli = {
					target = ROOT
					type = cb_insult
					months = 60
				}
			}
			every_owned_province = {
				if = {
					limit = {
						religion = ibadi
					}
				ibadi_rebels = 1
				}
				if = {
					limit = {
						religion = shiite
					}
				shiite_rebels = 1
				}
				if = {
					limit = {
						religion = sunni
					}
				sunni_rebels = 1
				}
			}
		}
	}

	option = { #We will build the Third Temple in a different location
		name = jewish_flavor.5.b
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = {
					treasury = 5000
				}
			}
		}
		add_treasury = -5000
		add_prestige = 20
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -30
		}
		hidden_effect = {
			379 = {
				set_province_flag = third_temple_built
			}
		}
		if = {
			limit = {
				has_dlc = "Origins"
			}
			add_church_power = 10
		}
		if = {
			limit = {
				has_dlc = "Leviathan"
			}
			379 = {
				set_great_project_tier_3 = {
					type = holy_city_jerusalem
				}
			}
		}
		else = {
			palestine_area = {
				add_province_modifier = {
					name = "third_temple_provincial_modifier"
					duration = -1
				}
			}
			add_country_modifier = {
				name = "third_temple_country_modifier"
				duration = -1
			}
		}
	}
	
	option = { #It is not the moment
		name = jewish_flavor.5.c
		ai_chance = {
			factor = 50
		}
		if = {
			limit = {
				has_dlc = "Origins"
			}
			add_church_power = -10
		}
		add_estate_loyalty = {
			estate = estate_church
			loyalty = -10
		}
		add_estate_influence_modifier = {
			estate = estate_church
			influence = 10
			duration = 7300
			desc = estate_not_built_temple
		}
	}
}
