# The Sikh Faith - Guru Nanak founds Sikhism
province_event = {
	id = sikhism_flavor.1
	title = "sikhism_flavor.1.t"
	desc = "sikhism_flavor.1.d"
	picture = STREET_SPEECH_eventPicture

	trigger = {
		is_year = 1480
		NOT = { is_religion_enabled = sikhism }
		OR = {
			AND = {
				religion = hinduism
				any_neighbor_province = {
					religion_group = muslim
				}
			}
			AND = {
				religion_group = muslim
				any_neighbor_province = {
					religion = hinduism
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 260
		modifier = {
			factor = 0.4
			province_id = 2075 # Doaba
		}
		modifier = {
			factor = 0.6
			culture = panjabi
		}
	}

	option = {
		name = "sikhism_flavor.1.a"
		owner = { enable_religion = sikhism }
		change_religion = sikhism
		add_province_triggered_modifier = religious_center
		add_permanent_province_modifier = {
			name = "religious_zeal_at_conv"
			duration = 4500
		}
		random_neighbor_province = {
			limit = {
				is_capital = no
				NOT = { religion = sikhism }
			}
			change_religion = sikhism
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 4500
			}
		}
		random_neighbor_province = {
			limit = {
				is_capital = no
				NOT = { religion = sikhism }
			}
			change_religion = sikhism
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 4500
			}
		}
	}
}

# Guru Arjan
country_event = {
	id = sikhism_flavor.5
	title = "sikhism_flavor.5.t"
	desc = "sikhism_flavor.5.d"
	picture = ADVISOR_eventPicture

	trigger = {
		religion = sikhism
		is_year = 1581
		is_month = 9
		NOT = { is_year = 1606 }
		current_guru_is = guru_ram_das
		NOT = { has_global_flag = guru_event_happening }
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_global_flag = guru_event_happening
	}

	after = {
		clr_global_flag = guru_event_happening
	}

	# Builds the Harmandir Sahib
	option = {
		name = "sikhism_flavor.5.a"
		if = {
			limit = {
				owns = 2075 # Punjab
				2075 = {
					NOT = { has_building = temple }
					num_free_building_slots = 1
				}
			}
			2075 = {
				add_building = temple
			}
		}
		else_if = {
			limit = {
				any_owned_province = {
					NOT = { has_building = temple }
				}
			}
			random_owned_province = {
				limit = {
					NOT = { has_building = temple }
					num_free_building_slots = 1
				}
				add_building = temple
			}
		}
		else = {
			add_adm_power = 50
		}
		appoint_next_guru = yes
	}
}
