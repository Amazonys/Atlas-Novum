#The Religious Mandate
country_event = {
	id = new_flavour_bav.18
	title = new_flavour_bav.18.name
	desc = new_flavour_bav.18.desc
	picture = REFORM_eventPicture

	fire_only_once = yes
	
	mean_time_to_happen = {
		months = 60
	}

	trigger = {
		OR = {
			tag = BAV
			tag = UBV
			tag = LBV
			tag = ING
		}
		is_religion_enabled = protestant
		religion = catholic
		NOT = {
			had_global_flag = {
				flag = reformation_started
				days = 3650
			}
		}
		NOT = {
			employed_advisor = {
				religion = protestant
			}
		}
		has_regency = no
		any_owned_province = {
			religion = root
		}
	}

	option = { #Institute the Mandate	
		name = new_flavour_bav.18.a

		add_adm_power = -50

		add_country_modifier = {
			name = bav_religious_mandate_mod
			duration = -1
		}

		reduce_estate_church_loyalty_effect = yes

		ai_chance = {
			factor = 2
		}
	}
	option = { #Don't do it
		name = new_flavour_bav.18.b
		
		if = {
			limit = {
				any_owned_province = {
					is_capital = no
					NOT = { religion = protestant }
				}
			}
			random_owned_province = {
				limit = {
					is_capital = no
					religion = root
					NOT = { has_province_modifier = religious_zeal_at_conv }
					NOT = { has_province_modifier = expelled_minority }
				}
				change_religion = protestant
			}
			else = {
				random_owned_province = {
					limit = {
						religion = root
					}
					change_religion = protestant
				}
			}
		}
		
		ai_chance = {
			factor = 1
		}
	}
}
