# Local Saints and Feasts
country_event = {
	id = patriarch_events.1
	title = patriarch_events.1.t
	desc = patriarch_events.1.d
	picture = ORTHODOX_eventPicture

	trigger = {
		has_dlc = "Third Rome"
		religion = orthodox
		any_owned_province = {
			has_state_patriach = yes
			has_owner_culture = no
			religion = orthodox
		}
	}
	
	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_state_patriach = yes
					has_owner_culture = no
					religion = orthodox
				}
				save_event_target_as = local_saint_state_province
			}
		}
	}

	option = {
		name = patriarch_events.1.a #
		event_target:local_saint_state_province = {
			area = {
				limit = { owned_by = ROOT }
				add_province_modifier = {
					name = "saints_enforced"
					duration = 5475
				}
			}
		}
	}
	option = {
		name = patriarch_events.1.b #
		event_target:local_saint_state_province = {
			area = {
				limit = { owned_by = ROOT }
				add_local_autonomy = 10
			}
		}
	}
}

# Heresy
country_event = {
	id = patriarch_events.4
	title = patriarch_events.4.t
	desc = patriarch_events.4.d
	picture = ORTHODOX_eventPicture

	trigger = {
		has_dlc = "Third Rome"
		religion = orthodox
		any_owned_province = {
			has_state_patriach = yes
			is_capital = no
			religion = orthodox
			is_in_capital_area = yes
			has_owner_culture = no
			NOT = {
				any_neighbor_province = {
					owned_by = ROOT
					is_capital = yes
				}
			}
		}
	}
	
	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_state_patriach = yes
					is_capital = no
					religion = orthodox
					is_in_capital_area = yes
					has_owner_culture = no
					NOT = {
						any_neighbor_province = {
							owned_by = ROOT
							is_capital = yes
						}
					}
				}
				save_event_target_as = judaizers_province
			}
		}
	}

	option = {
		name = patriarch_events.4.a #
		add_patriarch_authority = 0.05
		event_target:judaizers_province = {
			area = {
				limit = { owned_by = ROOT }
				add_province_modifier = {
					name = "rooting_out_heresy"
					duration = 3650
				}
			}
		}
		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = patriarch_events.4.b #
		add_patriarch_authority = -0.15
		ai_chance = {
			factor = 0
		}
	}
}

country_event = {
	id = patriarch_events.5
	title = patriarch_events.5.t
	desc = patriarch_events.5.d
	picture = ORTHODOX_eventPicture

	trigger = {
		has_dlc = "Third Rome"
		religion = orthodox
		any_owned_province = {
			has_state_patriach = yes
			is_capital = no
			religion = orthodox
			is_in_capital_area = yes
			NOT = { has_province_flag = land_theft_from_monastics }
		}
	}
	
	is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					has_state_patriach = yes
					is_capital = no
					religion = orthodox
					is_in_capital_area = yes
					NOT = { has_province_flag = land_theft_from_monastics }
				}
				save_event_target_as = monastic_province
			}
		}
	}

	option = {
		name = patriarch_events.5.a #

					name = "took_land_from_monastics"

				}
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = patriarch_events.5.b #
		add_patriarch_authority = 0.05

			factor = 1
		}
	}
