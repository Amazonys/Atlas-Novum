					}

					}
				}
				else = {
					NOT = {

					}
				}
			}
		}
		emperor = {
			vassalize = ROOT
		}
	}
	
	option = {
		name = "EVTOPTB9074"
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				if = {
					limit = {
						has_dlc = "Emperor"
					}
					will_back_reform = emperor_privilegia_de_non_appelando
				}
				else = {
					will_back_reform = privilegia_de_non_appelando
				}
			}			
			
		}
		every_owned_province = {
			limit = {
				is_part_of_hre = yes

		}

			set_in_empire = no
		}
		emperor = {
			add_opinion = { who = ROOT modifier = opinion_left_empire }
		}
		add_opinion = { who = emperor modifier = opinion_united_empire }
		elector = no
	
	}
}

	id = 9075
	title = "EVTNAME9075"
	desc = "EVTDESC9075"
	picture = HRE_eventPicture

	major = yes
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTA9075"
		change_tag = HLR
		on_change_tag_effect = yes
		restore_country_name_effect = yes
		add_country_modifier = {
			name = "rome_reborn"
			duration = -1
		}
		custom_tooltip = hre_core_tt
		hidden_effect = {
			every_province = {
				limit = {
					is_part_of_hre = yes
					OR = {
						owned_by = ROOT
						owner = { is_neighbor_of = ROOT }
					}
					NOT = { is_core = ROOT }
				}
				add_core = emperor
			}
		}
		custom_tooltip = hre_inheritance_tt
		hidden_effect = {
			every_country = {
				limit = {
					is_part_of_hre = yes
					is_emperor = no
					num_of_cities = 1
				}
				country_event = { id = 9076 }
			}
		}
		dismantle_hre = yes
		set_government_rank = 3
		if = {
			limit = { has_dlc = "Domination" }
			add_government_reform = holy_imperial_monarchy_reform
		}
		add_adm_power = 250
		if = {
			limit = {
				has_custom_ideas = no
			}
			country_event = { id = ideagroups.1 } #Swap Ideas
		}
		swap_non_generic_missions = yes
	}
}
