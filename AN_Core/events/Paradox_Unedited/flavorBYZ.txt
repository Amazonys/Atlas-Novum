country_event = {
	id = flavor_byz.1
	title = flavor_byz.1.t
	desc = flavor_byz.1.d
	picture = WOUNDED_SOLDIERS_eventPicture
	
	is_triggered_only = yes
	major = yes
	major_trigger = {
		byz_is_in_middle_east_or_europe = yes
	}
	
	goto = 112
	
	option = {
		name = flavor_byz.1.a
		
		ai_chance = {
			factor = 1
		}
		
		112 = {
			custom_tooltip = BYZ_this_province_will_lose_development
			
			custom_tooltip = " "
			
			hidden_effect = {
				export_to_variable = {
					which = BYZ_amount_of_dev
					value = trigger_value:development
				}
				multiply_variable = {
					which = BYZ_amount_of_dev
					value = 0.75
				}
				while = {
					limit = {
						check_variable = {
							which = BYZ_amount_of_dev
							value = 1
						}
						OR = {
							base_tax = 2
							base_production = 2
							base_manpower = 2
						}
					}
					random_list = {
						1 = {
							trigger = {
								base_tax = 2
							}
							add_base_tax = -1
							ROOT = { add_treasury = 75 }
						}
						1 = {
							trigger = {
								base_production = 2
							}
							add_base_production = -1
							ROOT = { add_treasury = 75 }
						}
						1 = {
							trigger = {
								base_manpower = 2
							}
							add_base_manpower = -1
							ROOT = { add_treasury = 75 }
						}
					}
					change_variable = {
						which = BYZ_amount_of_dev
						value = -1
					}
				}
				set_variable = {
					which = BYZ_amount_of_dev
					value = 0
				}
			}
			
			add_permanent_province_modifier = {
				name = BYZ_chastening_of_venice
				duration = -1
			}
		}
		capital_scope = {
			add_province_modifier = {
				name = "byzantine_treasures_returned"
				duration = 7300
			}
		}
	}
	
	option = {
		name = flavor_byz.1.b
		
		ai_chance = {
			factor = 0
		}
		
		if = {
			limit = {
				has_dlc = "Domination"
			}
			add_country_modifier = { 
				name = BYZ_may_construct_galleass
				duration = -1
			}
			custom_tooltip = BYZ_may_only_build_in_112
			hidden_effect = {
				112 = {
					add_province_modifier = {
						name = BYZ_local_galleass
						duration = -1
					}
				}
			}
		}
		else = {
			112 = {
				add_province_modifier = {
					name = BYZ_arsenal
					duration = -1
				}
			}
		}
	}

}
