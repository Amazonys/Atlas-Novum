namespace = flavor_arw
normal_or_historical_nations = yes

	
#Caribs expand towards the mouth of the Orinoco
country_event = {
	id = flavor_arw.1
	title = flavor_arw.1.t
	desc = flavor_arw.1.d
	picture = BORDER_TENSION_eventPicture
	fire_only_once = yes

	trigger = {
		tag = ARW
		any_owned_province = {
			colonial_region = colonial_colombia
		}
		CAB = {
			owns = 2803 #Orinoco Delta
			NOT = { alliance_with = ROOT }
			NOT = { has_opinion = { who = ROOT value = 50 } }
		}
	}

	mean_time_to_happen = {
		months = 12
	}
	
	
	option = {
		name = flavor_arw.1.a #Let us claim the Orinoco Delta!
		ai_chance = { factor = 95 }
		add_claim = 2803
	}
	option = {
		name = flavor_arw.1.b #We must prepare our defences instead.
		ai_chance = { factor = 5 }
		add_country_modifier = {
			name = "arw_preparing_defences"
			duration = 3650
		}
	}
}

#European Settlers in the Guyana
country_event = {
	id = flavor_arw.2
	title = flavor_arw.2.t
	desc = flavor_arw.2.d
	picture = BORDER_TENSION_eventPicture
	fire_only_once = yes

	
	trigger = {
		tag = ARW
		capital_scope = {
			area = guyana_area
		}
		any_known_country = {
			is_neighbor_of = ROOT
			OR = {
				capital_scope = {
					continent = europe
				}
				colonial_parent = {
					capital_scope = {
						continent = europe
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 24
	}
	
	
	option = {
		name = flavor_arw.2.a #Establish Trading Relations with the Europeans
		ai_chance = { factor = 25 }
		random_known_country = {
			limit = {
				is_neighbor_of = ROOT
				OR = {
					capital_scope = {
						continent = europe
					}
					colonial_parent = {
						capital_scope = {
							continent = europe
						}
					}
				}
			}
			add_opinion = {
				who = ROOT
				modifier = arw_traders
			}
		}
	}
	option = {
		name = flavor_arw.2.b #
		ai_chance = { factor = 25 }
		random_known_country = {
			limit = {
				is_neighbor_of = ROOT
				OR = {
					capital_scope = {
						continent = europe
					}
					colonial_parent = {
						capital_scope = {
							continent = europe
						}
					}
				}
			}
			add_opinion = {
				who = ROOT
				modifier = arw_hostility
			}
		}
	}
}
