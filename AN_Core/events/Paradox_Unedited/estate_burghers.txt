#Advancement of the Merchant classes

	id = burghers_estate_events.1
	title = burghers_estate_events.1.t

	picture = BURGHER_ESTATE_eventPicture
	
	trigger = {
		has_estate = estate_burghers

			duration = 5475
		}
	}

#Demands for city Privileges
country_event = {
	id = burghers_estate_events.4
	title = burghers_estate_events.4.t
	desc = burghers_estate_events.4.d
	picture = BURGHER_ESTATE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_estate = estate_burghers
		any_owned_province = {
			is_overseas = no
			has_seat_in_parliament = no
			OR = {
				development = 10
				province_has_center_of_trade_of_level = 1
			}
			is_capital = no
			is_territory = no
			is_city = yes
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_BURGHERS_ADVANCE
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 0.5
			NOT = { religion_group = christian }
		}
		modifier = {
			factor = 1.2
			any_owned_province = {
				is_overseas = no
				is_capital = no
				has_seat_in_parliament = no
				OR = {
					development = 10
					province_has_center_of_trade_of_level = 1
				}
				has_river_estuary_trigger = yes
				is_city = yes
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_overseas = no
					has_seat_in_parliament = no
					OR = {
						development = 10
						province_has_center_of_trade_of_level = 1
					}
					is_capital = no
					is_territory = no
					is_city = yes
				}
				save_event_target_as = growing_city
			}
		}
	}
	
	option = {
		name = burghers_estate_events.4.a #Ok
		give_estate_land_share_medium = { estate = estate_burghers }

			estate = estate_burghers
			loyalty = 15
		}
		event_target:growing_city = {
			add_base_tax = 2
		}
	}
	option = {
		name = burghers_estate_events.4.b #Ok
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = EST_VAL_CITY_PRIVILEGES_DENIED
			influence = -10
			duration = 5475
		}
	}
}

#Immoral Prices
#Force the Burghers to lose influence or face revolt in Burgher controlled provinces.
country_event = {
	id = burghers_estate_events.5
	title = burghers_estate_events.5.t
	desc = burghers_estate_events.5.d
	picture = BURGHER_ESTATE_eventPicture
	
	trigger = {
		has_estate = estate_burghers
		estate_territory = {
			estate = estate_burghers
			territory = 1
		}
		any_owned_province = {
			OR = {
				owner = { government = republic }
				development = 10
				province_trade_power = 5
			}
			is_capital = no
			is_city = yes
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_FREE_OF_MEDDLING
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_burghers
				modifier = EST_VAL_PRICE_MEDDLING
			}
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			inflation = 5 #Rising Prices
		}
	} 
	
	immediate = {
		hidden_effect = {
			while = {
				limit = {
					NOT = { 
						development_in_provinces = {
							has_province_flag = bread_revolt_vs_burghers
							value = estate
							estate = estate_burghers
						}
					}
					any_owned_province = {
						OR = {
							owner = { government = republic }
							development = 10
							province_trade_power = 5
						}
						is_capital = no
						is_city = yes
						NOT = { has_province_flag = bread_revolt_vs_burghers }
					}
					check_reducing_estate_revolt_size_more_trigger = {
						FLAG = bread_revolt_vs_burghers
					}
				}
				random_owned_province = {
					limit = {
						OR = {
							owner = { government = republic }
							development = 10
							province_trade_power = 5
						}
						is_capital = no
						is_city = yes
						NOT = { has_province_flag = bread_revolt_vs_burghers }
					}
					set_province_flag = bread_revolt_vs_burghers
				}
			}
		}
	}

	option = {
		name = burghers_estate_events.5.a #Force them to sell at a just price.
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.2
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 45
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_burghers
					influence = 65
				}
			}
		}
		hidden_effect = {
			every_owned_province = {
				limit = { has_province_flag = bread_revolt_vs_burghers }
				clr_province_flag = bread_revolt_vs_burghers
			}
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = -20
		}
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = EST_VAL_PRICE_MEDDLING
			influence = -10
			duration = 5475
		}
	}
	option = {
		name = burghers_estate_events.5.b #We should not meddle with the Burghers.
		ai_chance = {
			factor = 25
			modifier = {
				factor = 2.0
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 45
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_burghers
					influence = 65
				}
			}
		}
		add_estate_influence_modifier = {
			estate = estate_burghers
			desc = EST_VAL_FREE_OF_MEDDLING
			influence = 10
			duration = 5475
		}
		every_owned_province = {
			limit = { has_province_flag = bread_revolt_vs_burghers }
			clr_province_flag = bread_revolt_vs_burghers
			spawn_rebels = {
				type = anti_tax_rebels
				size = 1
			}
		}
		add_estate_loyalty = {
			estate = estate_burghers
			loyalty = 15
		}
	}
	option = {
		name = burghers_estate_events.5.c #Pay for the bread.
		ai_chance = {
