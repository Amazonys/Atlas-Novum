# Start of the Pueblo Revolt
country_event = {
	id = flavor_pue.2
	title = flavor_pue.2.t
	desc = flavor_pue.2.d
	picture = REVOLUTION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		primary_culture = pueblo #PUE, OHK, ISL, ACO, KER, ZNI
		has_dlc = "Leviathan"
		is_year = 1680
		NOT = { is_year = 1692 }
		any_known_country = {
			technology_group = western
			NOT = { alliance_with = ROOT }
			OR = {
				NOT = {
					has_opinion = {
						who = ROOT
						value = 0
					}
				}
				owns_core_province = 875
				war_with = ROOT
			}
		}
	}
	
	immediate = {
		if = {
			limit = {
				NOT = { tag = PUE }
			}
			add_stability = -1
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 0.75
			is_revolutionary = yes
		}
	}
	
	option = { 
		name = flavor_pue.2.a
		add_country_modifier = {
			name = "pue_popes_rebellion"
			duration = 3650
		}
		set_country_flag = pue_pueblo_revolt
		define_general = {
			name = "Po'pay"
			shock = 4
			fire = 2
			manuever = 4
			siege = 3
		}
	}
}

# End of the Pueblo Revolt 
country_event = {
	id = flavor_pue.3
	title = flavor_pue.3.t
	desc = flavor_pue.3.d
	picture = WOUNDED_SOLDIERS_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Leviathan"
		primary_culture = pueblo
		has_country_flag = pue_pueblo_revolt
		any_known_country = {
			technology_group = western
			war_with = ROOT
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	option = { 
		name = flavor_pue.3.a
		ai_chance = { factor = 90 }
		trigger = {
			war_score = 50
		}
		add_country_modifier = {
			name = pue_end_of_the_revolt
			duration = 3650
		}
	}
	
	option = { 
		name = flavor_pue.3.b
		trigger = {
			NOT = { war_score = 50 }
		}
		add_mil_power = -20
		every_owned_province = {
			add_unrest = 1
		}
	}
	
	option = { 
		name = flavor_pue.3.c
		trigger = {
			any_known_country = {
				technology_group = western
				alliance_with = ROOT
			}
		}
		add_stability = 1
	}
}

# Murals
country_event = {
	id = flavor_pue.4
	title = flavor_pue.4.t
	desc = flavor_pue.4.d
	picture = ARTWORK_eventPicture
	
	trigger = {
		has_dlc = "Leviathan"
		primary_culture = pueblo
		owns_core_province = 875
		875 = {
			has_great_project = {
				type = monument
				tier = 1
			}
		}
		NOT = { has_country_modifier = pue_spider_grandmother }
	}
	
	mean_time_to_happen = {
		months = 240
	}
	
	option = { #restore the murals
		name = flavor_pue.4.a
		trigger = {
			NOT = { has_country_flag = pue_restored_murals }
		}
		add_treasury = -25
		add_prestige = 5
		add_country_modifier = {
			name = pue_spider_grandmother
			duration = 7300
		}
		set_country_flag = pue_restored_murals
	}
	
	option = { #keep restoring them
		name = flavor_pue.4.b
		trigger = {
			has_country_flag = pue_restored_murals
		}
		add_treasury = -50
		add_prestige = 5
		add_country_modifier = {
			name = pue_spider_grandmother
			duration = 7300
		}
	}
	
	option = { #ignore them
		name = flavor_pue.4.c
		add_adm_power = 25
	}
}
