	id = flavor_may.1
	title = flavor_may.1.t
	desc = flavor_may.1.d

		OR = {
			tag = MAY
			tag = KIC
			tag = ITZ

		}
		religion_group = christian
		any_owned_province = {
			NOT = {
				religion_group = christian
			}
		}
		NOT = {
			has_country_modifier = may_popol_vuh
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = flavor_may.1.a

			name = "may_popol_vuh"
			duration = 3650
		}
	}
	option = {
		name = flavor_may.1.b
		add_prestige = 5
		add_papal_influence = 2
	}
}

country_event = {
	id = flavor_may.3
	title = flavor_may.3.t
	desc = flavor_may.3.d
	picture = RELIGIOUS_TURMOIL_eventPicture
	
	fire_only_once = yes

	trigger = {
		religion = mesoamerican_religion

		}

					region = carribeans_region

				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 32
	}
	
	option = {
		name = flavor_may.3.a
		change_religion = catholic
		add_stability = -1

			add_opinion = { who = ROOT modifier = may_convert }
			country_event = {
				id = flavor_may.4
				days = 7
				tooltip = flavor_may.4.tt
			}
		}
	}
	option = {
		name = flavor_may.3.b
		add_prestige = 5
		set_country_flag = may_kan_ek_flag
	}
}

			name = MAY_merchant_travels
