#Hussites march on Prague
country_event = {
	id = flavor_boh.1
	title = flavor_boh.1.t
	desc = flavor_boh.1.d
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = HUSSITE_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = CIVIL_WAR_eventPicture
	}

	fire_only_once = yes

	trigger = {
		tag = BOH
		owns = 266
		is_year = 1446
		NOT = { is_year = 1500 }
		is_at_war = no
		OR = {
			religion = catholic
			religion = hussite
		}
		is_free_or_tributary_trigger = yes
	}
	
	mean_time_to_happen = {
		months = 17
	}
	immediate = {
		hidden_effect = {
			set_country_flag = first_hussite_king
		}
	}
	
	option = { # Convert to Hussite
		ai_chance = { factor = 10 }
		name = flavor_boh.1.a
		set_country_flag = hussite_regency_mission_flag
		if = {
			limit = {
				has_regency = no
			}
			tooltip = {
				kill_ruler = yes
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = {}
		}
		change_religion = hussite
		define_ruler = {
			name = "Jirí"
			dynasty = "z Podebrad"
			adm = 5
			dip = 4
			mil = 3
			religion = hussite
			culture = czech
			age = 25
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		if = {
			limit = {
				has_dlc = "Rights of Man" 
			}
			hidden_effect = { add_ruler_personality_ancestor = { key = zealot } }
		}
		add_reform_desire = 0.02
	}
	
	option = { # Hussite King, Catholic nation - tolerance option
		ai_chance = { factor = 90 }
		set_country_flag = hussite_regency_mission_flag
		name = flavor_boh.1.c
		add_papal_influence = -100
		if = {
			limit = {
				has_regency = no
			}
			tooltip = {
				kill_ruler = yes
			}
		}
		if = {
			limit = {
				has_heir = yes
			}
			remove_heir = {}
		}
		define_ruler = {
			name = "Jirí"
			dynasty = "z Podebrad"
			adm = 5
			dip = 4
			mil = 3
			religion = hussite
		}
		if = {
			limit = {
				has_states_general_mechanic = yes
			}
			change_statists_vs_orangists = 1
		}
		add_reform_desire = 0.01
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		PAP = {
			country_event = { id = flavor_boh.9 }
		}
	}
	
	option = { # Fight them
		name = flavor_boh.1.b
		set_country_flag = hussite_regency_mission_flag
		266 = {
			spawn_rebels = {
				type = pretender_rebels
				size = 1
				leader = "Jirí"
				leader_dynasty = "z Podebrad"
			}		
		}
		random_owned_province = {
			limit = {
				NOT = { province_id = 266 }
			}
			spawn_rebels = {
				type = pretender_rebels
				size = 1
			}
		}
		if = {
			limit = { religion = catholic }
			add_papal_influence = 10
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
	}
}

#The Heretic King
country_event = {
	id = flavor_boh.2
	title = flavor_boh.2.t
	desc = flavor_boh.2.d
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = HUSSITE_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = COURT_eventPicture
	}

	is_triggered_only = yes

	trigger = {
		tag = BOH
		OR = {
			AND = {
				religion = catholic
				ruler_religion = hussite
			}
			religion = hussite
		}
		has_regency = no
		NOT = { has_ruler_flag = hussite_king }
		NOT = { is_religion_enabled = protestant }
		NOT = {
			check_variable = {
				which = hussite_changes_to_repent
				value = 3
			}
		}
		had_country_flag = {
			flag = first_hussite_king
			days = 182
		}
	}
	
	immediate = {
		hidden_effect = {
			set_ruler_flag = hussite_king
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = hussite_changes_to_repent
							value = 1
						}
					}
				}
				set_variable = {
					which = hussite_changes_to_repent
					value = 1
				}					
			}
			else_if = {
				limit = {
					check_variable = {
						which = hussite_changes_to_repent
						value = 1
					}
				}
				change_variable = {
					which = hussite_changes_to_repent
					value = 1
				}
			}
		}
	}
	
	option = { # reconciliation
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0.1
				is_year = 1470
			}
		}
		name = flavor_boh.2.a
		if = {
			limit = { exists = PAP }
			add_papal_influence = 5
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
		add_reform_desire = -0.01
		if = {
			limit = {
				religion = hussite
			}
			change_religion = catholic
			every_owned_province = {
				limit = {
					religion = hussite
				}
				change_religion = catholic
			}
		}
		if = {
			limit = {
				ruler_religion = hussite
			}
			set_ruler_religion = catholic
		}
		if = {
			limit = {
				heir_religion = hussite
			}
			set_heir_religion = catholic
		}
	}
	option = {
		name = flavor_boh.2.b
		ai_chance = {
			factor = 10
		}
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		add_reform_desire = 0.005
		if = {
			limit = {
				religion = catholic
			}
			excommunicate = ROOT
		}
		else_if = {
			limit = {
				religion = hussite
			}
			custom_tooltip = rejected_reconciliation_tt
			set_ruler_flag = hussite_heretic # catholics can cleanse heresy against you for 15 years
		}
		if = {
			limit = {
				exists = PAP
			}
			PAP = { country_event = { id = flavor_boh.9 } }
		}
	}
}

		ai_chance = {
			factor = 20
		}
		trigger = {
			religion = hussite
		}
		name = "flavor_boh.7.a"
		if = {
			limit = {
				has_dlc = "Art of War"
			}
			custom_tooltip = remain_hussite_tt # warning that you won't get a side for the league war
		}
		add_prestige = 10
		add_church_power = 10
	}
	
	option = { # Switch to Protestant
		ai_chance = {
			factor = 80
		}
		trigger = {
			religion = hussite
		}
		name = "flavor_boh.7.b"
		change_religion = protestant
		every_owned_province = {
			limit = {
				religion = hussite
			}
			change_religion = protestant
		}
		if = {
			limit = {
				tag = BOH
			}
			every_country = {
				limit = {
					NOT = { tag = BOH }
					OR = {
						religion = hussite
						any_owned_province = {
							religion = hussite
						}
					}
				}
				country_event = {
					id = flavor_boh.7
					days = 3
				}
			}
		}
	}
	
	option = { # Hussite provinces convert
		trigger = {
			NOT = { 
				religion = hussite 
			}
		}
		name = "flavor_boh.7.b"
		every_owned_province = {

				religion = hussite
			}
			change_religion = protestant

# Winter King in Prague
country_event = {
	id = flavor_boh.8
	title = flavor_boh.8.t
	desc = flavor_boh.8.d
	picture = COURT_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		tag = BOH
		is_emperor = no
		religion = catholic
		any_owned_province = {
			religion = protestant
			OR = {
				culture = czech
				culture = czech_new
			}
		}
		is_year = 1580
	}
	
	
	option = {
		name = flavor_boh.8.a
		ai_chance = { factor = 90 }	
		change_religion = protestant
		if = {
			limit = {
				emperor = { religion = catholic }
			}
			emperor = {
				add_casus_belli = {	
					target = BOH
					type = cb_religious_conformance
					months = 1200
				}
				add_opinion = {
					who = root
					modifier = boh_protestant_king
				}										
			}
		}

		every_neighbor_country = {
			limit = {
				is_emperor = no
				religion = catholic
				num_of_cities = 5
			}
			add_casus_belli = {	
				target = BOH
				type = cb_religious_conformance
				months = 1200
			}	
			add_opinion = {
				who = root
				modifier = boh_protestant_king
			}							
		}
	}
}
