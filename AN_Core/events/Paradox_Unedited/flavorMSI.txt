#The federal Mossi Kingdom
country_event = {
	id = flavor_msi.1
	title = flavor_msi.1.t
	desc = flavor_msi.1.d
	picture = MALI_DECLINE_eventPicture
	
	trigger = {
		NOT = { has_country_flag = msi_mossi_kingdom_event }
		NOT = { has_reform = mossi_federal_kingdom }
		tag = MSI
	}

	mean_time_to_happen = {
		months = 1
	}

	immediate = {
		hidden_effect = {
			set_country_flag = msi_mossi_kingdom_event
		}
	}
	
	option = {	#
		name = flavor_msi.1.a
		if = {
			limit = {
				NOT = { government = monarchy }
			}
			change_government = monarchy
		}
		add_government_reform = mossi_federal_kingdom
		ai_chance = { factor = 1 }
	}
}

#Manpower from the other kingdoms
country_event = {
	id = flavor_msi.2
	title = flavor_msi.2.t
	desc = flavor_msi.2.d
	picture = AFRICAN_DISCIPLINE_eventPicture
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			clr_country_flag = collecting_manpower_from_kingdoms
			set_variable = {
				which = federal_manpower_gain
				value = 0
			}
			export_to_variable = {
				which = manpower_modifier_var
				value = modifier:global_manpower_modifier
			}
			change_variable = {
				which = manpower_modifier_var
				value = 1
			}
			while = {
				limit = {
					any_owned_province = {
						has_province_modifier = gathers_soldiers_for_the_ruler
						controlled_by = ROOT	#A province which is occupied by the enemy should not be able to send any kind of manpower
					}
				}
				random_owned_province = {
					limit = {
						has_province_modifier = gathers_soldiers_for_the_ruler
						controlled_by = ROOT
					}
					remove_province_modifier = gathers_soldiers_for_the_ruler
					ROOT = {
						export_to_variable = {
							which = province_manpower
							value = trigger_value:base_manpower
							who = PREV
						}
						change_variable = {
							which = federal_manpower_gain
							which = province_manpower
						}
					}
				}
			}
			multiply_variable = {
				which = federal_manpower_gain
				which = manpower_modifier_var
			}
			set_variable = {
				which = federal_manpower_gain_tooltip
				which = federal_manpower_gain
			}
			multiply_variable = {
				which = federal_manpower_gain_tooltip
				value = 1000
			}
			hidden_effect = {
				every_owned_province = {	#Clean up
					limit = {
						has_province_modifier = gathers_soldiers_for_the_ruler
					}
					remove_province_modifier = gathers_soldiers_for_the_ruler
				}
			}
		}
	}
	
	option = {	#
		name = flavor_msi.2.a
		#add_years_of_owned_provinces_manpower = {
		#	years = 1
		#	trigger = {
		#		has_province_modifier = gathers_soldiers_for_the_ruler
		#	}
		#	custom_tooltip = flavor_msi.2.a.tt
		#}

		custom_tooltip = flavor_msi.2.a.tt
		hidden_effect = {
			while = {
				limit = {
					check_variable = {
						which = federal_manpower_gain
						value = 1
					}
				}
				add_manpower = 1
				subtract_variable = {
					which = federal_manpower_gain
					value = 1
				}
			}
			multiply_variable = {
				which = federal_manpower_gain
				value = 1000
			}
			while = {
				limit = {
					check_variable = {
						which = federal_manpower_gain
						value = 1
					}
				}
				add_manpower = 0.001
				subtract_variable = {
					which = federal_manpower_gain
					value = 1
				}
			}
		}
		ai_chance = { factor = 1 }
	}
}

#Tribal allegiance
country_event = {
	id = flavor_msi.5
	title = flavor_msi.5.t
	desc = flavor_msi.5.d
	picture = AFRICAN_BATTLE_eventPicture
	
	trigger = {
		has_government_attribute = has_federal_kingdoms
		had_country_flag = {
			flag = fed_kingdom_allegience_support
			days = 180
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			clr_country_flag = fed_kingdom_allegience_support
			set_variable = {
				which = allegience_tooltip
				value = 0
			}
			every_owned_province = {
				limit = {
					culture = ROOT
					NOT = { is_capital_of = ROOT }
				}
				ROOT = {
					change_variable = {
						which = allegience_tooltip
						value = 3
					}
				}
			}
		}
	}
	
	option = {	#
		name = flavor_msi.5.a
		custom_tooltip = flavor_msi.5.a.tt
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = ROOT
					NOT = { is_capital_of = ROOT }
				}
				ROOT = { add_tribal_allegiance = 3 }
				add_local_autonomy = 5
			}
		}
		ai_chance = { factor = 1 }
	}
}
