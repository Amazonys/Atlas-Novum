#Nobles Demand Exlusive right to Commissions
country_event = {
	id = nobles_estate_events.1
	title = nobles_estate_events.1.t
	desc = nobles_estate_events.1.d
	picture = NOBLE_ESTATE_eventPicture
	
	trigger = {
		has_estate = estate_nobles
		estate_influence = {
			estate = estate_nobles
			influence = 70
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_nobles
				modifier = EST_VAL_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_nobles
				modifier = EST_VAL_NO_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			}
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = nobles_estate_events.1.a #
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.2
				estate_loyalty = {
					estate = estate_nobles
					loyalty = 50
				}
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 60
					}
				}
			}
			modifier = {
				factor = 0.2
				has_estate = estate_burghers
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 30
					}
				}
			}
			modifier = {
				factor = 0.2
				has_estate = estate_vaisyas
				NOT = {
					estate_loyalty = {
						estate = estate_vaisyas

					}
				}
			}
		}
		add_estate_loyalty = {
			estate = estate_nobles

		}

			limit = {
				has_estate = estate_burghers

				estate = estate_burghers

				has_estate = estate_vaisyas

				estate = estate_vaisyas

			name = "only_noble_officers"

		}

			estate = estate_nobles
			desc = EST_VAL_EXCLUSIVE_RIGHT_TO_BE_OFFICERS_NOBLES
			influence = 10
			duration = 7300
		}
	}
	option = {
		name = nobles_estate_events.1.b #
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.1
				has_estate = estate_burghers
				NOT = {
					estate_loyalty = {
						estate = estate_burghers
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 0.1
				has_estate = estate_vaisyas
				NOT = {
					estate_loyalty = {
						estate = estate_vaisyas
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 30
					}
				}
			}
		}

			loyalty = -20

#Nobles encroach on Government Lands
country_event = {
	id = nobles_estate_events.3
	title = nobles_estate_events.3.t
	desc = nobles_estate_events.3.d
	picture = NOBLE_ESTATE_eventPicture
	goto = noble_encroachment
	
	trigger = {
		has_estate = estate_nobles
		num_of_cities = 15
		any_owned_province = {
			is_overseas = no
			is_capital = no
			is_colony = no
			has_seat_in_parliament = no
			is_territory = no
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_capital = no
					is_overseas = no
					is_colony = no
					has_seat_in_parliament = no
					is_territory = no
				}
				save_event_target_as = noble_encroachment
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
		modifier = {
			factor = 2.0
			estate_influence = {
				estate = estate_nobles
				influence = 80
			}
			NOT = {
				estate_loyalty = {
					estate = estate_nobles
					loyalty = 80
				}
			}
		}
	}
	
	option = {
		name = nobles_estate_events.3.a #Allow it.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0.25
				estate_territory = {
					estate = estate_nobles
					territory = 12
				}
			}
			modifier = {
				factor = 0.25
				estate_influence = {
					estate = estate_nobles
					influence = 60
				}
			}
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 30
					}
				}
			}
		}
		event_target:noble_encroachment = {
			add_local_autonomy = 25
		}
		give_estate_land_share_small = { estate = estate_nobles }
	}
	option = {
		name = nobles_estate_events.3.b #Take it Back.
		ai_chance = {
			factor = 50
			modifier = {
				factor = 1.1
				NOT = {
					estate_territory = {
						estate = estate_nobles
						territory = 15
					}
				}
			}
			modifier = {
				factor = 0.25
				NOT = {
					estate_loyalty = {
						estate = estate_nobles
						loyalty = 35
					}
				}
			}
		}
		event_target:noble_encroachment = {
			random_list = {
				50 = {
					spawn_rebels = {
						type = noble_rebels
						size = 1
					}
				}
				50 = {  }
			}
		}
		add_estate_loyalty = {
			estate = estate_nobles
			loyalty = -15
		}
	}
}
