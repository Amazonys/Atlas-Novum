# Atabegs Declare Independence
country_event = {
	id = georgian_crisis.3
	title = georgian_crisis.3.t
	desc = georgian_crisis.3.d
	picture = HORDE_ON_HORSEBACK_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes 
	
	trigger = {
		has_disaster = georgian_crisis
		OR = {
			AND = {
				owns = 2204
				owns = 4303
			}
			SME = {
				is_subject_of = ROOT
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "georgian_crisis.3.a"
		if = {
			limit = {
				SME = {
					is_subject_of = ROOT
				}
			}
			SME = {
				grant_independence = yes
			}
			declare_war_with_cb = {
				who = SME
				casus_belli = cb_core
				war_goal_province = 2204
			}
			hidden_effect = {
				SME = {
					set_dynasty = "Jaqeli"
					change_government = monarchy
					add_government_reform = autocracy_reform
					change_religion = orthodox
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.8
							cavalry = 0.2
						}
					}
				}
			}
		}
		else_if = {
			limit = {
				owns = 2204
				owns = 4303
			}
			release = SME
			declare_war_with_cb = {
				who = SME
				casus_belli = cb_core
				war_goal_province = 2204
			}
			hidden_effect = {
				SME = {
					set_dynasty = "Jaqeli"
					change_government = monarchy
					add_government_reform = autocracy_reform
					change_religion = orthodox
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.8
							cavalry = 0.2
						}
					}
				}
			}
		}
	}
	
	option = {
		name = "georgian_crisis.3.b"
		add_legitimacy = -20
		if = {
			limit = {
				SME = {
					is_subject_of = ROOT
				}
			}
			SME = {
				grant_independence = yes
			}
			add_truce_with = SME
		}
		else_if = {
			limit = {
				owns = 2204
				owns = 4303
			}
			release = SME
			add_truce_with = SME
			hidden_effect = {
				SME = {
					set_dynasty = "Jaqeli"
					change_government = monarchy
					add_government_reform = autocracy_reform
					change_religion = orthodox
					capital_scope = {
						build_to_forcelimit = {
							infantry = 0.8
							cavalry = 0.2
						}
					}
				}
			}
		}
	}
}
