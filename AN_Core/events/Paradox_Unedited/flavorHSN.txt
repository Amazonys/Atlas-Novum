country_event = {
	id = flavor_hsn.1
	title = flavor_hsn.1.t
	desc = flavor_hsn.1.d
	picture = {	
		trigger = {
			has_dlc = "Rights of Man"
		}
		picture = SWORD_OF_OSMAN_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Rights of Man" }
		}
		picture = GOOD_WITH_MONARCH_eventPicture
	}
	
	major = yes
	
	is_triggered_only = yes
	
	option = {
		name = flavor_hsn.1.a
		override_country_name = AYB
		set_country_flag = is_AYB_flag
		set_country_flag = has_overriden_name_flag
		custom_tooltip = hsn_empower_ideas_tt
		if = {
			limit = {
				NOT = {
					government_rank = 2
				}
			}
			set_government_rank = 2
		}
		add_country_modifier = {
			name = "hsn_rise_of_the_eagle"
			duration = 7300
		}
		hidden_effect = {
			set_country_flag = hsn_ayyubid_ideas
			swap_free_idea_group = yes
		}
	}
}

country_event = {
	id = flavor_hsn.6
	title = flavor_hsn.6.t
	desc = flavor_hsn.6.d
	picture = {	
		trigger = {
			OR = {
				has_dlc = "Cradle of Civilization"
				has_dlc = "Star and Crescent"
			}
		}
		picture = muslimgfx_GREAT_BUILDING_eventPicture
	}
	picture = {	
		trigger = {
			NOT = {
				OR = {
					has_dlc = "Cradle of Civilization"
					has_dlc = "Star and Crescent"
				}
			}
		}
		picture = GREAT_BUILDING_eventPicture
	}
	
	trigger = {
		OR = {
			tag = HSN
			was_tag = HSN
		}
		owns_core_province = 382
		is_at_war = no
		382 = {
			has_tax_building_trigger = yes
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	fire_only_once = yes
	
	option = {
		name = flavor_hsn.6.a
		add_years_of_income = -0.5
		382 = {
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
		382 = {
			upgrade_tax_building = yes
		}
	}
	
	option = {
		name = flavor_hsn.6.b
		add_adm_power = 50
	}
	
}
