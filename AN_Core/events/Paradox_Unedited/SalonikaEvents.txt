namespace = salonika

# Jew seek refuge in Salonika
country_event = {
	id = salonika.1
	title = "salonika.1.t"
	desc = "salonika.1.d"
	picture = DHIMMI_ESTATE_eventPicture
	goto = 148
	
	fire_only_once = yes
	
	trigger = {
		owns_core_province = 148
		religion_group = muslim
		148 = {
			NOT = { religion = ROOT }
		}
		OR = {
			CAS = { has_country_modifier = alhambra_decree }
			SPA = { has_country_modifier = alhambra_decree }
		}
	}
	
	mean_time_to_happen = {
		months = 30
	}
	
	option = {
		ai_chance = { factor = 1 }
		name = salonika.1.a
		148 = {
			add_province_modifier = {
				name = "selanik_jews"
				duration = -1
			}
		}
		if = {
			limit = {
				has_estate = estate_dhimmi
			}
			add_estate_loyalty = {
				estate = estate_dhimmi
				loyalty = 15
			}
		}
		else = {
			add_prestige = 10
		}
	}
	option = {
		ai_chance = { factor = 0 }
		name = salonika.1.b
		if = {
			limit = {
				has_estate = estate_dhimmi
			}
			add_estate_loyalty = {
				estate = estate_dhimmi
				loyalty = -15
			}
		}
		else = {
			add_prestige = -5
		}
	}
}

# Jewish population soars in Salonika
country_event = {
	id = salonika.2
	title = "salonika.2.t"
	desc = "salonika.2.d"
	picture = DHIMMI_ESTATE_eventPicture
	goto = 148
	
	fire_only_once = yes
	
	trigger = {
		NOT = {	num_of_estate_privileges = { estate = estate_dhimmi	value = 4 } }
		religion_group = muslim
		148 = {
			NOT = { religion = ROOT }
			owned_by = ROOT
			has_province_modifier = selanik_jews
		}
	}
	
	mean_time_to_happen = {
		months = 600
	}
	
	option = {
		ai_chance = { factor = 25 }
		name = salonika.2.a
		if = {
			limit = {
				has_estate = estate_dhimmi
				has_country_flag = unlocked_privilege_estate_dhimmi_guarantee_of_traditions
				NOT = { has_estate_privilege = estate_dhimmi_guarantee_of_traditions }
			}
			set_estate_privilege = estate_dhimmi_guarantee_of_traditions
		}
		else_if = {
			limit = {
				has_estate = estate_dhimmi
				NOT = { has_estate_privilege = estate_dhimmi_guaranteed_autonomy }
				NOT = { has_country_flag = unlocked_privilege_estate_dhimmi_guarantee_of_traditions }
			}
			set_estate_privilege = estate_dhimmi_guaranteed_autonomy
		}
		else_if = {
			limit = {
				has_estate = estate_dhimmi
				NOT = { has_estate_privilege = estate_dhimmi_land_rights }
			}
			set_estate_privilege = estate_dhimmi_land_rights
		}
		else_if = {
			limit = {
				has_estate = estate_dhimmi
				NOT = { has_estate_privilege = estate_dhimmi_lighter_dhimmi_taxes }
			}
			set_estate_privilege = estate_dhimmi_lighter_dhimmi_taxes
		}
		else_if = {
			limit = {
				has_estate = estate_dhimmi
				NOT = { has_estate_privilege = estate_dhimmi_tolerance }
				NOT = { has_country_flag = unlocked_privilege_estate_dhimmi_guarantee_of_traditions }
			}
			set_estate_privilege = estate_dhimmi_tolerance
		}
		148 = {
			change_religion = jewish
			add_base_production = 4
		}
	}
	option = {
		ai_chance = { factor = 75 }
		name = salonika.2.b
		148 = {
			add_base_production = 1
		}
	}
}
