# Foreign Trade
country_event = {
	id = flavor_sio.1
	title = flavor_sio.1.t
	desc = flavor_sio.1.d
	picture = CORRUPTION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Leviathan"
		primary_culture = dakota
		any_owned_province = {
			NOT = { has_province_modifier = sio_foreigner_trade }
		}
		if = {
			limit = {
				exists = FRA
			}
			FRA = {
				NOT = { war_with = ROOT }
				any_owned_province = {
					continent = north_america
				}
			}
		}
		else_if = {
			limit = {
				exists = GBR
			}
			GBR = {
				NOT = { war_with = ROOT }
				any_owned_province = {
					continent = north_america
				}
			}
		}
		else_if = {
			limit = {
				exists = SPA
			}
			SPA = {
				NOT = { war_with = ROOT }
				any_owned_province = {
					continent = north_america
				}
			}
		}
		else = {
			is_at_war = no
			is_year = 1600
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = flavor_sio.1.a
		random_owned_province = {
			limit = { is_core = ROOT }
			add_province_modifier = {
				name = sio_foreigner_trade
				duration = 7300
			}
		}
		if = {
			limit = {
				OR = {
					has_reform = native_war_band_reform
					has_reform = native_settle_down_reform
				}
			}			
			ROOT = {
				add_reform_progress_small_effect = yes
			}
		}
	}
}

# Feast of the Dead
country_event = {
	id = flavor_sio.5
	title = flavor_sio.5.t
	desc = flavor_sio.5.d
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Leviathan"
		is_year = 1659
		OR = {
			908 = { owned_by = ROOT }
			calc_true_if = {
				all_province = {
					region = great_lakes_region
					owned_by = ROOT
				}
				amount = 3
			}
		}
		NOT = {	has_country_modifier =  "sio_feast" }
	}
	
	mean_time_to_happen = {
		months = 24
	}
	
	option = {
		name = flavor_sio.5.a
		if = {
			limit = {
				any_known_country ={
					government = native
					religion = totemism
				}
			}
			random_known_country = {
				limit = {
					government = native
					religion = totemism
				}
				add_opinion = {
					modifier = sio_feast_of_the_dead
					who = ROOT
				}
			}
			random_known_country = {
				limit = {
					government = native
					religion = totemism
				}
				add_opinion = {
					modifier = sio_feast_of_the_dead
					who = ROOT
				}
			}
		}
		else = {
			random_known_country = {
				add_opinion = {
					modifier = sio_feast_of_the_dead
					who = ROOT
				}
			}
			add_dip_power = 50
		}
		if = {
			limit = {
				has_idea_group = diplomatic_ideas
			}
			add_country_modifier = {
				name = sio_feast
				duration = 3650
			}
		}
	}
}
