	id = 4043
	title = "EVTNAME4043"
	desc = "EVTDESC4043"
	picture = MERCHANTS_TALKING_eventPicture

	trigger = {
		has_country_modifier = india_trade_co
		NOT = {
			any_owned_province = {
				OR = {
					region = west_india_region
					region = coromandel_region
					region = deccan_region
					region = bengal_region
					region = malaya_region
					region = moluccas_region
					region = indonesia_region
					region = indo_china_region
					region = burma_region
					region = japan_region
					region = south_china_region
					region = north_china_region
				}
				has_port = yes
			}
		}
	}

	mean_time_to_happen = {
		months = 24
		
		modifier = {
			factor = 0.9
			inflation = 10
		}
		modifier = {
			factor = 0.9
			inflation = 20
		}
		modifier = {
			factor = 0.9
			num_of_loans = 1
		}
		modifier = {
			factor = 0.9
			num_of_loans = 2
		}
		modifier = {
			factor = 1.1
			trader = 2
		}
		modifier = {
			factor = 1.1
			trader = 3
		}
		modifier = {
			factor = 1.1
			colonial_governor = 2
		}
		modifier = {
			factor = 1.1
			colonial_governor = 3
		}
	}

	option = {
		name = "EVTOPTA4043"		# What a disaster!
		remove_country_modifier = india_trade_co
		add_prestige = -2
	}
}
