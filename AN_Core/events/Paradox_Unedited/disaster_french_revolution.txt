country_event = {
	id = french_revolution.1
	title = "flavor_fra.EVTNAME3144"
	desc = "flavor_fra.EVTDESC3144"
	picture = LIBERUM_VETO_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = { log = "[Root.GetName] has the French Revolution Disaster." }
		if = {
			limit = {
				any_owned_province = {
					development = 20
					highest_supply_limit_in_area = yes
					is_in_capital_area = yes
					fort_level = 1
					is_capital = no
				}
			}
			random_owned_province = { 
				limit = {
					development = 20
					highest_supply_limit_in_area = yes
					is_in_capital_area = yes
					fort_level = 1
					is_capital = no
				}
				if = {
					limit = {
						NOT = { has_dlc = "Emperor" }
					}
					spawn_rebels = {
						type = revolutionary_rebels
						size = 3		
						win = yes
						unrest = 10
					}
				}
				else = {
					set_province_flag = revolution_spawn_rebels_here
				}
			}
		}
		else = {
			random_owned_province = { 
				limit = {
					highest_supply_limit_in_area = yes
					is_capital = no
				}
				if = {
					limit = {
						NOT = { has_dlc = "Emperor" }
					}
					spawn_rebels = {
						type = revolutionary_rebels
						size = 3		
						win = yes
						unrest = 10
					}
				}
				else = {
					set_province_flag = revolution_spawn_rebels_here
				}
			}
		}
		if = {
			limit = {
				NOT = { has_dlc = "Emperor" }
			}
			random_owned_province = {
				limit = {
					is_capital = yes
				}
				spawn_rebels = {
					type = revolutionary_rebels
					size = 2
				}
			}
		}
		add_stability = -3
		set_country_flag = had_revolution
		if = {
			limit = {
				is_bankrupt = yes
			}
			set_country_flag = no_revolution_bankruptcy_event
		}
	}

	option = {		# Dire times are ahead of us
		name = "EVTOPTA3011"
		trigger = { NOT = { has_dlc = "Emperor" } }
		add_prestige = -10
	}
	option = {
		name = revolution.1.A			# Crush them
		trigger = { has_dlc = "Emperor" }
		set_country_flag = revolution_disaster_reactionary
		random_owned_province = {
			limit = {
				has_province_flag = revolution_spawn_rebels_here
			}
			clr_province_flag = revolution_spawn_rebels_here
			spawn_rebels = {
				type = revolutionary_rebels
				size = 3		
				win = yes
				unrest = 10
			}
		}
		random_owned_province = {
			limit = {
				is_capital = yes
			}
			spawn_rebels = {
				type = revolutionary_rebels
				size = 2
			}
		}
		ai_chance = {
			factor = 1
		}
	}
	option = {
		name = revolution.1.B			# Vive la Revolution!
		trigger = { has_dlc = "Emperor" }
		set_country_flag = revolution_disaster_revolutionary
		set_country_flag = revolution_disaster_immediate_revolution
		custom_tooltip = revolution_disaster_go_for_it_tooltip
		random_owned_province = {
			limit = {
				has_province_flag = revolution_spawn_rebels_here
			}
			clr_province_flag = revolution_spawn_rebels_here
			spawn_rebels = {
				type = pretender_rebels
				size = 3		
				win = yes
				unrest = 10
			}
			if = {
				limit = {
					any_neighbor_province = {
						owned_by = root
					}
				}
				random_neighbor_province = {
					limit = { owned_by = root }
					spawn_rebels = {
						type = pretender_rebels
						size = 2
						win = yes
						unrest = 10
					}
				}
			}
		}
		hidden_effect = {
			country_event = { id = french_revolution.2 }
		}
		ai_chance = {
			factor = 2
			modifier = {
				factor = 2
				revolution_target_exists = no
			}
			modifier = {
				factor = 0
				is_emperor = yes
			}
		}
	}
}

	id = french_revolution.5
	title = "flavor_fra.EVTNAME3146"
	desc = {

			NOT = { has_dlc = "Emperor" }
		}
		picture = REVOLUTION_eventPicture
	}
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_country_flag = had_execution_of_king_louis_event_flag }
		OR = {
			has_country_modifier = the_reign_of_terror
			has_country_flag = the_directory_flag
		}
		is_revolutionary_republic_trigger = yes
		any_country = {
			capital_scope = {
				continent = root
			}
			NOT = { is_subject_of = root }
			NOT = { alliance_with = root }
			is_revolutionary = no
		}
	}

	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = had_execution_of_king_louis_event_flag
			clr_country_flag = ruler_of_france_exiled
			clr_country_flag = heir_of_france_exiled
			if = {
				limit = {
					any_rival_country = {
						is_neighbor_of = root
						is_revolutionary = no
					}
				}
				random_rival_country = {
					limit = {
						is_neighbor_of = root
						is_revolutionary = no
					}
					save_event_target_as = flee_to_country
				}
			}
			else_if = {
				limit = {
					any_rival_country = {
						government = monarchy
						is_revolutionary = no
					}
				}
				random_rival_country = {
					limit = {
						government = monarchy
						is_revolutionary = no
					}
					save_event_target_as = flee_to_country
				}
			}
			else_if = {
				limit = {
					any_rival_country = {
						exists = yes
						is_revolutionary = no
					}
				}
				random_rival_country = {
					limit = {
						exists = yes
						is_revolutionary = no
					}
					save_event_target_as = flee_to_country
				}
			}
			else = {
				random_country = {
					limit = {
						NOT = { is_subject_of = root }
						NOT = { alliance_with = root }
						is_revolutionary = no
						capital_scope = {
							continent = root
						}
						OR = {
							is_neighbor_of = root
							NOT = {
								ROOT = {
									any_neighbor_country = {
										NOT = { is_subject_of = root }
										capital_scope = {
											continent = root
										}
									}
								}
							}
						}
					}
					save_event_target_as = flee_to_country
				}
			}
		}
	}

		name = "flavor_fra.EVTOPTA3146"
		add_stability_or_adm_power = yes
		set_country_flag = fra_monarch_executed
		every_country = {
			limit = {
				government = monarchy
				capital_scope = {
					continent = root
				}
				NOT = {
					border_distance = {
						who = root
						distance = 150
					}
				}
			}
			add_opinion = {
				who = ROOT
				modifier = executed_monarch
			}
		}
		hidden_effect = {
			if = {
				limit = {
					any_owned_province = {
						continent = root
						is_core = ROOT
						has_port = yes
						is_capital = no
						controlled_by = root
					}
					any_owned_province = {
						continent = root
						is_core = ROOT
						is_capital = no
						religion = root
						controlled_by = root
					}
				}
