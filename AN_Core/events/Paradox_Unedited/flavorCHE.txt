# European Contact
country_event = {
	id = flavor_che.3
	title = flavor_che.3.t
	desc = flavor_che.3.d
	picture = PLAGUE_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Leviathan"
		tag = CHE
		if = {
			limit = {
				OR = {
					exists = CAS
					exists = SPA
					exists = ARA
				}
			}
			OR = {
				CAS = {
					any_owned_province = {
						continent = north_america
					}
				}
				SPA = {
					any_owned_province = {
						continent = north_america
					}
				}
				ARA = {
					any_owned_province = {
						continent = north_america
					}
				}
			}
		}		
		else = {
			NOT = { stability = 0 }
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = flavor_che.3.a
		add_stability = -1
	}
}

# Treaty With Foreign Traders
country_event = {
	id = flavor_che.4
	title = flavor_che.4.t
	desc = flavor_che.4.d
	picture = CORRUPTION_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Leviathan"
		tag = CHE
		is_year = 1684
		owns_core_province = 4879
		NOT = {
			any_owned_province = {
				has_province_modifier = che_foreigner_trade
			}
		}
		if = {
			limit = {
				exists = ENG
			}
			ENG = {
				NOT = { war_with = ROOT }
				any_owned_province = {
					continent = north_america
				}
			}
		}
		else_if = {
			limit = {
				exists = GBR
			}
			GBR = {
				NOT = { war_with = ROOT }
				any_owned_province = {
					continent = north_america
				}
			}
		}
		else_if = {
			limit = {
				exists = USA
			}
			USA = {
				NOT = { war_with = ROOT }
				any_owned_province = {
					continent = north_america
				}
			}
		}
		else = {
			is_at_war = no
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = flavor_che.4.a
		random_owned_province = {
			limit = { is_core = ROOT }
			add_province_modifier = {
				name = che_foreigner_trade
				duration = 7300
			}
		}
		if = {
			limit = {
				OR = {
					has_reform = native_war_band_reform
					has_reform = native_settle_down_reform
				}
			}			
			ROOT = {
				add_government_reform = native_trading_with_foreigners_reform 
			}
		}
	}
}

# Cherokee National Council
country_event = {
	id = flavor_che.5
	title = flavor_che.5.t
	desc = flavor_che.5.d
	picture = ELECTION_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Leviathan"
		tag = CHE
		OR = {
			AND = {
				has_reform = native_clan_council_reform
				total_development = 100
			}
			has_reform = native_becomes_republic_reform
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
		option = {
		name = flavor_che.5.a
		if = {
			limit = {
				NOT = { stability = 3}
			}
			add_stability = 1
			add_adm_power = 50
			add_reform_progress_small_effect = yes
		}
		else = {
			add_adm_power = 100
			add_reform_progress_small_effect = yes
		}
	}
}
