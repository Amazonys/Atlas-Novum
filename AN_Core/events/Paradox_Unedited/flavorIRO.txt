# An Unforgivable Affront
country_event = {
	id = flavor_iro.1
	title = flavor_iro.1.t
	desc = flavor_iro.1.d
	picture = DEATH_OF_HEIR_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		primary_culture = iroquois
		any_known_country = {
			religion = totemism
			NOT = {
				has_opinion = {
					who = ROOT
					value = 0
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 300
	}
	
	option = {
		name = flavor_iro.1.a
		add_manpower = -1
		set_country_flag = iro_unforgivable_affront
		random_neighbor_country = {
			limit = {
				religion = totemism
				NOT = { alliance_with = ROOT }
			}
			ROOT = {
				add_opinion = {	
					who = PREV
					modifier = opinion_iroquois_affront
				}
			}
		}
	}
}

# A New Captive
country_event = {
	id = flavor_iro.3
	title = flavor_iro.3.t
	desc = flavor_iro.3.d
	picture = LIBERUM_VETO_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		primary_culture = iroquois
		NOT = { has_country_flag = iro_adopted_captive }
		NOT = { has_country_flag = iro_iron_fist }
	}
	
	option = { #Banish him
		name = flavor_iro.3.a
		ai_chance = {
			factor = 0
		}
		add_adm_power = -50
		random_neighbor_country = {
			limit = {
				religion = totemism
			}
			add_opinion = {
				who = PREV
				modifier = iroquoian_iron_fist_opinion
			}
		}
		random_owned_province = {
			add_unrest = 1
		}
		add_country_modifier = {
			name = "iroquoian_iron_fist"
			duration = 3650
		}
		set_country_flag = iro_iron_fist
	}
	
	option = { #Adopt him
		name = flavor_iro.3.b
		ai_chance = {
			factor = 100
		}
		set_country_flag = iro_adopted_captive
		add_stability = 1
		capital_scope = {
			add_unrest = -2
		}
	}
}

# Mother of Nations
country_event = {
	id = flavor_iro.4
	title = flavor_iro.4.t
	desc = flavor_iro.4.d
	desc = flavor_iro.4.d
	picture = ADVISOR_eventPicture 
	
	fire_only_once = yes
	
	trigger = {
		primary_culture = iroquois
		had_recent_war = 2
		is_at_war = no
		has_country_flag = iroquois_take_captives
		NOT = { has_country_flag = iro_mother_of_nations }
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = iro_mother_of_nations
		}
	}
	
	mean_time_to_happen = {
		months = 120
	}
	
	option = {
		name = flavor_iro.4.a
		capital_scope = {
			add_province_modifier = {
				name = "mother_of_nations"
				duration = 1825
			}
		}
		every_owned_province = {
			if = {
				limit = {
					culture = iroquois
					NOT = { is_core = ROOT }
				}
				add_core = ROOT
			}
		}
		define_advisor = {
			type = statesman
			name = "Jigonsaseh"
			discount = yes
			skill = 2
			female = yes
			culture = iroquois 
			religion = totemism
		}
	}
}
