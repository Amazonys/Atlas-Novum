#Missing Consort! :o (Orpheus Quest start)
country_event = {
	id = totemism_flavor.1
	title = totemism_flavor.1.t
	desc = totemism_flavor.1.d
	picture = BAD_WITH_MONARCH_eventPicture
	
	trigger = {
		has_dlc = "Leviathan"
		religion = totemism
		is_female = no
		has_consort = yes
		NOT = { has_country_flag = orpheus_quest }
		NOT = { has_country_flag = adventurous_ruler }
	}
	
	is_triggered_only = yes
	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 1
		
		modifier = {
			factor = 0.75
			OR = {
				consort_has_personality_ancestor = { key = secretive }
				consort_has_personality = secretive_personality
			}
		}
	}
	
	option = {
		name = totemism_flavor.1.a # Send people to find her
		ai_chance = { factor = 50 }
		add_manpower = -1
		set_country_flag = orpheus_quest
	}
	
	option = {
		name = totemism_flavor.1.b # Hire people to find her
		ai_chance = { factor = 25 }
		add_treasury = -15
		set_country_flag = orpheus_quest
	}
	
	option = {
		name = totemism_flavor.1.c # Send ruler
		ai_chance = { factor = 10 }
		capital_scope = {
			add_unrest = 1
		}
		set_country_flag = orpheus_quest
		set_country_flag = adventurous_ruler
	}

	option = {
		name = totemism_flavor.1.e # We'll find a new consort
		ai_chance = { factor = 15 }
		divorce_consort_effect = yes
	}
}

# Consort Sighted! (Orpheus Quest 2)
country_event = {
	id = totemism_flavor.2
	title = totemism_flavor.2.t
	desc = totemism_flavor.2.d
	picture = EXPLORERS_eventPicture
	
	trigger = {
		has_dlc = "Leviathan"
		religion = totemism
		any_owned_province = {
			is_capital = no
			religion = totemism
			num_of_units_in_province = {
				who = ROOT
				amount = 0.1
			}
		}
		has_country_flag = orpheus_quest
	}
	
	immediate = {
		clr_country_flag = orpheus_quest
		hidden_effect = {
			clr_country_flag = orpheus_quest
			random_owned_province = {
				limit = {
					religion = totemism
					is_capital = no
				}
				save_event_target_as = orpheus_province
			}
		}
	}
	
	mean_time_to_happen = {
		months = 2
	}
	
	option = { # She demands to see the ruler
		name = totemism_flavor.2.a
		trigger = {
			NOT = { has_country_flag = adventurous_ruler }
		}
		country_event = {
			id = totemism_flavor.3 
			days = 10
		}
	}
	
	option = { # Take her by force
		name = totemism_flavor.2.e
		trigger = {
			event_target:orpheus_province = {
				num_of_units_in_province = {
					who = ROOT
					amount = 2
				}
			}
		}
		event_target:orpheus_province = {
			add_unrest = -1
		}
		hidden_effect = {
			country_event = {
				id = totemism_flavor.5
			}
		}
		set_country_flag = taken_by_force
	}
	
	option = { # Let her go
		name = totemism_flavor.2.b
		divorce_consort_effect = yes
	}
	
	option = { # Ruler finds her!
		name = totemism_flavor.2.c
		trigger = {
			has_country_flag = adventurous_ruler
		}
		hidden_effect = {
			random_list = {
				90 = { # Gods let her come back
					country_event = {
						id = totemism_flavor.3
					}
				}
				10 = { # She runs away
					country_event = {
						id = totemism_flavor.5
					}
					divorce_consort_effect = yes
				}
			}
		}
	}
}

# Exploration of Legendary Location (finding figurines etc)
country_event = {
	id = totemism_flavor.9
	title = totemism_flavor.9.t
	desc = totemism_flavor.9.d
	picture = ARTWORK_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			capital_scope = {
				random_neighbor_province = {
					limit = {
						has_province_modifier = tot_legendary_location
					}
					save_event_target_as = legendary_location
				}
			}
		}
	}
	
	option = {
		name = totemism_flavor.9.a
		if = {
			limit = {
				has_estate = estate_church
			}
			add_estate_loyalty = {
				estate = estate_church
				loyalty = 10
			}
		}
		if = {
			limit = {
				OR = {
					has_reform = native_becomes_monarchy_reform
					has_reform = native_becomes_republic_reform
					has_reform = native_becomes_theocracy_reform
					has_reform = native_becomes_horde_reform
				}
				event_target:legendary_location = {	
					NOT = { has_building = temple }
				}
			}
			event_target:legendary_location = {	
				add_building = temple
			}
		}
		else_if = {
			limit = {
				event_target:legendary_location = {	
					NOT = { has_building = native_ceremonial_fire_pit }
				}
				government = native
			}
			event_target:legendary_location = {	
				add_building = native_ceremonial_fire_pit
			}
		}
		else_if = {
			limit = {
				government = native
				event_target:legendary_location = {	
					NOT = { has_building = native_longhouse }
				}
			}
			event_target:legendary_location = {	
				add_building = native_longhouse
			}
		}
		else = {
			add_prestige = 10
		}
		set_country_flag = legendary_location_explored
	}
}

# Maintenance of Legendary Location
country_event = {
	id = totemism_flavor.10
	title = totemism_flavor.10.t
	desc = totemism_flavor.10.d
	picture = EXPLORERS_eventPicture
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	trigger = {
		has_dlc = "Leviathan"
		religion = totemism
		has_country_flag = legendary_location_explored
		NOT = {
			any_owned_province = {
				has_province_modifier = tot_legendary_location_maintained
			}
		}
		capital_scope = {
			any_neighbor_province = {
				has_province_modifier = tot_legendary_location
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			capital_scope = {
				random_neighbor_province = {
					limit = {
						has_province_modifier = tot_legendary_location
					}
					save_event_target_as = legendary_location
				}
			}
		}
	}
	
	option = { 	# appoint advisor to take care of it
		name = totemism_flavor.10.a
		trigger = {
			OR = {
				has_adm_advisor = yes
				has_dip_advisor = yes
			}
		}
		if = {
			limit = {
				has_adm_advisor = yes
			}
			remove_advisor_by_category = ADM
		}
		else_if = {
			limit = {
				has_dip_advisor = yes
			}
			remove_advisor_by_category = DIP 
		}
		event_target:legendary_location = {
			add_province_modifier = {
				name = tot_legendary_location_maintained
				duration = 1825
			}
		}
	}
	
	option = { # appoint heir 
		name = totemism_flavor.10.b
		trigger = {
			has_heir = yes
		}
		random_list = {
			50 = { }
			50 = { change_heir_adm = 1 }
		}
		if = {
			limit = {
				OR = {
					heir_has_good_adm_personality = yes
					heir_has_personality_ancestor = { key = pious }
				}
			}
			add_heir_personality = pious_personality
		}
		event_target:legendary_location = {
			add_province_modifier = {
				name = tot_legendary_location_maintained
				duration = 1825
			}
		}
		hidden_effect = { country_event = { id = totemism_flavor.12 days = 5474 } }
	}
	
	option = { # donate some money
		name = totemism_flavor.10.c
		add_treasury = -10
		event_target:legendary_location = {
			add_province_modifier = {
				name = tot_legendary_location_maintained
				duration = 1825
			}
		}
	}
	
	option = { # it will be fine like this
		name = totemism_flavor.10.e
	}
}
