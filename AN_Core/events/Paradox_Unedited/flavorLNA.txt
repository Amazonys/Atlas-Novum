
namespace = flavor_lna
normal_or_historical_nations = yes

# The Emerald Buddha by Daniel Tolman
country_event = {
	id = flavor_lna.1
	title = flavor_lna.1.t
	desc = flavor_lna.1.d
	picture = BUDDHA_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		religion_group = eastern
		FROM = {
			OR = {
				has_province_modifier = emerald_buddha
				has_province_modifier = emerald_buddha_alternate
			}
			NOT = {
				owned_by = ROOT
			}
			NOT = {
				owner = {
					OR = {
						is_in_war = {
							defenders = PREV
							defenders = ROOT
						}
						is_in_war = {
							attackers = PREV
							attackers = ROOT
						}
					}
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			FROM = {
				save_global_event_target_as = emerald_buddha_province
				remove_province_modifier = emerald_buddha
				remove_province_modifier = emerald_buddha_alternate
			}
		}
	}
	
	option = {
		name = flavor_lna.1.a
		add_prestige = 10
		hidden_effect = {
			FROM = {
				owner = {
					country_event = {
						id = flavor_lna.2
					}
				}
			}
		}
		custom_tooltip = emerald_buddha_tt
		capital_scope = {
			add_permanent_province_modifier = {
				name = emerald_buddha
				duration = -1
			}
		}
	}
}

# Emerald Buddha lost
country_event = {
	id = flavor_lna.2
	title = flavor_lna.2.t
	desc = flavor_lna.2.d
	picture = BUDDHA_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_lna.2.a
		custom_tooltip = emerald_buddha_tt
		if = {
			limit = {
				religion_group = eastern
			}
			add_prestige = -20
			add_legitimacy = -10
		}
		else = {
			add_prestige = -5
		}
	}
	
	after = {
		clear_global_event_target = emerald_buddha_province
	}
}

# Siamese Absolutism
country_event = {
	id = flavor_lna.3
	title = flavor_lna.3.t
	desc = flavor_lna.3.d
	picture = BIG_BOOK_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_lna.3.a
		if = {
			limit = {
				NOT = {
					government = monarchy
				}
			}
			change_government = monarchy
		}
		add_government_reform = siamese_absolutism
	}
	
	option = {
		name = flavor_lna.3.b
		add_prestige = 25
	}
}
