# The Marathi Revolt
country_event = {
	id = flavor_dlh.1
	title = "flavor_dlh.EVTNAME1"
	desc = "flavor_dlh.EVTDESC1"
	picture = ANGRY_MOB_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		religion_group = muslim
		is_year = 1650
		NOT = { tag = MUG }
		NOT = { exists = MAR } # Marathas
		any_owned_province = {
			OR = {
				area = maidan_area
				area = desh_area
				culture = marathi
			}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_dlh.EVTOPTA1"
		every_owned_province = {
			limit = {
				OR = {
					area = maidan_area
					area = desh_area
					culture = marathi
				}
			}
			add_core = MAR
			add_unrest = 10
		}
	}
	option = {
		name = "flavor_dlh.EVTOPTB1"
		every_owned_province = {
			limit = {
				OR = {
					area = maidan_area
					area = desh_area
					culture = marathi
				}
				is_capital = no
			}
			add_local_autonomy = 100
		}
	}
}

# The Rise of Bahlul Lodi

	id = flavor_dlh.5
	title = "flavor_dlh.5.t"
	desc = "flavor_dlh.5.d"
	picture = COURT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = DLH
		is_at_war = no
		SRH = {
			dynasty = "Lodi"

		}
		NOT = { has_global_flag = dlh_bahlul_lodi_flag }
		is_year = 1445
		NOT = { is_year = 1450 }
	}
	
	immediate = {
		set_global_flag = dlh_bahlul_lodi_flag
	}
	
	mean_time_to_happen = {
		months = 3
	}
	
	option = {
		name = "flavor_dlh.5.a" #We must rein him in.

			factor = 100
		}
		custom_tooltip = flavor_dlh.5.a.tt

		add_prestige = 10
	}
	option = {
		name = "flavor_dlh.5.b" #We have lost control over Sirhind and the Punjab.

			factor = 0
		}
