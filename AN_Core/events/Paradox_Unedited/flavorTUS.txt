#The emperor condemns the Duchy's elevation
country_event = {
	id = flavor_tus.2
	title = flavor_tus.2.t
	desc = flavor_tus.2.d
	picture = REFORM_eventPicture

	fire_only_once = yes
	
	trigger = {
		tag = TUS
		NOT = {
			is_year = 1600
		}
		is_emperor = no
		NOT = {
			reverse_has_opinion = {
				who = emperor
				value = 75 
			}
		}
		reverse_has_opinion = {
			who = PAP
			value = 50
		}
		religion = catholic
		NOT = {
			num_of_cities = 6
		}
		government = monarchy
	}

	option = {
		name = flavor_tus.2.a
		ai_chance = {
			factor = 80
		}
		emperor = {
			add_opinion = {
				who = TUS
				modifier = opinion_emperor_tus_refused
			}
		}
		PAP = {
			add_opinion = {
				who = TUS
				modifier = opinion_papal_tus_refused
			}
		}
		add_legitimacy = -10
		add_prestige = 10
		add_country_modifier = {
			name = emperor_condemns_tuscany
			duration = 7300
		}
	}

	option = {
		name = flavor_tus.2.b 	#We yield to the emperor
		add_prestige = -15
		emperor = {
			add_opinion = {
				who = TUS
				modifier = opinion_emperor_tus_yielded
			}
		}
		PAP = {
			add_opinion = {
				who = TUS
				modifier = opinion_papal_tus_yielded
			}
		}
	}
}

#Savonarola Rises
country_event = {
	id = flavor_tus.9
	title = flavor_tus.9.t
	desc = flavor_tus.9.d
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = BONFIRE_OF_THE_VANITIES_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = RELIGIOUS_TURMOIL_eventPicture
	}

	fire_only_once = yes
	is_triggered_only = yes	
	
	immediate = {
		hidden_effect = {
			set_country_flag = savonarola_disaster_flag
			set_variable = {
				which = savonarola_support_variable
				value = 0
			}
			set_variable = {
				which = savonarola_decisions_variable
				value = 0
			}
			set_country_flag = bonfire_of_the_vanities_disaster_flag
		}
	}
		
	#Ruling Family Flees!
	option = {
		name = flavor_tus.9.a
		ai_chance = {
			factor = 75
		}
		if = {
			limit = {
				government = monarchy
			}
			hidden_effect = {
				if = {
					limit = {
						has_heir = yes
					}
					exile_heir_as = flo_exiled_heir
					set_country_flag = flo_exiled_heir_flag
				}
				exile_ruler_as = { name = flo_exiled_ruler }
			}
			define_ruler = {
				name = "Girolamo"
				dynasty = "Savonarola"
				age = 40
				ADM = 5
				DIP = 5
				MIL = 3
				claim = 50
			}
			hidden_effect = {
				add_ruler_personality = infertile_personality
				add_ruler_personality_ancestor = { key = zealot }
			}
		}
		else = {
			exile_ruler_as = { name = flo_exiled_ruler }
			define_ruler = {
				name = "Girolamo Savonarola"
				age = 40
				ADM = 5
				DIP = 5
				MIL = 3
			}
			hidden_effect = {
				add_ruler_personality_ancestor = { key = zealot }
			}
			add_government_reform = millenarian_theocracy_reform
		}
		set_ruler_flag = savonarola_flag
		add_republican_tradition = 20
	}
}

# Event about Papal disapproval
country_event = {
	id = flavor_tus.19
	title = flavor_tus.19.t
	desc = flavor_tus.19.d
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = BONFIRE_OF_THE_VANITIES_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = POPE_PREACHING_eventPicture
	}
	
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		has_ruler_flag = savonarola_flag
		had_country_flag = {
			flag = bonfire_of_the_vanities_disaster_flag
			days = 90
		}
	}
	
	option = { # Declare the Pope a heretic
		name = flavor_tus.19.a
		ai_chance = { factor = 100 }
		excommunicate = ROOT
		add_papal_influence = -200
		if = {
			limit = {
				PAP = {
					owns = 118
				}
			}
