namespace = granada_civil_war

# granada_civil_war
country_event = {
	id = granada_civil_war.1
	title = "granada_civil_war.1.t"
	desc = "granada_civil_war.1.desc"
	picture = CIVIL_WAR_eventPicture

	major = yes
	# this = country seeing popup, from = original event scope
	major_trigger = {
		capital_scope = {
			OR = {
				region = iberia_region
				area = northern_morocco_area
				area = western_morocco_area
				area = marrekesh_area
			}
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					any_owned_province = {
						is_core = ROOT
						is_capital = no
					}
				}
				random_owned_province = {
					limit = {
						is_core = ROOT 
						is_capital = no
					}
					spawn_rebels = {
						type = pretender_rebels
						religion = ROOT
						size = 2
						leader = "Yûsuf"
						leader_dynasty = "Nasrid"
						win = yes
					}
					add_local_autonomy = 25
				}
			}
			else = {
				capital_scope = {
					spawn_rebels = {
						type = pretender_rebels
						religion = ROOT
						size = 2
						leader = "Yûsuf"
						leader_dynasty = "Nasrid"
					}
				}
			}
			set_country_flag = had_granada_succession_war
		}
	}
	
	
	option = {
		name = "granada_civil_war.1.a"			# Crush them!
		add_prestige = -10
	}
}

			adm = 2
			dip = 1
			mil = 4
			culture = ROOT
			religion = ROOT
			block_disinherit = yes
		}
		hidden_effect = {

		}
	}
