# Sunken Treasure
country_event = {
	id = pirates.10
	title = "pirates.10.t"
	picture = FLAGSHIP_SUNK_eventPicture
	desc = "pirates.10.d"
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			has_dlc = "Golden Century"
			has_dlc = "Lions of the North"
		}
		has_government_attribute = is_pirate_republic_reform
		capital_scope = {
			region = carribeans_region
		}
		if = {
			limit = {
				has_dlc = "El Dorado"
			}
			carribeans_region = {
				recent_treasure_ship_passage = yes
			}
		}
		any_country = {
			capital_scope = {
				continent = europe
			}
			any_subject_country = { # potosi or mexico
				OR = {
					owns_core_province = 852
					owns_core_province = 795
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					capital_scope = {
						continent = europe
					}
					any_subject_country = { # potosi or mexico
						OR = {
							owns_core_province = 852
							owns_core_province = 795
						}
					}
				}
				save_event_target_as = sunken_ship_country
			}
		}
	}
	
	option = {
		name = "pirates.10.a"
		add_faction_influence = {
			faction = pr_captains
			influence = 10
		}
	}
	
	option = {
		name = "pirates.10.b"
		add_faction_influence = {
			faction = pr_buccaneers
			influence = 10
		}
		add_yearly_sailors = -5
		random_list = {
			10 = {
				add_years_of_income = 2
			}
			90 = {
				# nothing happens
			}
		}
	}
}

# Privateer turns pirate
country_event = {
	id = pirates.14
	title = "pirates.14.t"
	picture = PIRATE_CAPTAINS_eventPicture
	desc = "pirates.14.d"
	
	is_triggered_only = yes
	
	trigger = {
		OR = {
			has_dlc = "Golden Century"
			has_dlc = "Lions of the North"
		}
		has_government_attribute = is_pirate_republic_reform
		num_of_ports = 1
		any_country = {
			num_of_heavy_ship = 3
			any_trade_node = {
				privateer_power = { country = prev share = 1 }
			}
			capital_scope = {
				has_discovered = ROOT
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					num_of_heavy_ship = 3
					any_trade_node = {
						privateer_power = { country = prev share = 1 }
					}
					capital_scope = {
						has_discovered = ROOT
					}
				}
				save_event_target_as = privateer_origin_country
			}
		}
		hidden_effect = {
			set_saved_name = {
				key = defecting_privateer
				type = advisor
				scope = event_target:privateer_origin_country
			}
		}
	}
	
	option = {
		name = "pirates.14.a"
		add_faction_influence = {
			faction = pr_captains
			influence = 10
		}
		create_admiral = {
			tradition = 50
			name = defecting_privateer
		}
		if = {
			limit = {
				capital_scope = {
					has_port = yes
				}
			}
			capital_scope = {
				heavy_ship = ROOT
			}
		}
		else = {
			random_owned_province = {
				limit = {
					has_port = yes
				}
				heavy_ship = ROOT
			}
		}
		reverse_add_opinion = {
			who = event_target:privateer_origin_country
			modifier = opinion_defected_privateer
		}
	}
}
