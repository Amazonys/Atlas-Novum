# Bribing the Sejm
country_event = {
	id = elective_monarchy.3
	title = elective_monarchy.3.t
	desc = elective_monarchy.3.d
	picture = SEJM_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Res Publica"
		government = monarchy
		years_of_income = 0.5
		if = {
			limit = {
				exists = POL
			}
			knows_country = POL
			POL = {
				has_government_attribute = has_polish_election
				heir_nationality = ROOT
			}
		}
		else_if = {
			limit = {
				exists = PLC
			}
			knows_country = PLC
			PLC = {
				has_government_attribute = has_polish_election
				heir_nationality = ROOT
			}
		}
		else = {
			any_known_country = {
				has_government_attribute = has_polish_election
				heir_nationality = ROOT
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}	

	option = {
		name = elective_monarchy.3.a
		add_years_of_income = -0.5
		add_prestige = 3
		random_known_country = {
			limit = {
				has_government_attribute = has_polish_election
				heir_nationality = ROOT
			}
			add_heir_support = 25
		}		
	}
	option = {
		name = elective_monarchy.3.b
		add_prestige = -1
		random_known_country = {
			limit = {
				has_government_attribute = has_polish_election
				heir_nationality = ROOT
			}
			add_heir_support = -10
		}
	}	
}

				exists = PLC
			}

			}
		}
		else = {
			any_known_country = {

		}
	}

# [Root.Monarch.GetName] Inherits Title
country_event = {
	id = elective_monarchy.9
	title = elective_monarchy.9.t
	desc = elective_monarchy.9.d
	picture = GOOD_WITH_MONARCH_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Res Publica"
		has_government_attribute = has_polish_election
		ruler_is_foreigner = yes
		any_known_country = {
			dynasty = ROOT
			is_subject = no
			same_continent = root
			any_owned_province = {
				NOT = { is_claim = ROOT }
				NOT = { is_core = ROOT }
				is_in_capital_area = yes
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		random_known_country = {
			limit = {
				dynasty = ROOT
				is_lesser_in_union = no
				same_continent = root
				any_owned_province = {
					NOT = { is_claim = ROOT }
					NOT = { is_core = ROOT }
					is_in_capital_area = yes
				}
			}
			random_owned_province = {
				limit = {
					NOT = { is_claim = ROOT }
					NOT = { is_core = ROOT }
					is_in_capital_area = yes
				}
				save_event_target_as = claim_province
			}
		}
	}

	option = {
		name = elective_monarchy.9.a
		goto = claim_province
		event_target:claim_province = {
			random_owned_province = {
				limit = {
					NOT = { is_claim = ROOT }
					NOT = { is_core = ROOT }
					is_in_capital_area = yes
				}
				add_claim = ROOT
			}
		}
	}

	option = {
		name = elective_monarchy.9.b
		add_prestige = -5
	}
}
