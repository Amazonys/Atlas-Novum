# The Nieszawa Privileges-'Conventiones Particulares'
country_event = {
	id = flavor_pol.3467
	title = "flavor_pol.EVTNAME3467"
	desc = "flavor_pol.EVTDESC3467"
	picture = LIBERUM_VETO_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		#NOT = {	num_of_estate_privileges = { estate = estate_nobles	value = 4 } }
		OR = {
			AND = {
				tag = POL
				NOT = { exists = PLC }
			}
			tag = PLC
		}
		is_year = 1454
		NOT = { is_year = 1485 }
	}
	
	mean_time_to_happen = {
		months = 50
	}
	
	immediate = { 
		hidden_effect = { 
			set_country_flag = pol_had_nieszawa_privileges_country_flag 
			random_owned_province = {
				limit = {
					continent = europe
				}
				save_event_target_as = pol_reble_province
			}
		} 
	}
	option = {
		name = "flavor_pol.EVTOPTA3467" # Accept the Conventions
		if = {
			limit = {
				NOT = { has_estate_privilege = estate_nobles_nieszawa_privileges }
			}
			if = {
				limit = {
					num_of_estate_privileges = { estate = estate_nobles	value = 6 }
				}
				if = {
					limit = {
						has_estate_privilege = estate_nobles_nobility_primacy
					}
					remove_estate_privilege = estate_nobles_nobility_primacy
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_land_rights
					}
					remove_estate_privilege = estate_nobles_land_rights
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_officer_corp
					}
					remove_estate_privilege = estate_nobles_officer_corp
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_levies
					}
					remove_estate_privilege = estate_nobles_levies
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_advisors
					}
					remove_estate_privilege = estate_nobles_advisors
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_right_of_counsel
					}
					remove_estate_privilege = estate_nobles_right_of_counsel
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_supremacy_over_crown
					}
					remove_estate_privilege = estate_nobles_supremacy_over_crown
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_supremacy_over_crown
					}
					remove_estate_privilege = estate_nobles_supremacy_over_crown
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_strong_duchies
					}
					remove_estate_privilege = estate_nobles_strong_duchies
				}
				else_if = {
					limit = {
						has_estate_privilege = estate_nobles_better_integration
					}
					remove_estate_privilege = estate_nobles_better_integration
				}
			}
			set_estate_privilege = estate_nobles_nieszawa_privileges
		}
		else = { set_estate_privilege = estate_nobles_nieszawa_privileges }
	}
	option = {
		name = "flavor_pol.EVTOPTB3467" # Refuse to sign it and defy the Magnates
		goto = pol_reble_province
		event_target:pol_reble_province = {
			polish_noble_rebels = 3
		}
	}
}

			}
		}
