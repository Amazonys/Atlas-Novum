# Turko-Polish Tension
country_event = {
	id = flavor_pol.3488
	title = "flavor_pol.EVTNAME3488"
	desc = "flavor_pol.EVTDESC3488"
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		OR = {
			AND = {
				tag = POL
				NOT = { exists = PLC }
			}
			tag = PLC
		}
		religion_group = christian
		exists = TUR
		NOT = { alliance_with = TUR }
		NOT = { is_subject_of = TUR }

		NOT = { has_country_modifier = pol_tur_tension_timer }
		OR = {
			AND = {
				is_neighbor_of = TUR
				TUR = {
					any_owned_province = {
						any_neighbor_province = {
							owned_by = ROOT
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
				}
			}
			any_known_country = {
				vassal_of = ROOT
				is_neighbor_of = TUR
				TUR = {
					any_owned_province = {
						any_neighbor_province = {
							owner = {
								vassal_of = ROOT
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
				}
			}
			any_known_country = {
				OR = {
					junior_union_with = ROOT
					senior_union_with = ROOT
				}
				is_neighbor_of = TUR
				TUR = {
					any_owned_province = {
						any_neighbor_province = {
							owner = {
								OR = {
									junior_union_with = ROOT
									senior_union_with = ROOT
								}
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = "flavor_pol.EVTOPTA3488" # We will defend Christendom
		add_country_modifier = {
			name = pol_tur_tension_timer
			duration = 3650
			hidden = yes
		}
		add_yearly_manpower = 1
		if = {
			limit = {
				is_neighbor_of = TUR
			}
			TUR = {
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owned_by = ROOT
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
		}
		if = {
			limit = {
				NOT = { is_neighbor_of = TUR }
				any_known_country = {
					vassal_of = ROOT
					is_neighbor_of = TUR
				}
			}
			TUR = {
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owner = {
								vassal_of = ROOT
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
		}
		if = {
			limit = {
				NOT = { is_neighbor_of = TUR }
				NOT = {
					any_known_country = {
						vassal_of = ROOT
						is_neighbor_of = TUR
					}
				}
				any_known_country = {
					OR = {
						junior_union_with = ROOT
						senior_union_with = ROOT
					}
					is_neighbor_of = TUR
				}
			}
			TUR = {
				random_owned_province = {
					limit = {
						any_neighbor_province = {
							owner = {
								OR = {
									junior_union_with = ROOT
									senior_union_with = ROOT
								}
							}
						}
						NOT = { is_core = ROOT }
						NOT = { is_claim = ROOT }
					}
					add_claim = ROOT
				}
			}
		}
	}
}
