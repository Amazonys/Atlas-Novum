# The yearly Tatar Yoke Tribute
country_event = {
	id = flavor_mos.3
	title = flavor_mos.3.t
	desc = flavor_mos.3.desc
	picture = TRIBUTE_TO_TATARS_eventPicture
	is_triggered_only = yes

	trigger = {
		has_dlc = "Domination"
		NOT = { has_global_flag = end_of_the_tatar_yoke }
		has_country_modifier = the_tatar_yoke
		466 = {
			owner = {
				NOT = { war_with = ROOT }
			}
		}
	}

	mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            export_to_variable = {
                which = mos_tribute_to_pay
                value = monthly_income
                who = ROOT
            }
            multiply_variable = {
                which = mos_tribute_to_pay
                value = 2
            }
        }
    }

	#Pay the Annual Tribute
	option = {
		name = flavor_mos.3.a
		custom_tooltip = flavor_mos.3.a.tt
		hidden_effect = {
			466 = {
				owner = {
					country_event = {
						id = flavor_mos.4
					}
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				OR = {
					NOT = { years_of_income = 0.3 }
					466 = {
						owner = {
							OR = {
								is_at_war = yes
								NOT = { manpower_percentage = 0.5 }
								NOT = { mil_power_cost = 50 }
							}
						}
					}
				}
			}
		}
	}

	#Refuse to pay the Annual Tribute
	option = {
		name = flavor_mos.3.b
		custom_tooltip = flavor_mos.3.b.tt
		hidden_effect = {
			466 = {
				owner = {
					country_event = {
						id = flavor_mos.5
					}
				}
			}
		}
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				years_of_income = 0.3
				466 = {
					owner = {
						is_at_war = no
						manpower_percentage = 0.5
						mil_power_cost = 50
					}
				}
			}
		}
	}
}

# Third Period of the Muscovite Civil War - The blinding of Vasily II
country_event = {
	id = flavor_mos.7
	title = flavor_mos.7.t
	desc = flavor_mos.7.desc
	picture = BATTLE_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		has_dlc = "Domination"
		tag = MOS
		has_ruler_flag = has_vasily_ii
		num_of_cities = 2
	}
	
	mean_time_to_happen = {
		months = 50
	}
	
	#
	option = {
		name = flavor_mos.7.a
		exile_ruler_as = {
			name = vasily_ii
		}
		exile_heir_as = ivan_veliky
		hidden_effect = {
			if = {
				limit = { has_leader = "Dmitry Shemiaka Rurikovich" }
				define_leader_to_ruler = {
					name = "Dmitry Shemiaka Rurikovich"
					adm = 2
					dip = 1
					mil = 5
					claim = 40
				}
				kill_leader = {
					type = "Dmitry Shemiaka Rurikovich"
				}
			}
			else = {
				define_ruler = {
					name = "Dmitry Shemiaka"
					dynasty = "Rurikovich"
					min_age = 20
					max_age = 40
					adm = 2
					dip = 1
					mil = 5
					claim = 40
				}
			}
			set_ruler_flag = dmitry_shemiaka_in_power
		}
		add_ruler_modifier = {
			name = vasily_ii_in_exile
			duration = -1
			desc = vasily_ii_in_exile_desc
		}
		
		ai_chance = {
			factor = 1
		}
	}
}

#The Rule of Dmitry
country_event = {
	id = flavor_mos.8
	title = flavor_mos.8.t
	desc = flavor_mos.8.desc
	picture = BATTLE_eventPicture
	fire_only_once = yes
	goto = capital
	trigger = {
		tag = MOS
		has_ruler_flag = dmitry_shemiaka_in_power
		had_ruler_flag = {
			flag = dmitry_shemiaka_in_power
			days = 730
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	immediate = {
		hidden_effect = {
			capital_scope = {
				save_event_target_as = capital
			}
		}
	}
	
	#Move against Dmitry!
	option = {
		name = flavor_mos.8.a
		set_ruler = vasily_ii
		set_heir = ivan_veliky
		capital_scope = {
			spawn_rebels = {
				type = pretender_rebels
				size = 2
				leader = "Dmitry Shemiaka"
				leader_dynasty = "Rurikovich"
			}
		}
		set_country_flag = dmitry_shemiaka_defeated
		ai_chance = {
			factor = 1
		}
	}
	
	#Abdicate the throne to Dmitry
	option = {
		name = flavor_mos.8.b
		remove_country_modifier = vasily_ii_in_exile
		add_legitimacy = 20
		set_country_flag = mos_civil_war_ended_flag
		ai_chance = {
			factor = 0
		}
	}
}
