				add_permanent_claim = ROOT
			}
			lower_silesia_area = {
				add_permanent_claim = ROOT
			}
			hidden_effect = {
				remove_casus_belli = {
					type = cb_restore_personal_union
					target = BOH
				}
			}
		}
	}

    develop_silesia = {
	icon = mission_war_chest
	required_missions = { recover_silesia }
    position = 5

		      silesia_area = {

			silesia_area = {
              add_base_tax = 2

			lower_silesia_area = {
              add_base_tax = 2
