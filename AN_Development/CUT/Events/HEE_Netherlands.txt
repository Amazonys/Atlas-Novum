#Dutch events (Netherlands++)

namespace = hee_NED

#Generaliteitslanden
province_event = {
	id = hee_NED.1
	title = hee_NED.1.t
	desc = hee_NED.1.d
	picture = CITY_DEVELOPMENT_eventPicture

	trigger = {
		owner = {
			tag = NED
			OR = {
				religion = protestant
				religion = reformed
			}
			government = republic
			NOT = {
				has_idea_group = humanist_ideas
			}
			is_at_war = no
		}
		has_owner_religion = no
		OR = {
			culture = dutch
			culture = wallonian
			culture = flemish
		}
		is_capital = no
		NOT = {
			has_province_modifier = hee_generality_lands
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = {
		name = hee_NED.1.a
		add_province_modifier = {
			name = hee_generality_lands
			duration = -1
		}
	}
}

#The revolt of Adolf van Egmond
country_event = {
	id = hee_NED.3
	title = hee_NED.3.t
	desc = hee_NED.3.d
	picture = BATTLE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = GEL
		NOT = {
			is_year = 1480
		}
		exists = BUR
		NOT = {
			num_of_cities = 4
		}
		owns = 99
		is_year = 1450
	}

	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.8
			NOT = {
				legitimacy = 80
			}
		}
	}

	option = {
		name = hee_NED.3.a
		random_owned_province = {
			spawn_rebels = {
				type = noble_rebels
				size = 2
				leader = "Adolf van Egmond"
			}
		}
		hidden_effect = {
			add_country_modifier = {
				name = hee_adolf_revolted
				duration = 3650
				hidden = yes
			}
		}
	}

	option = {
		name = hee_NED.3.b
		add_dip_power = -150
	}
}

#Compromising with Adolf
country_event = {
	id = hee_NED.4
	title = hee_NED.4.t
	desc = hee_NED.4.d
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes

	trigger = {
		has_country_modifier = hee_adolf_revolted
		is_subject = no
		OR = {
			any_country = {
				junior_union_with = BUR
			}
			BUR = {
				num_of_cities = 15
			}
		}
		is_at_war = no
		BUR = {
			is_at_war = no
		}
	}

	mean_time_to_happen = {
		months = 30
	}

	option = {
		name = hee_NED.4.a
		BUR = {
			country_event = { id = hee_NED.5 days = 10 }
		}
		add_prestige = -10
		custom_tooltip = hee_NED.4.tt
	}

	option = {
		name = hee_NED.4.b
		99 = {
			add_province_modifier = {
				name = hee_adolfs_unrest
				duration = 3650
			}
		}
		add_legitimacy = 5
	}
}
