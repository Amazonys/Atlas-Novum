#St. Francis of Xavier
country_event = {
	id = hee_NAV.10
	title = hee_NAV.10.t
	desc = hee_NAV.10.d
	picture = RELIGIOUS_CONVERSION_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = NAV
			AND = {
				NOT = {
					exists = NAV
				}
				tag = SPA
			}
		}
		is_year = 1500
		NOT = {
			is_year = 1551
		}
		OR = {
			religion = catholic
			religion = protestant
			religion = reformed
		}
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.9
			has_idea_group = religious_ideas
		}
	}

	option = { #He should serve as a minister
		name = hee_NAV.10.a
		ai_chance = {
			factor = 15
		}
		define_advisor = {
			type = inquisitor
			name = "Francis of Xavier"
			skill = 2
			discount = yes
		}
	}

	option = { #God leads his path
		name = hee_NAV.10.b
		ai_chance = {
			factor = 85
		}
		add_dip_power = 150
		hidden_effect = {
			random_province = {
				limit = {
					is_capital = no
					OR = {
						region = japan_region
						region = indonesia_region
						region = deccan_region
					}
				}
				province_event = { id = hee_NAV.12 days = 365 }
			}
			random_province = {
				limit = {
					is_capital = no
					OR = {
						region = japan_region
						region = indonesia_region
						region = deccan_region
					}
				}
				province_event = { id = hee_NAV.12 days = 365 random = 90}
			}
		}
		custom_tooltip = hee_NAV.10.tt
	}
}

#Ciudadela de Pamplona
country_event = {
	id = hee_NAV.11
	title = hee_NAV.11.t
	desc = hee_NAV.11.d
	picture = FORT_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = NAV
			AND = {
				tag = SPA
				NOT = {
					exists = NAV
				}
			}
			AND = {
				tag = ARA
				NOT = {
					exists = NAV
				}
			}
		}
		owns = 210
		is_year = 1570
		NOT = {
			is_year = 1630
		}
		exists = FRA
		NOT = {
			alliance_with = FRA
		}
		NOT = {
			has_opinion = {
				who = FRA
				value = 50
			}
		}
	}

	mean_time_to_happen = {
		months = 420
		modifier = {
			factor = 0.7
			truce_with = FRA
		}
		modifier = {
			factor = 0.8
			is_rival = FRA
		}
		modifier = {
			factor = 0.8
			FRA = {
				is_rival = ROOT
			}
		}
	}

	option = { #Build a great fort!
		name = hee_NAV.11.a
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				num_of_loans = 5
			}
		}
		add_years_of_income = -0.25
		210 = {
			add_province_modifier = {
				name = hee_ciudadela_pamplona
				duration = -1
			}
		}
		if = {
			limit = {
				NOT = {
					210 = {
						fort_level = 2
					}
				}
			}
			210 = {
				add_building = fort_15th
			}
		}
	}

	option = { #Let's develop a different strategy
		name = hee_NAV.11.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 100
				num_of_loans = 5
			}
		}
		add_mil_power = 20
	}
}
