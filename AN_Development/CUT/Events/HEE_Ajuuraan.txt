#Integrating Mogadishu currency

country_event = {
	id = hee_AJU.1
	title = hee_AJU.1.t
	desc = hee_AJU.1.d
	picture = CORRUPTION_eventPicture

	fire_only_once = yes

	trigger = {
		tag = AJU
		OR = {
			AND = {
				NOT = {
					exists = MDI
				}
				owns_core_province = 1205
			}
			MDI = {
				is_subject_of = AJU
			}
		}
	}

	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.5
			has_idea_group = economic_ideas
		}
	}

	option = {
		name = hee_AJU.1.a
		add_country_modifier = {
			name = hee_mogadishu_currency
			duration = -1
		}
		1205 = {
			add_province_modifier = {
				name = hee_mogadishu_administration
				duration = -1
			}
		}
	}

	option = {
		name = hee_AJU.1.b
		add_stability = 1
	}
}

	id = hee_AJU.2
	title = hee_AJU.2.t
	desc = hee_AJU.2.d
	picture = HARBOUR_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = AJU
			tag = MDI
		}
		owns_core_province = 1205
		is_at_war = no
		is_year = 1500
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = hee_AJU.2.a
		ai_chance = {
			factor = 95
		}
		1205 = {
			add_base_production = 1
			add_province_modifier = {
				name = hee_toob_menadir
				duration = -1
			}
		}
		MAM = {
			country_event = { id = hee_AJU.3 }
		}
		SYR = {
			country_event = { id = hee_AJU.3 }
		}
		if = {
			limit = {
				TUR = {
					OR = {
						owns_core_province = 382
						owns_core_province = 361
					}
				}
			}
			country_event = { id = hee_AJU.3 }
		}
	}
	option = {
		name = hee_AJU.2.b
		ai_chance = {
			factor = 5
		}
		add_prestige = 15
	}
}
