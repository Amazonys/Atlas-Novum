#Pressure from the Shilluk

		tag = FNJ

	mean_time_to_happen = {
		months = 24

		modifier = {
			factor = 0.5
			is_year = 1505
		}
		modifier = {

			is_year = 1510
		}
	}

		name = "BtPLegacyUpperNileEvents.EVTOPTA1"

		2800 = {
			cede_province = FNJ
			add_core = FNJ
			change_culture = luo_culture
			change_religion = shamanism
		}
		set_capital = 2800
		5290 = {
			cede_province = SLK
			add_core = SLK
			remove_core = FNJ
			rename_capital = "Fashoda"
		}
		hidden_effect = {
			SLK = {
				define_ruler = {
					name = "Nyikaango I"
					dynasty = "Shilluk"
					age = 25
					claim = 95
					regency = no
					female = no
					adm = 5
					dip = 5
					mil = 5
				}
			}
		}
		add_prestige = -25
		add_stability = -1
		add_mil_power = 100
		add_army_tradition = 25
		add_country_modifier = {
			name = "btpl_funj_momentum"
			duration = 7300
		}
	}

	option = { # Resist the Shilluk
		name = "BtPLegacyUpperNileEvents.EVTOPTB1"
		ai_chance = { factor = 5 }
		add_prestige = 10
		5290 = { add_core = SLK }

			spawn_rebels = {
				type = nationalist_rebels
				leader = "Nyikaango"
				size = 2
				friend = SLK
			}
		}
	}
}

#Funj Expand Northwards
country_event = {
	id = BtPLegacyUpperNileEvents.3
	title = "BtPLegacyUpperNileEvents.EVTNAME3"
	desc = "BtPLegacyUpperNileEvents.EVTDESC3"
	picture = NATIVES_eventPicture

	fire_only_once = yes

	trigger = {
		tag = FNJ
		has_country_modifier = btpl_funj_momentum
		OR = {
			2799 = { owned_by = ALO }
			2799 = { owned_by = ABD }
		}
	}

	mean_time_to_happen = {
		months = 2

		modifier = {
			factor = 0.5
			is_year = 1502
		}
		modifier = {
			factor = 0.5
			is_year = 1504
		}
	}

	option = { # Make use of this momentum
		name = "BtPLegacyUpperNileEvents.EVTOPTA3"
		ai_chance = { factor = 95 }
		capital_scope = { build_to_forcelimit = { infantry = 1.0 } }
		add_country_modifier = {
			name = "btpl_funj_militarisation"
			duration = 7300
		}
		every_province = {
			limit = {
				OR = {
					owned_by = ALO
					owned_by = ABD
				}
			}
			add_claim = FNJ
		}
		if = {
			limit = { exists = ALO }
			declare_war_with_cb = {
				who = ALO
				casus_belli = cb_conquest
			}
		}
		if = {
			limit = { exists = ABD }
			declare_war_with_cb = {
				who = ABD
				casus_belli = cb_conquest
			}
		}
	}

	option = { # We dont need to expand to be great
		name = "BtPLegacyUpperNileEvents.EVTOPTB3"
		ai_chance = { factor = 5 }
		remove_country_modifier = btpl_funj_momentum
		add_country_modifier = {
			name = "btpl_funj_demilitarisation"
			duration = 7300
		}
	}
}
