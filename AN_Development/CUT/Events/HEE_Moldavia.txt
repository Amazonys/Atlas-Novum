#Athleta Christi

  id = hee_MOL.3
  title = hee_MOL.3.t
  desc = hee_MOL.3.d
    picture = RELIGIOUS_WARS_eventPicture

	fire_only_once = yes

	trigger = {
            tag = MOL
            NOT = {
				vassal_of = TUR
			}
            is_year = 1475

			}
            owns = 1756
            owns = 2405

		}

    mean_time_to_happen = {
		months = 200
    }

	option = {
		name = hee_MOL.3.a
		add_ruler_modifier = {
			name = athleta_christi
			duration = -1
		}
	}
}
