# The Long Compromise

country_event = {
	id = hee_HAM.1
	title = hee_HAM.1.t
	desc = hee_HAM.1.d
	picture = DEBATE_REPUBLICAN_eventPicture

	fire_only_once = yes

	trigger = {
		tag = HAM
		owns_core_province = 44
		44 = {
			religion = protestant
		}
	}

	mean_time_to_happen = {
		months = 4
	}

	option = {
		name = hee_HAM.1.a
		44 = {
			add_province_modifier = {
				name = hee_long_compromise
				duration = 3650
			}
		}
		add_stability = 1
	}
}
