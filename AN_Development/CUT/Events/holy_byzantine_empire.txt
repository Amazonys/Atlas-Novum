# Renovatio triggers for Emperor...
country_event = {
	id = holy_byzantine_empire.1
	title = "EVTNAME9075"
	desc = "EVTDESC9075"
	picture = HRE_eventPicture

	major = yes
	is_triggered_only = yes

	option = {
		name = "EVTOPTA9075"
		change_tag = ROM
		add_country_modifier = {
			name = "rome_reborn"
			duration = -1
		}
		every_province = {
			limit = {
				is_part_of_hre = yes
				OR = {
					owned_by = ROOT
					owner = { is_neighbor_of = ROOT }
				}
				NOT = { is_core = ROOT }
			}
			add_core = emperor
		}
		every_country = {
			limit = {
				is_part_of_hre = yes
				is_emperor = no
				num_of_cities = 1
			}
			country_event = { id = holy_byzantine_empire.2 }
		}
		dismantle_hre = yes
		set_government_rank = 3
		add_adm_power = 250
	}
}
