country_event = {
	id = IR_flavor_HAB.1
	title = "IR_flavor_HAB.1.t"
	desc = "IR_flavor_HAB.1.d"
	picture = COURT_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = HAB
			tag = STY
			tag = HUN
		}
		is_year = 1455
		NOT = { is_year = 1460 }
		OR = {
			HAB = { has_ruler = "Ladislaus Postumus" }
			HUN = { has_ruler = "Ladislaus Postumus" }
		}
	}

	mean_time_to_happen = {
		months = 30
	}


	immediate = {
		hidden_effect = {
			if = { limit = { OR = { AND = { tag = HAB ai = yes } tag = STY } }
				HAB = { kill_ruler = yes }
				STY = {
					inherit = HAB
				}
				set_capital = 134
			}
			if = { limit = { tag = HAB ai = no }
				HAB = {
					inherit = STY
						define_ruler = {
							name = "Friedrich III"
							dynasty = "von Habsburg"
							#birth_date = 1415.9.21
							adm = 2
							dip = 5
							mil = 1
							culture = ROOT
							religion = ROOT
						}
						define_consort = {
							country_of_origin = POR
							name = "Eleanor"
							dynasty = "de Avis"
							birth_date = 1434.9.18
							#death_date = 1467.9.3
							female = yes
							adm = 2
							dip = 2
							mil = 1
						}
					}
			}
			HUN = {
				if = { limit = { has_ruler = "Ladislaus Postumus" } kill_ruler = yes
				define_ruler = {
					name = "Mátyás"
					dynasty = Corvin
					#birth_date = 1443.2.23
					culture = ROOT
					religion = ROOT
					adm = 4
					dip = 5
					mil = 5
					}
				}
				else_if = {
					limit = { has_heir = "Ladislaus Postumus" }
				kill_heir = yes
				#fire event to elect a ruler?
				}
			}
			BOH = {
				if = { limit = { has_ruler = "Ladislaus Postumus" } kill_ruler = yes
				define_ruler = {
					name = "Jirí"
					dynasty = "z Podebrad"
					#birth_date = 1420.4.23
					adm = 5
					dip = 4
					mil = 3
					culture = ROOT
					religion = ROOT
					}
				}
				else_if = {
					limit = { has_heir = "Ladislaus Postumus" }
				kill_heir = yes
				#fire event to elect a ruler?
				}
			}
		}
	}

	option = {
		name = "IR_flavor_HAB.1.a"
	}
}
