country_event = {
	id = loss_of_malacca.100
	title = "loss_of_malacca.100.t"
	desc = {
		trigger = { 596 = { owned_by = MLC } }
		desc = "loss_of_malacca.100.d1"
	}
	desc = {
		trigger = { 596 = { NOT = { owned_by = MLC } } }
		desc = "loss_of_malacca.100.d2"
	}
	picture = ANGRY_MOB_eventPicture


	is_triggered_only = yes
	major = yes
	#mean_time_to_happen = {
	#	days = 1
	#}


	option = {
		trigger = { 596 = { owned_by = MLC } }
		name = "loss_of_malacca.100.a"
		add_legitimacy = 50
		add_stability = 3
		add_years_of_income = 2
		discover_mission_cede_alt = {
			PROVINCE_ID = 6459
			COST = 25
			TRADE_GOOD = tropical_wood #MUCH BETTER IF YOU WIN
		}
		every_province = {
			limit = { region = malaya_region NOT = { OR = { is_core = PREV is_claim = PREV } } }
			add_claim = PREV
		}
	}
	option = {
		trigger = { 596 = { NOT = { owned_by = MLC } } }
		name = "loss_of_malacca.100.b"
		change_tag = RIA
		#set_capital = 597
		release = PRK
		release = JOH
		hidden_effect = {
			discover_mission_cede_alt = {
				PROVINCE_ID = 6459
				COST = 0
				TRADE_GOOD = fish
			}
			every_core_province = {
				limit = { NOT = { owned_by = PREV } }
				remove_core = PREV
			}
			JOH = {
				every_owned_province = {
					add_claim = RIA
				}
			}
		}
	}
}

	id = loss_of_malacca.2
	title = "loss_of_malacca.2.t"
	desc = "loss_of_malacca.2.d"
	picture = MERCHANTS_TALKING_eventPicture

	is_triggered_only = yes


	trigger = { NOT = { has_country_flag = loss_of_malacca.2_happened } }

	immediate = { set_country_flag = loss_of_malacca.2_happened }


	option = {		#It was inevitable
		name = "loss_of_malacca.2.a"
		random_owned_province = {
			limit = { NOT = { is_capital = yes } base_tax = 2 base_manpower = 2 }
			add_base_tax = -1
			add_base_manpower = -1
		}
		capital_scope = {
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
	option = {		#They should stay where they are
		name = "loss_of_malacca.2.b"
		add_legitimacy = 5
	}
}

#Pahang reclaims land
country_event = {
	id = loss_of_malacca.3
	title = "loss_of_malacca.3.t"
	desc = "loss_of_malacca.3.d"
	picture = RELIGIOUS_WAR_eventPicture

	is_triggered_only = yes

	trigger = { NOT = { has_country_flag = loss_of_malacca.3_happened }  }

	immediate = { set_country_flag = loss_of_malacca.3_happened }

	option = {		#We cannot do anything!
		trigger = { PAH = { exists = yes } }
		ai_chance = { factor = 50 }
		name = "loss_of_malacca.3.a"
		every_owned_province = {
			limit = { is_core = PAH }
			cede_province = PAH
		}
	}
	option = {		#We cannot do anything!
		trigger = { PAH = { exists = no } any_owned_province = { is_core = PAH } }
		ai_chance = { factor = 50 }
		name = "loss_of_malacca.3.a"
		release = PAH
	}
	option = {		#Pahan is ours!
		ai_chance = { factor = 50 }
		name = "loss_of_malacca.3.b"
		random_owned_province = {
			limit = { is_core = PAH }
			spawn_rebels = {
				type = nationalist_rebels
				religion = sunni
				size = 2
			}
		}
	}
}
