			}

					NOT = { num_of_cities = 5 }
				}
			}
		}
		#set_country_flag = cri_seek_tur_protection

			limit = {
				has_regency = yes
				has_heir = no
			}

				name = "Mengli"
				dynasty = "Giray"
				age = 33
				claim = 75
			}

				limit = {
					has_states_general_mechanic = yes
				}
				change_statists_vs_orangists = 1
			}
			if = {
				limit = {
					has_dlc = "Rights of Man"
				}
				hidden_effect = { add_ruler_personality = conqueror_personality }
			}
		}
		if = {
			limit = {
				has_regency = yes
				has_heir = yes
			}

				name = "Mengli"

				limit = {

			limit = { is_capital = no }
			spawn_rebels = {
				size = 1
				type = pretender_rebels
			}
		}
	}

		name = "flavor_cri.1.b" #We will be better of without Ottoman involvement.

			factor = 50
			modifier = {
				factor = 2

		#trigger = {

		#}
		ai_chance = {
			factor = 100
		}
		tooltip = { create_march = CRI }
		CRI = {
			country_event = { id = flavor_cri.4 }
		}
		add_dip_power = -50 #Propping Up Weak Ruler
		add_mil_power = -50 #Propping Up Weak Ruler
		add_adm_power = -50 #Propping Up Weak Ruler
		every_province = {
			limit = { crimea_gazaria_common_provinces = yes }

				add_claim = TUR
			}
			if = {
				limit = { owned_by = CRI is_capital = no }
				cede_province = TUR
				add_core = TUR
			}
		}
	}
	option = {
		name = "flavor_cri.2.b" #Let us not risk upsetting the balance of power in the area.
		#trigger = {
		#	CRI = { has_country_flag = cri_seek_tur_protection }
		#}
		ai_chance = {
			factor = 0
		}
		CRI = {
			country_event = { id = flavor_cri.5 }
			add_opinion = {
				who = TUR
				modifier = refused_aid_in_crimean_succession
			}
		}
		every_known_country = {
			limit = { is_neighbor_of = CRI }
			add_opinion = {
				who = TUR
				modifier = stayed_out_of_crimean_succession
			}
		}
	}
}
