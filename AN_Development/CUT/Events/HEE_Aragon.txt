#The Assassination of Pedro de Arbués
country_event = {
	id = hee_ARA.1
	title = hee_ARA.1.t
	desc = hee_ARA.1.d
	picture = ASSASSINATION_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = ARA
			tag = SPA
		}
		NOT = {
			is_year = 1495
		}
		is_year = 1480
		owns_core_province = 214
		religion = catholic
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.3
			advisor = inquisitor
		}
		modifier = {
			factor = 0.75
			advisor = theologian
		}
	}

	option = {
		name = hee_ARA.1.a
		ai_chance = {
			factor = 90
		}
		if = {
			limit = {
				advisor = inquisitor
			}
			kill_advisor = inquisitor
		}
		if = {
			limit = {
				NOT = { advisor = inquisitor }
			}
			kill_advisor = random
		}
		add_adm_power = 25
		add_papal_influence = 10
		214 = {
			add_province_modifier = {
				name = hee_arbues_saint_province
				duration = 7300
			}
		}
		2990 = {
			add_province_modifier = {
				name = hee_arbues_saint_province
				duration = 7300
			}
		}
		211 = {
			add_province_modifier = {
				name = hee_arbues_saint_province
				duration = 7300
			}
		}
		add_country_modifier = {
			name = hee_arbues_saint
			duration = 10950
		}
	}
	option = {
		name = hee_ARA.1.b

		}
