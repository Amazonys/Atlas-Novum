# The Rough Wooing

country_event = {
	id = hee_ENG.1
	title = hee_ENG.1.t
	desc = hee_ENG.1.d
	picture = BATTLE_eventPicture

	fire_only_once = yes

	trigger = {
		tag = ENG
		is_subject = no
		exists = SCO
		SCO = {
			is_subject = no
			OR = {
				is_female = yes
				has_heir = no
				AND = {
					has_regency = yes
					has_female_heir = yes
				}
			}
		}
		is_year = 1500
		NOT = {
			is_year = 1600
		}
		NOT = {
			exists = GBR
		}
		is_at_war = no
	}

	mean_time_to_happen = {
		months = 60
		modifier = {
			factor = 0.5
			truce_with = SCO
		}
	}

	option = {
		name = hee_ENG.1.a
		add_casus_belli = {
			type = cb_fabricated_claims
			target = SCO
			months = 120
		}
	}
}
