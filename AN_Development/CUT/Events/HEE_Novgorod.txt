#Alliance oiffer to Lithuania
country_event = {
	id = hee_NOV.5
	title = hee_NOV.5.t
	desc = hee_NOV.5.d
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	option = {
		name = hee_NOV.5.a
		ai_chance = {
			factor = 100
		}
		create_alliance = FROM
		add_opinion = {
			who = FROM
			modifier = hee_NOV_gift
		}
		FROM = {
			add_opinion = {
				who = LIT
				modifier = hee_NOV_gift
			}
		}
	}

	option = {
		name = hee_NOV.5.b
		ai_chance = {
			factor = 0
		}
		NOV = {
			add_casus_belli = {
				target = LIT
				type = cb_insult
				months = 300
			}
		}
	}
}
