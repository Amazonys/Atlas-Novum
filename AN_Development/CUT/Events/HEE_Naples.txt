# Pedro Alvarez de Toledo

country_event = {
	id = hee_NAP.1
	title = hee_NAP.1.t
	desc = hee_NAP.1.d
	picture = ADVISOR_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = NAP
			tag = TTS
		}
		is_year = 1520
		NOT = {
			is_year = 1560
		}
	}

	mean_time_to_happen = {
		months = 300
		modifier = {
			factor = 0.8
			adm = 4
		}
		modifier = {
			factor = 0.8
			is_subject = yes
		}
	}

	#We need to centralize our state.
	option = {
		name = hee_NAP.1.a
		add_adm_power = 25
		add_dip_power = 25
		add_country_modifier = {
			name = hee_pedro_alvarez
			duration = 7300
		}
	}

	#Naples must become a military power to rival any in Europe.
	option = {
		name = hee_NAP.1.b
		add_mil_power = 50
	}
}
