#Marathi events
namespace = hee_MAR

#Maratha Uprising
country_event = {
	id = hee_MAR.1
	title = hee_MAR.1.t
	desc = hee_MAR.1.d
	picture = BATTLE_eventPicture

	major = yes
	fire_only_once = yes

	trigger = {
		is_year = 1620
		NOT = {
			is_year = 1690
		}
		NOT = {
			exists = MAR
		}
		any_owned_province = {
			culture = marathi
			religion = hinduism
		}
		OR = {
			AND = {
				ai = no
				num_of_cities = 10
			}
			AND = {
				ai = yes
				num_of_cities = 6
			}
		}
		NOT = {
			any_country = {
				has_country_modifier = hee_marathi_enemy
			}
		}
		NOT = {
			primary_culture = marathi
		}
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 5
			religion = hinduism
		}
		modifier = {
			factor = 1.5
			accepted_culture = marathi
		}
		modifier = {
			factor = 0.8
			religion = sunni
		}
		modifier = {
			factor = 1.5
			ai = no
			is_at_war = yes
		}
		modifier = {
			factor = 0.8
			NOT = {
				num_of_cities = 11
			}
		}
	}

	immediate = {
		hidden_effect = {
				add_country_modifier = { #Keep track of previous owner
					name = hee_marathi_enemy
					duration = -1
					hidden = yes
				}
			random_owned_province = {
				limit = {
					culture = marathi
					religion = hinduism
				}
				add_core = MAR
			}
			release = MAR
			MAR = {
				define_ruler = {
					name = "Shivaji"
					dynasty = "Bhonsle"
					ADM = 4
					DIP = 4
					MIL = 5
				}
			}
			every_province = {
				limit = {
					culture = marathi
				}
				add_claim = MAR
			}
		}
	}

	option = {
		name = hee_MAR.1.a
		hidden_effect = {
			MAR = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.7
						cavalry = 0.3
					}
				}
			}
		}
		MAR = {
			declare_war_with_cb = {
				who = ROOT
				casus_belli = cb_independence_war
			}
		}
	}
}


#Fort Defects
province_event = {
	id = hee_MAR.2
	title = hee_MAR.2.t
	desc = hee_MAR.2.d
	picture = CONQUEST_eventPicture

	fire_only_once = yes

	trigger = {
		owner = {
			OR = {
				AND = {
					has_country_modifier = hee_marathi_enemy
					num_of_cities = 5
					war_with = MAR
				}
				AND = {
					any_subject_country = {
						has_country_modifier = hee_marathi_enemy
					}
					num_of_cities = 6
					war_with = MAR
				}
			}
		}
		culture = marathi
		any_neighbor_province = {
			owned_by = MAR
		}
		is_capital = no
		NOT = {
			owner = {
				has_country_modifier = hee_no_more_defections
			}
		}
	}

	mean_time_to_happen = {
		months = 18
		modifier = {
			factor = 0.7
			religion = hinduism
		}
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					owner = {
						has_country_modifier = hee_defected_once
					}
				}
				add_country_modifier = {
					name = hee_no_more_defections
					duration = -1
					hidden = yes
				}
			}
		}
	}

	option = {
		name = hee_MAR.2.a
		add_core = MAR
		cede_province = MAR
		hidden_effect = {
			MAR = {
				country_event = { id = hee_MAR.3 }
			}
		}
		MAR = {
			infantry = ROOT
			infantry = ROOT
			cavalry = ROOT
		}
		hidden_effect = {
			add_country_modifier = {
				name = hee_defected_once
				duration = -1
				hidden = yes
			}
		}
	}
}
