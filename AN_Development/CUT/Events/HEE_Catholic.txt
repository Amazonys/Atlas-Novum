# Growth of Popular Piety

country_event = {
	id = hee_catholic.2
	title = hee_catholic.2.t
	desc = hee_catholic.2.d
	picture = RELIGION_eventPicture

	trigger = {
		NOT = {
			is_religion_enabled = protestant
		}
		religion = catholic
		reform_desire = 0.02
		NOT = {
			has_country_flag = hee_popular_piety
		}
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 1.2
			advisor = inquisitor
		}
		modifier = {
			factor = 0.9
			advisor = theologian
		}
	}

	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = hee_popular_piety_timer
				duration = 18250
				hidden = yes
			}
		}
	}

	# These people should listen to the pope
	option = {
		name = hee_catholic.2.a
		trigger = {
			ai = no
		}
		add_country_modifier = {
			name = hee_piety_resisted
			duration = 3650
		}
		add_papal_influence = 15
		add_reform_desire = 0.01
	}

	# People should always strive for new ways to express their faith
	option = {
		name = hee_catholic.2.b
		add_country_modifier = {
			name = hee_piety_supported
			duration = 3650
		}
		add_papal_influence = -15
	}
}
