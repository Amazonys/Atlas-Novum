			}
		}

		}
		allow = {
			tier_2_formable_trigger = {
				PROVINCES=spa_core_provinces
			}
			NOT = { current_age = age_of_discovery }
			is_at_war = no
		}
		effect = {
			swap_non_generic_missions = yes
			#Integrate Cokings and unions
			every_subject_country = { 
				limit = { OR = { is_subject_of_type = personal_union is_subject_of_type = coking } }
				ROOT = { inherit = PREV } 
			}
			# integrate fiefs
			custom_tooltip = integrate_fiefs.tooltip
			hidden_effect = {
				every_subject_country = {
					limit = { is_subject_of_type = fief NOT = { num_of_cities = 9 } }

						add_core = ROOT
						cede_province = ROOT
					}
				}
			}
			change_tag = SPA

				limit = { NOT = { government_rank = 5 } }
