can_make_core = {
	condition = {
		tooltip = "BELONGS_TO_ELG"
		potential = {
			owner = {
				has_country_flag = nhs_formed_greek
				NOT = { has_country_flag = nhs_no_greek_nation }
				nhs_is_elysia_late = yes
				OR = {
					NOT = { exists = ELG }
					ELG = {
						is_subject_of = PREV
					}
				}
			}
			NOT = { has_province_modifier = nhs2_base_of_operations }
			NOT = { province_id = 151 }
			OR = {
				NOT = { has_province_flag = nhs_invasion_keep_province }
				owner = {
					has_country_flag = nhs_invasion_over
				}
			}
			nhs_greek_areas = yes
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "BEING_SETTLED"
		potential = {
			owner = {
				OR = {
					tag = NHS
					nhs_is_elysia_late = yes
				}
			}
			OR = {
				province_id = 2550
				province_id = 952
				province_id = 950
				province_id = 2554
				province_id = 957
			}
			NOT = { has_province_flag = nhs_got_core }
			owned_by = ROOT
		}
		allow = {
			always = no
		}
	}
	condition = {
		tooltip = "NHS_MILITARY_ORDER"
		potential = {
			owner = {
				tag = VGD
				is_subject = yes
			}
			NOT = { is_capital = yes }
		}
		allow = {

	}
