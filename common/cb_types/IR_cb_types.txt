		}
		if = {
			limit = { OR = { tag = ENG tag = FRA } NOT = { has_global_flag = HYW_end }}
			FROM = {
				is_subject = no

					NOT = { num_of_cities = ROOT }

					}
				}
			}
		}
		else_if = {
			limit = { is_part_of_hre = yes }

			}
		}
		else = {
			FROM = {

				any_subject_country = {
