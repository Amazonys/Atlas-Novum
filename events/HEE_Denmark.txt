		months = 180

			factor = 0.1

		}

		name = hee_DAN.1.a
		add_stability = 1

				id = hee_DAN.2 days = 2
			}
		}

	}

	option = {
		name = hee_DAN.1.b

		add_legitimacy = -10

}

#Hans Tausen
country_event = {
	id = hee_DAN.3
	title = hee_DAN.3.t
	desc = hee_DAN.3.d
	picture = PRAYING_eventPicture	

	fire_only_once = yes

	trigger = {
		OR = {
			tag = DAN
			AND = {
				tag = SCA
				primary_culture = danish
			}
		}
		is_religion_enabled = protestant
		religion = catholic
		NOT = {
			is_year = 1530
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}

	immediate = {
			random_owned_province = {
				limit = {
					is_capital = no
				}
				add_province_modifier = {
					name = hee_tausen
					duration = -1
				}
			}
	}

	option = {
		name = hee_DAN.3.a
		random_owned_province = {
			limit = {
				has_province_modifier = hee_tausen
			}
			change_religion = protestant
			add_province_modifier = {
				name = religious_zeal_at_conv
				duration = 3650
			}
		}
	}

	option = {
		name = hee_DAN.3.b
		random_owned_province = {
			limit = {
				has_province_modifier = hee_tausen
			}
			add_province_modifier = {

			}

	}
}

	desc = hee_DAN.12.d
	picture = KING_SICK_IN_BED_eventPicture

	fire_only_once = yes

	trigger = {
		OR = {
			tag = DAN
			AND = {
				tag = SCA
				primary_culture = danish
			}
		}
		government = monarchy
		is_year = 1750
		NOT = {
			is_year = 1790
		}
		OR = {
			has_any_disaster = no
			has_disaster = aspiration_for_liberty
		}
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.8
			NOT = {
				ADM = 3
			}
		}
		modifier = {
			factor = 0.8
			NOT = {
				DIP = 3
			}
		}
		modifier = {
			factor = 0.8
			NOT = {
				MIL = 3
			}
		}
		modifier = {
			factor = 0.6
			has_disaster = aspiration_for_liberty
		}
	}

	option = {
		name = hee_DAN.12.a
		ai_chance = {
			factor = 90
		}
		if = {
			limit = {
				has_heir = no
			}
			define_heir = {
				dynasty = ROOT
				age = 1
			}
		}
		define_ruler = {
			name = "(Regent) Johann Friedrich Struensee"
			dynasty = "Struensee"
			ADM = 5
			DIP = 3
			MIL = 2
			regency = yes
			age = 26
		}
		set_ruler_flag = hee_struensee_timer
		add_country_modifier = {
			name = hee_denmark_struensee
			duration = -1
		}
		add_ruler_modifier = {	#Compatability with Lex Talionis	
			name = is_usurper
			duration = -1
		}
		custom_tooltip = hee_DAN.12.tt
	}

	option = {
		name = hee_DAN.12.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 100
				OR = {
					ADM = 5
					DIP = 5
					MIL = 5
				}
			}
		}
		add_stability = -1
		add_dip_power = -20
		add_mil_power = -20
	}
}
