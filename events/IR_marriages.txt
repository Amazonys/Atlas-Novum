namespace = IR_marriages


# The Iberian Wedding
country_event = {
	id = IR_marriages.1
	title = "IR_marriages.1.t"
	desc = "IR_marriages.1.d"
	picture = DIPLOMACY_eventPicture

	major = yes
	
	fire_only_once = yes
	
	
	trigger = {
		NOT = { exists = SPA }
		OR = {
			AND = {
				tag = CAS
				NOT = { war_with = ARA }
				ARA = {
					exists = yes
					ai = yes
					was_player = no
					is_subject = no
					is_neighbor_of = ROOT
				}
			}
			AND = {
				tag = ARA
				NOT = { war_with = CAS }
				ai = no
				CAS = {
					exists = yes
					ai = yes
					was_player = no
					is_subject = no
					is_neighbor_of = ROOT
				}
			}
		}
		is_year = 1469
		is_subject = no
		num_of_cities = 10
	}

	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.1
			OR = {
				AND = {
					tag = CAS
					ARA = {
						has_regency = yes 
					}
				}
				AND = {
					tag = ARA
					CAS = { 
						has_regency = yes 
					}
				}
			}
		}
	}
	
	
	option = {		# Let us bind their Dynasty to ours
		name = "flavor_spa.EVTOPTA3716"
		ai_chance = { factor = 100 }
		if = {
			limit = { tag = CAS }
			CAS = { release = ARA }
			ARA = { release = CAS }
			create_union = ARA
			add_opinion = {
				who = ARA
				modifier = spa_iberian_wedding
			}
			ARA = {
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding
				}
				add_liberty_desire = -50
			}
			every_known_country = {
				limit = {
					marriage_with = ARA
				}
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding_negative
				}
			}
			define_ruler = {
 				name = "Isabella"
 				dynasty = "de Trastámara" 
				age = 18
				claim = 75
				regency = no 
				female = yes
				adm = 4
				dip = 5
				mil = 2
			}
			define_consort = {
 				name = "Ferdinand"
 				dynasty = "de Trastámara"
				age = 17
  				adm = 3
 				dip = 4
 				mil = 3
 				female = no
			}
		}
		if = {
			limit = { tag = ARA }
			CAS = { release = ARA }
			ARA = { release = CAS }
			create_union = CAS
			add_opinion = {
				who = CAS
				modifier = spa_iberian_wedding
			}
			CAS = {
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding
				}
				add_liberty_desire = -50
			}
			every_known_country = {
				limit = {
					marriage_with = CAS
				}
				add_opinion = {
					who = ROOT
					modifier = spa_iberian_wedding_negative
				}
			}
			define_ruler = {
 				name = "Ferdinand"
 				dynasty = "de Trastámara"
				age = 17
				claim = 75
				adm = 3
 				dip = 4
 				mil = 3
 				female = no
			}
			define_consort = {
 				name = "Isabella"
 				dynasty = "de Trastámara" 
				age = 18
				female = yes
				adm = 4
				dip = 5
				mil = 2
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		#hidden_effect = { country_event = { id = castilian_succession.1 months = 60 random = 24 } }
	}
	option = {		# No let us marry a local talent instead
		name = "flavor_spa.EVTOPTB3716"
		ai_chance = { factor = 1 }
		add_stability_or_adm_power = yes
		if = {
			limit = { tag = CAS }
			ARA = {
				add_opinion = {
					who = ROOT
					modifier = spa_declined_iberian_wedding
				}
			}
		}
		if = {
			limit = { tag = ARA }
			CAS = {
				add_opinion = {
					who = ROOT
					modifier = spa_declined_iberian_wedding
				}
			}
		}
		if = {
			limit = {
				has_consort = no
			}
			define_consort = {
				ADM = 3
				MIL = 3
				DIP = 3
			}
		}
	}
}

#Election of Kasimir
country_event = {
	id = IR_marriages.2
	title = IR_marriages.2.t
	desc = IR_marriages.2.d

	picture = LIBERUM_VETO_eventPicture
	
	fire_only_once = yes
	major = yes
	
	trigger = {
		tag = POL
		is_subject = no
		has_regency = yes
		LIT = {
			ai = yes
			exists = yes
			was_player = no
			is_subject = no
			NOT = { junior_union_with = POL }
			dynasty = "Jagiellon"
		}
		is_year = 1445
		NOT = { is_year = 1450 }
	}
	
	mean_time_to_happen = {
		months = 36
		modifier = {
			factor = 0.5
			is_year = 1446
		}
		modifier = {
			factor = 0.01
			is_year = 1447
		}
		modifier = {
			factor = 0.001
			is_year = 1448
		}
	}
	
	option = {
		name = IR_marriages.2.a #We must retain the Jagiellon Dynasty
		ai_chance = { 
			factor = 99 
		}
		create_union = LIT
		country_event = {
			id = flavor_pol.20
		}
	}
	
	option = {
		name = IR_marriages.2.b #Appoint a Local Noble.
		ai_chance = { 
			factor = 1
		}
		define_ruler = {
			DIP = 4
			ADM = 4
			MIL = 4
			claim = 60
		}
		LIT = {
			add_opinion = {
				who = ROOT
				modifier = lit_denied_birthright
			}
		}
	}
}
