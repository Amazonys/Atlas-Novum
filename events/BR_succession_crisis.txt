#Succession Crisis in Byzantine Empire, intended to be altered to become a disaster instead.
country_event = {
    id = succession_crisis.34
    title = succession_crisis.34.t
    desc = succession_crisis.34.d
    picture = KING_SICK_IN_BED_eventPicture
	
    is_triggered_only = yes
        
	mean_time_to_happen = {
		days = 1
	}
    major = yes
    
    immediate = { #Adjusted to give a bit more rebels so it isn't so easy to get out of, meaning events are more likely to fire in the meantime.
        hidden_effect = {
            set_country_flag = had_byz_succ_crisis
            add_stability = -3
            add_country_modifier = {
                name = byz_succession_crisis
                duration = 9125
            }
            random_owned_province = {
                limit = {
                    is_capital = no
                }
                spawn_rebels = {
                    type = noble_rebels
                    size = 1
                }
            }
			random_owned_province = {
                limit = {
                    is_capital = no
                }
                spawn_rebels = {
                    type = pretender_rebels
                    size = 2
                }
            }
            if = {
                limit = {
                    TUR = {
                        is_great_power = yes
                    }
                    NOT = { is_great_power = yes }
                }
                random_owned_province = {
                    limit = {
                        is_capital = no
                    }
                    spawn_rebels = {
                        type = noble_rebels
                        size = 1
                        friend = TUR
                    }
                }
				random_owned_province = {
                    limit = {
                        is_capital = no
                    }
                    spawn_rebels = {
                        type = pretender_rebels
                        size = 2
                        friend = TUR
                    }
                } 
            }
            
        }
    }
    
    option = {
        name = succession_crisis.34.a
        trigger = {
            has_regency = no
            has_heir = yes
        }
        kill_ruler = yes
        add_legitimacy = -100
		set_country_flag = byz_heir_defend_succession
    }
    
    option = {
        name = succession_crisis.34.b
        trigger = {
            has_regency = yes
        }
        add_legitimacy = -100
    }
    
    option = {
        name = succession_crisis.34.c
        trigger = {
            has_regency = no
            has_heir = no
        }
        define_ruler = {
            name = "(Interregnum)"
            adm = 0
            dip = 0
            mil = 0
            fixed = yes
            regency = yes
        }
        add_legitimacy = -100
    }
}

#Gothic Pretenders Arise
country_event = {

    desc = succession_crisis.35.d
    picture = CONQUEST_eventPicture
    
    trigger = { #Commented some trigger requirements due to them either being present already as part of disaster trigger or me not thinking they make sense.

        #NOT = { stability = 2 }
        #government = monarchy
		OR = { #Overhauled to allow for the event even though Theodoro does not exist, much like Montferrat works.
			owns = 2410 #Theodoro/Doros/Cherson/Whatever
			FEO = {
				exists = yes
				has_regency = no
			}
		}
        #NOT = { num_of_rebel_armies = 1 }
    }
    
    is_triggered_only = yes
        
	mean_time_to_happen = {
		days = 1
	}
    
    fire_only_once = yes
    
    option = {
        name = succession_crisis.35.a
        add_legitimacy = 10
        if = {
            limit = {
                FEO = {
                    exists = yes
                    dynasty = "of Gothia"
                }
            }
            random_owned_province = {
                limit = {
                    is_capital = no
                }
                spawn_rebels = {
                    type = pretender_rebels
                    size = 2
                    leader = "Alexander of Gothia"
                    friend = FEO
                }
            }
        }
        if = {
            limit = {
                FEO = {
                    exists = yes
                }
            }
            random_owned_province = {
                limit = {
                    is_capital = no
                }
                spawn_rebels = {
                    type = pretender_rebels
                    size = 1
                    friend = FEO
                }
            }
        }
    }
    
    option = {
		trigger = {
			NOT = {
				has_country_flag = accepted_pretender
			}
		}
        name = succession_crisis.35.b
		if = {
			limit = {
				FEO = {
					exists = yes
                    dynasty = "of Gothia"
				}
			}
			define_ruler = {
				name = "(Interregnum)"
				adm = 0
				dip = 0
				mil = 0
				fixed = yes
				regency = yes
			}
			kill_heir = yes
			set_country_flag = accepted_goth
			country_event = { id = succession_crisis.45 }
		}
		else = {
			kill_heir = yes
			define_ruler = {
				adm = 1
				dip = 1
				mil = 1
				claim = 50
				dynasty = "of Gothia"
			}			
		}
        set_country_flag = accepted_pretender
        
    }
}
