country_event = {
	id = colony_cleanup.1
	title = "colony_cleanup.t"
	desc = "colony_cleanup.d"
	picture = DIPLOMACY_eventPicture
	
	#is_triggered_only = yes
	hidden = yes

	trigger = {
		any_subject_country = { 
			is_colonial_nation = yes
			capital_scope = { colonial_region = colonial_alaska }
			any_owned_province = {
				is_city = yes
				NOT = { OR = { has_province_modifier = soft_wasteland_1 has_province_modifier = soft_wasteland_2 has_province_modifier = soft_wasteland_3 } }
				NOT = { colonial_region = colonial_alaska }	
				NOT = { has_province_flag = allowed_alaska }
			}
		}
	}
	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = "colony_cleanup.a" #fix the provinces
		ai_chance = { factor = 100 }
		every_subject_country = {
			limit = { capital_scope = { colonial_region = colonial_alaska } }
			every_owned_province = {
				limit = { NOT = { colonial_region = colonial_alaska } NOT = { has_province_flag = allowed_alaska } }
				add_territorial_core = ROOT
				cede_province = ROOT
			}
		}
	}
	option = {
		name = "colony_cleanup.b" #allow them to keep provinces
		ai_chance = { factor = 0 }
		every_subject_country = {
			limit = { capital_scope = { colonial_region = colonial_alaska } }
			every_owned_province = {
				limit = { NOT = { colonial_region = colonial_alaska } NOT = { has_province_flag = allowed_alaska } }
				set_province_flag = allowed_alaska
			}
		}
	}
}
