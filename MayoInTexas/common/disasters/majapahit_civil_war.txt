#Second Majapahit Civil War
#Second civil war which led to the downfall of Majapahit

majapahit_civil_war = {
	potential = {
		tag = MAJ
		NOT = { has_country_flag = had_maj_civil_war }
		is_free_or_tributary_trigger = yes
		normal_or_historical_nations = yes
		uses_doom = no
	}


	can_start = {
		current_age = age_of_discovery
		has_any_disaster = no
		OR = {
			has_regency = yes
			NOT = { legitimacy = 60 }
                        has_heir = no
		}
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_lesser_in_union = yes
		}
	}
	
	
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { legitimacy = 60 }
		}
		modifier = {
			factor = 1
			has_heir = no
		}
		modifier = {
			factor = 1
			NOT = { stability = 0 }
		}
		modifier = {
			factor = 0.5
			NOT = { ADM = 2 }
		}		
		modifier = {
			factor = 0.5
			NOT = { DIP = 2 }
		}		
		modifier = {
			factor = 0.5
			NOT = { MIL = 2 }
		}	
		modifier = {
			factor = -1
			stability = 3
		}
	}




	
	can_end = {
		stability = 2
                legitimacy = 90
		religious_unity = 0.8
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = {
			has_spawned_rebels = pretender_rebels
		}
	}
	



	modifier = {
		global_unrest = 3
		stability_cost_modifier = 0.50
                land_morale = -0.1
                naval_morale = -0.1
	}	
	
	
	on_start = maj_civil_war.1
	on_end = maj_civil_war.100
	
	on_monthly = {
		events = {
		}
		random_events = { 
			1000 = 0
                        100 = maj_civil_war.2
                        100 = maj_civil_war.3
                        50 = maj_civil_war.4
                        100 = maj_civil_war.5
                        50 = maj_civil_war.6
#                       50 = maj_civil_war.7			#Error related to this. Cutting it for now -Draiocht
#                        50 = maj_civil_war.8			#Error related to this. Cutting it for now -Draiocht
		}
	}
}
