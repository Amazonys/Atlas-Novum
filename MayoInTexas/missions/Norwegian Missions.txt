	slot = 1
	generic = no
	ai = yes

		tag = NOR
	}
	has_country_shield = yes

	nor_win_independence = {
		icon = mission_monarch_in_throne_room
		required_missions = { }
		trigger = {
			is_subject = no
			owns_core_province = 17
			is_at_war = no
		}
		effect = {
			#Copied from Austria's missions, adjusted to become a vassalization.
			if = {
				limit = {
					exists = SWE

						is_subject = no

					type = cb_restore_personal_union

				}
			}
			else = {
				#Claims
				give_permaclaims_for_mission = { mission=nor_king_of_sweden }
			}
			give_permaclaims_for_mission = { mission=nor_retake_iceland }
		}
	}
	nor_king_of_sweden = {

					area = northern_norway

				add_base_manpower = 4
