# INSTRUCTIONS:

# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

royal_marriage = {
	condition = {
		tooltip = MARWAR
		potential = {
		}
		allow = {
			NOT = { war_with = FROM }
		}
	}
	condition = {
		tooltip = IR_NO_FIEF_MARRY
		potential = {
			OR = {
				FROM = { is_subject_of = ROOT is_subject_of_type = fief }
				ROOT = { is_subject_of = FROM is_subject_of_type = fief }
			}
		}
		allow = {
			always = no
		}
	}
}

remove_electorate = {
	condition = {
		tooltip = IR_NO_REMOVE_ELECTORATE
		potential = {
		}
		allow = {
			has_global_flag = remove_electorate_allowed
		}
	}
}

annexationaction = {
	condition = {
		tooltip = ANNEX_TOO_LOW
		potential = {
			FROM = {
				vassal_of = ROOT
			}
		}
		allow = {
			FROM = {
				has_opinion = {
					who = ROOT
					value = 190
				}
			}
		}
	}
	condition = {
		tooltip = INTEGRATEFIEFTOOLOWDESC
		potential = {
			FROM = { 
				OR = {
					is_subject_of_type = fief 
					is_subject_of_type = dominion 
				}
			}
		}
		allow = {
			FROM = {
				check_variable = {
					which = IR_subj_cohesion
					value = 100
				}
			}
		}
	}
	effect = {
		hidden_effect = { 
			every_subject_country = {
				limit = { OR = { is_subject_of_type = fief is_subject_of_type = dominion } }
				subtract_variable = { which = IR_subj_cohesion value = 30 }
			}
		}
	}
}


integrationaction = {
	condition = {
		tooltip = INTEGRATETOOLOWDESC
		potential = {
			always = no
		}
		allow = {
			FROM = {
				has_opinion = {
					who = ROOT
					value = 190
				}
			}
		}
	}
}

vassalaction = {
	condition = {
		tooltip = VASSALINVALID
		potential = {
		}
		allow = {
			alliance_with = FROM
			FROM = {
				has_opinion = {
					who = ROOT
					value = 190
				}
			}
			FROM = { is_at_war = no }
		}
	}
}

abandon_union_action = {
	condition = {
		potential = {
		}
		allow = {
			ROOT = {
				senior_union_with = FROM
				is_at_war = no
			}
		}
	}
	condition = {
		potential = {
			always = no
		}
		allow = {
			ROOT = {
				is_subject_of_type = dominion
			}
		}
	}
}


form_coalition = {
	condition = {
		tooltip = MINAGGRESSIVEEXPANSION
		potential = {	
		}
		allow = {
			has_opinion_modifier = {
				modifier = aggressive_expansion
				who = FROM
			}		
			NOT = {
				has_opinion_modifier = {
					modifier = aggressive_expansion
					who = FROM
					value = -50
				}
			}
		}
	}
