#Norwegian Events

namespace = hee_NOR

#Copper Mines on Røros

country_event = {
	id = hee_NOR.1
	title = hee_NOR.1.t
	desc = hee_NOR.1.d
	picture = TRADEGOODS_eventPicture

	fire_only_once = yes

	trigger = {
		tag = NOR
		owns = 20
		is_year = 1640
		NOT = {
			is_year = 1750
		}
	}

	mean_time_to_happen = {
		months = 360
	}

	option = {
		name = hee_NOR.1.a
		20 = {
			change_trade_goods = copper
		}
	}
	option = {
		name = hee_NOR.1.b
		add_treasury = 50
	}
}

#Saint Olav’s Shrine for Overlord

country_event = {
	id = hee_NOR.2
	title = hee_NOR.2.t
	desc = hee_NOR.2.d
	picture = IMPORTANT_STATUE_eventPicture

	fire_only_once = yes

	trigger = {
		NOR = {
			is_subject_of = ROOT
		}
		OR = {
			religion = protestant
			religion = reformed
		}
	}

	mean_time_to_happen = {
		months = 120
		modifier = {
			factor = 0.5
			OR = {
				NOR = {
					religion = protestant
				}
				NOR = {
					religion = reformed 
				}
			}
		}
	}
	option = {
		name = hee_NOR.2.a
		ai_chance = {
			factor = 90
		}
		NOR = {
			random_owned_province = {
				limit = {
					NOT = { religion = ROOT }
				}
				change_religion = ROOT
			}
			country_event = {
				id = hee_NOR.3 days = 10 random = 5
			}
			add_treasury = 25
		}
	}
	option = {
		name = hee_NOR.2.b
		ai_chance = {
			factor = 10
		}
		add_prestige = -5
	}
}
