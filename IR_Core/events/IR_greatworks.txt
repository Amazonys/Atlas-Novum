# Pride and Prejudice By Jane Austen
country_event = {
	id = great_work.1
	title = great_work.1.t
	desc = great_work.1.d

	fire_only_once = yes

	trigger = {
	    OR = {
	    	tag = ENG
			tag = GBR
			tag = UNK
		}
		is_year = 1813
		government = monarchy
		has_regency = yes
	}

		months = 3
	}

	option = {
		name = great_work.1.a

			add_country_modifier = {

			}
		}
	  }
   }
