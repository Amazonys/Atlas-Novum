# Alvand Mizra or 
country_event = {
	id = IR_flavor_QAR.1
	title = "IR_flavor_QAR.1.t"
	desc = "IR_flavor_QAR.1.d"
	picture = COURT_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	major = yes
	
	trigger = {
		always = no
		tag = QAR
		is_year = 1445
		NOT = { has_country_flag = IR_flavor_QAR.1_happened } 
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = { set_country_flag = IR_flavor_QAR.1_happened }
	
	
	option = {		# Release Iraq under Alvand Mizra
		name = "IR_flavor_QAR.1.a" 
		ai_chance = { factor = 85 }
		hidden_effect = {
			#release = IRQ
			every_owned_province = {
				limit = { qar_iraq_provinces = yes }
				cede_province = IRQ
				remove_core = ROOT
				if = {
					limit = { NOT = { has_core = IRQ } }
					add_core = IRQ
					
				}
			}
			
		}
		create_subject = {
			subject = IRQ
			subject_type = fief
		}
		IRQ = {
			define_ruler = {
   				name = "Alvand Mizra"
   				dynasty = "Qara Qoyunlu"
    			adm = 3
    			dip = 1
   				mil = 4
    			age = 32
    			#claim = 95
    			culture = turcoman
			}
			set_capital = 411
			hidden_effect = {
				change_primary_culture = turcoman
			}	
		}
	}
	option = {	# Fahud
		name = "IR_flavor_QAR.1.b" 
		ai_chance = { factor = 15 }
		every_owned_province = {
			limit = { qar_iraq_provinces = yes }
			add_unrest = 5
		}
	}
}
