# New Shogun - join or not?
country_event = {
	id = japan.2
	title = japan.2.name
	desc = japan.2.desc
	picture = COURT_eventPicture
	
	is_triggered_only = yes
	
	option = { #be a daimyo
		name = japan.2.A

		event_target:new_shogunate = {
			vassalize = root
		}

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				num_of_cities = 10
			}
		}
	}
	option = { #don't be a daimyo
		name = japan.2.B
		
		event_target:new_shogunate = {
			add_opinion = {
				who = root
				modifier = opinion_defy_new_shogun
			}
			country_event = { id = japan.3 days = 5 }
		}

		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				NOT = { num_of_cities = 10 }
			}
		}
	}
}
