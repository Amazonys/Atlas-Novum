namespace = IR_strait

#Palk Strait - formed 1480 historically
province_event = { 
	id = IR_strait.1
	title = "IR_strait.1.t"
	desc = "IR_strait.1.d"
	picture = BATTLE_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		#always = no
		province_id = 6929
		NOT = { has_global_flag = palk_strait_cyclone_happened }
		NOT = { has_great_project = palk_strait }
        NOT = { has_construction = canal }
		owner = { is_at_war = no }
		controlled_by = owner
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.5
			owner = { NOT = { owns = 536 } }
		}
	}
	
	immediate = {
		hidden_effect = { set_global_flag = palk_strait_cyclone_happened }
	}

	option = {
		name = "IR_strait.1.a" 
		ai_chance = { factor = 90 }
		custom_tooltip = IR_strait.1_tooltip
		hidden_effect = { add_great_project = palk_strait }
		add_permanent_province_modifier = {
			name = palk_strait
			duration = -1
		}
	} 
	option = {
		name = "IR_strait.1.b" 
		ai_chance = { factor = 10 }
		add_treasury = -100
		add_manpower = -5
		custom_tooltip = IR_strait.2_tooltip
	} 
}

	title = "IR_strait.2.t"
	desc = "IR_strait.2.d"
	picture = CITY_DEVELOPMENT_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		#always = no
		province_id = 6929
		has_global_flag = palk_strait_cyclone_happened
		NOT = { owner = { has_country_flag = palk_strait_construct_happened } }
		NOT = { has_great_project = palk_strait }
        NOT = { has_construction = canal }
		is_year = 1600
		owner = { is_at_war = no }
		controlled_by = owner
	}
	
	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.5
			owner = { NOT = { owns = 536 } }
		}
	}
	
	immediate = { owner = { set_country_flag = palk_strait_construct_happened } }

	option = {
		name = "IR_strait.2.a" 
		ai_chance = { factor = 90 }
		custom_tooltip = IR_strait.3_tooltip
		add_treasury = -1000
		add_manpower = -10
		add_permanent_province_modifier = {
			name = palk_strait
			duration = -1
		}
		hidden_effect = { add_great_project = palk_strait }
	} 
	option = {
		name = "IR_strait.2.b" 
		ai_chance = { factor = 10 }
		add_prestige = 5
	} 
} 



#Uzboy River dries up
country_event = { 
	id = IR_strait.3
	title = "IR_strait.3.t"
	desc = "IR_strait.3.d"
	picture = BATTLE_eventPicture
	
	major = yes
	fire_only_once = yes
	
	trigger = {
		any_owned_province = {
			area = uzboy_area
		}
		NOT = { has_global_flag = uzboy_dry_happened }
		NOT = { 9433 = { has_great_project = uzboy_dry } }
	}
	
	mean_time_to_happen = {
		months = 1200
	}
	
	immediate = {
		hidden_effect = { set_global_flag = uzboy_dry_happened }
	}

	option = {
		name = "IR_strait.3.a" 
		ai_chance = { factor = 90 }
		#custom_tooltip = IR_strait.1_tooltip
		2349 = { remove_province_modifier = uzboy_estuary_modifier }
		hidden_effect = { 9433 = { add_great_project = uzboy_dry } }
	} 
	option = {
		name = "IR_strait.3.b" 
		ai_chance = { factor = 10 }
		add_years_of_income = -5
		add_manpower = -10
		#custom_tooltip = IR_strait.2_tooltip
	} 
}
