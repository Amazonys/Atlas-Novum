			create_subject = {
				subject_type = fief

			}
		}

			limit = { CAT = { exists = yes is_subject_of = ROOT } }

				if = {
					limit = { CAT = { NOT = { government = republic } } }

		}

	}

		name = "IR_flavor_FRA.26.c" 
		if = {

				limit = { OR = { area = catalonia_area area = urgell_area } }
