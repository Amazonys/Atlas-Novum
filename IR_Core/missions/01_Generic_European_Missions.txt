		NOT = {
			is_emperor = yes
		}
	}
	has_country_shield = no
	
	gen_hre_electors = {
		icon = mission_alliances
		required_missions = { }
		position = 3
		
		trigger = {
			custom_trigger_tooltip = {
				tooltip = gen_hre_electors.tooltip
				calc_true_if = {

		effect = {
			add_prestige = 10

			}
